001package org.gbif.checklistbank.index;
002
003
004import java.util.List;
005import java.util.concurrent.BlockingQueue;
006import java.util.concurrent.Callable;
007import java.util.concurrent.CompletionService;
008import java.util.concurrent.ExecutionException;
009import java.util.concurrent.ExecutorCompletionService;
010import java.util.concurrent.Future;
011import java.util.concurrent.RejectedExecutionHandler;
012import java.util.concurrent.SynchronousQueue;
013import java.util.concurrent.ThreadPoolExecutor;
014import java.util.concurrent.TimeUnit;
015
016import org.slf4j.Logger;
017import org.slf4j.LoggerFactory;
018
019/**
020 * This class was implemented to be executed only one time, and probably be executed by a job scheduler (CRON for example).
021 * The behavior of the pool of threads can be customized using the properties file, using this file
022 * * is possible define the values for: poolSize, maxPoolSize and keepAliveTime; and additionally for the
023 * e-mail configuration (an email will be sent at the end of the process):
024 *  smtp.server,email.to,email.from,email.subject,email.body,email.cc
025 * The thread pool is implemented using an {@link ExecutorCompletionService} and a {@link ThreadPoolExecutor} because
026 * some results can be return for each job: errors during the process or any other information.
027 */
028public abstract class ThreadPoolRunner<T> {
029
030
031  private static class DefaultRejectedExecutionHandler implements RejectedExecutionHandler {
032
033    @Override
034    public void rejectedExecution(Runnable job, ThreadPoolExecutor executor) {
035      ThreadPoolRunner.LOG.debug("Job has been rejected by the executor, putting on queue");
036      try {
037        executor.getQueue().put(job);
038      } catch (InterruptedException e1) {
039        LOG.error("Work discarded, thread was interrupted while waiting for space to schedule: {}", job);
040      }
041    }
042  }
043
044
045  /**
046   * the number of threads to keep in the pool, even if they are idle.
047   */
048  protected int poolSize = 0;
049
050  /**
051   * Executes each submitted task (agent synchronization) using one of possibly several pooled threads
052   */
053  private ThreadPoolExecutor threadPool = null;
054
055  /**
056   * Decouples the production of new asynchronous tasks from the consumption of the results of completed tasks
057   */
058  private CompletionService<T> completionService;
059
060  /**
061   * Queue of elements that will be executed by the {@link ThreadPoolExecutor}
062   */
063  private final BlockingQueue<Runnable> queue = new SynchronousQueue<Runnable>();
064
065  /**
066   * Logger for the {@link ThreadPoolRunner} class
067   */
068  private static final Logger LOG = LoggerFactory.getLogger(ThreadPoolRunner.class);
069
070  /**
071   * @param poolSize the number of threads to keep in the pool, even if they are idle.
072   */
073  public ThreadPoolRunner(int poolSize) {
074    this.poolSize = poolSize;
075  }
076
077  protected abstract Callable<T> newJob();
078
079  /**
080   * Initialize the {@link ThreadPoolExecutor} and {@link ExecutorCompletionService}
081   */
082  private void initThreadPool() {
083    threadPool = new ThreadPoolExecutor(
084      poolSize, poolSize, 0, TimeUnit.SECONDS, queue, new DefaultRejectedExecutionHandler()
085    );
086    completionService = new ExecutorCompletionService<T>(threadPool);
087  }
088
089
090  /**
091   * This method runs the synchronization for all the agent of the same type,
092   * creates a thread that will synchronized # of agents (defined by the agentsPerThread field)
093   */
094  public int run() {
095    // localTasksCount contains the number of threads created, this count will be used for shutting down the service
096
097    initThreadPool();
098
099    int localTasksCount = 0;
100    try {
101      // gets the agents with service type: this.serviceTypeCode
102      Callable<T> job = newJob();
103      while (job != null) {
104        try {
105          LOG.debug("Submitting job {}, tasks count={}", job, localTasksCount);
106          completionService.submit(job);
107          localTasksCount += 1;
108        } catch (Exception e) {
109          LOG.error("Error when submiting job: {}", localTasksCount);
110        }
111        job = newJob();
112      }
113      LOG.info("{} Jobs submitted succesfully!", localTasksCount);
114      shutdownService(localTasksCount);
115    } catch (Exception e) {
116      LOG.error("Error when submiting jobs: {}", e);
117    }
118
119    return localTasksCount;
120  }
121
122  /**
123   * Performs the shutdown activities: waits each thread to complete the process and collects the results.
124   */
125  protected void shutdownService(int tasksCount) {
126    try {
127      LOG.info("Initiating shutting down of the service");
128      while (tasksCount > 0) {
129        // waits for each task to return the results
130        Future<T> result = completionService.take();
131        tasksCount--;
132        try {
133          T taskResponse = result.get();
134          LOG.info("Job completed, {} still running", tasksCount);
135          taskResponseHook(taskResponse);
136
137        } catch (ExecutionException e) {
138          LOG.error("Error waiting a job to return results. {} still running jobs", tasksCount, e);
139        }
140        LOG.debug("Waiting for {} jobs to shutdown.", tasksCount);
141      }
142      List<Runnable> failedTasks = threadPool.shutdownNow();
143      LOG.debug("{} Tasks failed or forcibly removed from the thread pool", failedTasks.size());
144    } catch (Exception ignored) {
145      LOG.error("Thread pool prematurely terminated", ignored);
146    }
147    LOG.info("Shutdown complete");
148  }
149
150  /**
151   * Override this method if the main runner needs to consume the individual job results.
152   * @param taskResponse the finished job result
153   */
154  protected void taskResponseHook (T taskResponse) {
155
156  }
157}
