041/**
042 * This class contains the annotations required by {@link org.gbif.api.service.common.SearchService} and the Solr
043 * result/object mapping.
044 * The intention of having a separate class containing mostly annotations is to avoid the usage of specific Solr
045 * annotations in the model objects that are part of the API.
046 * Scoring used in this class:
047 * canonical_name:Abies^100
048 * scientific_name:Abies^10
049 * scientific_name:*Abies*^0.2
050 * vernacular_name:Abies^25
051 * vernacular_name:Abies^6
052 * vernacular_name:*Abies*^0.2
053 * kingdom:Abies
054 * phylum:Abies
055 * class:Abies
056 * order:Abies
057 * family:Abies^2
058 * genus:Abies^2
059 * subgenus:Abies^3
060 * species:Abies^5
061 * description:Abies^0.3
062 */
063@SearchMapping(
064  facets = {
065    @FacetField(name = "DATASET_KEY", field = "dataset_key"),
066    @FacetField(name = "HIGHERTAXON_KEY", field = "higher_taxon_nub_key"),
067    @FacetField(name = "STATUS", field = "taxonomic_status_key"),
068    @FacetField(name = "RANK", field = "rank_key"),
069    @FacetField(name = "THREAT", field = "threat_status_key"),
070    @FacetField(name = "EXTINCT", field = "extinct"),
071    @FacetField(name = "NOMENCLATURAL_STATUS", field = "nomenclatural_status_key"),
072    @FacetField(name = "NAME_TYPE", field = "name_type"),
073    @FacetField(name = "HABITAT", field = "marine")
074  },
075  fulltextFields = {
076    @FullTextSearchField(field = "canonical_name", exactMatchScore = 100.0d,
077      partialMatchScore = 1d, partialMatching = WildcardPadding.RIGHT),
078    @FullTextSearchField(field = "vernacular_name", highlightField = "vernacular_name_lang", exactMatchScore = 10d,
079      partialMatchScore = 0.75d, partialMatching = WildcardPadding.RIGHT),
080    @FullTextSearchField(field = "scientific_name", exactMatchScore = 5d,
081      partialMatchScore = 0.7d, partialMatching = WildcardPadding.RIGHT),
082    @FullTextSearchField(field = "description", exactMatchScore = 0.2d,
083      partialMatching = WildcardPadding.NONE),
084    @FullTextSearchField(field = "species", exactMatchScore = 5d, partialMatching = WildcardPadding.NONE),
085    @FullTextSearchField(field = "subgenus", exactMatchScore = 3d, partialMatching = WildcardPadding.NONE),
086    @FullTextSearchField(field = "genus", exactMatchScore = 2d, partialMatching = WildcardPadding.NONE),
087    @FullTextSearchField(field = "family", exactMatchScore = 2d, partialMatching = WildcardPadding.NONE),
088    @FullTextSearchField(field = "order", partialMatching = WildcardPadding.NONE),
089    @FullTextSearchField(field = "class", partialMatching = WildcardPadding.NONE),
090    @FullTextSearchField(field = "phylum", partialMatching = WildcardPadding.NONE),
091    @FullTextSearchField(field = "kingdom", partialMatching = WildcardPadding.NONE),
092  })
093@SuggestMapping(field = "canonical_name_as", phraseQueryField = "canonical_name_as_phrase")

096  private static Function<VernacularName, String> toNameLang = new Function<VernacularName, String>() {
097
098    @Override
099    public String apply(VernacularName obj) {
100      String lang = obj.getLanguage() == null ? "" : obj.getLanguage().getIso2LetterCode();
101      return Strings.nullToEmpty(obj.getVernacularName()) + lang;
102    }
103  };
104
105  private static Ordering<VernacularName> byNameAndLanguage = Ordering.natural().onResultOf(toNameLang);

383   * Populates the vernacular names from a solr response.
384   * @param vernacularNamesLanguages the vernacularNamesLanguages to set
385   */
386  public void setVernacularNamesSerialized(List<String> vernacularNamesLanguages) {
387    // deserialize them into vernacular names
388    List<VernacularName> vnames = Lists.newArrayListWithCapacity(vernacularNamesLanguages.size());
389    for (String v : vernacularNamesLanguages) {
390      vnames.add(NameUsageDocConverter.deserializeVernacularName(v));
391    }
392    super.setVernacularNames(vnames);
