001package org.gbif.checklistbank.index.model;
002
003import org.gbif.api.model.checklistbank.VernacularName;
004import org.gbif.checklistbank.index.NameUsageDocConverter;
005import org.gbif.common.search.model.HighlightableList;
006
007import java.util.List;
008
009import com.google.common.collect.Lists;
010
011/**
012 * A facade to a vernacular name stored in a serialized format in solr that allows to support solr highlighting on the
013 * vernacularName field of the complex VernacularName class.
014 */
015public class HighlightableVernacularName implements HighlightableList {
016
017  private final List<VernacularName> names;
018
019  public HighlightableVernacularName(List<VernacularName> names) {
020    this.names = names;
021  }
022
023  @Override
024  public List<String> getValues() {
025    List<String> values = Lists.newArrayList();
026    for (VernacularName d : names) {
027      values.add(NameUsageDocConverter.serializeVernacularName(d));
028    }
029    return values;
030  }
031
032  @Override
033  public void replaceValue(int index, String newValue) {
034    // watch out, we get a marked up serialized string here!
035    VernacularName v = NameUsageDocConverter.deserializeVernacularName(newValue);
036    names.get(index).setVernacularName(v.getVernacularName());
037  }
038}
