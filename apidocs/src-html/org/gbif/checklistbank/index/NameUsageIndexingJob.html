001package org.gbif.checklistbank.index;
002
003import org.gbif.api.model.checklistbank.Description;
004import org.gbif.api.model.checklistbank.Distribution;
005import org.gbif.api.model.checklistbank.NameUsage;
006import org.gbif.api.model.checklistbank.NameUsageComponent;
007import org.gbif.api.model.checklistbank.SpeciesProfile;
008import org.gbif.api.model.checklistbank.VernacularName;
009import org.gbif.checklistbank.service.UsageService;
010import org.gbif.checklistbank.service.mybatis.DescriptionServiceMyBatis;
011import org.gbif.checklistbank.service.mybatis.DistributionServiceMyBatis;
012import org.gbif.checklistbank.service.mybatis.SpeciesProfileServiceMyBatis;
013import org.gbif.checklistbank.service.mybatis.VernacularNameServiceMyBatis;
014
015import java.util.List;
016import java.util.Map;
017import java.util.concurrent.Callable;
018
019import com.google.common.collect.Lists;
020import com.google.common.collect.Maps;
021import org.apache.commons.lang3.time.StopWatch;
022import org.apache.solr.client.solrj.SolrServer;
023import org.apache.solr.common.SolrInputDocument;
024import org.slf4j.Logger;
025import org.slf4j.LoggerFactory;
026
027/**
028 * Executable job that creates a list of {@link SolrInputDocument} using a list of {@link NameUsage} objects.
029 */
030public class NameUsageIndexingJob implements Callable<Integer> {
031
032  private final Logger log = LoggerFactory.getLogger(getClass());
033
034  /**
035   * SolrServer instance.
036   */
037  private final SolrServer indexWriter;
038
039  /**
040   * Minimum usage key, inclusive, to process.
041   */
042  private final int startKey;
043
044  /**
045   * Maximum usage key, inclusive, to process.
046   */
047  private final int endKey;
048
049  /**
050   * Service layer.
051   */
052  private final UsageService nameUsageService;
053  private final VernacularNameServiceMyBatis vernacularNameService;
054  private final DescriptionServiceMyBatis descriptionService;
055  private final DistributionServiceMyBatis distributionService;
056  private final SpeciesProfileServiceMyBatis speciesProfileService;
057
058  private StopWatch stopWatch = new StopWatch();
059
060  /**
061   * {@link NameUsage}/{@link SolrInputDocument} converter.
062   */
063  private final NameUsageDocConverter solrDocumentConverter;
064
065  /**
066   * Default constructor.
067   */
068  public NameUsageIndexingJob(final SolrServer solr, final UsageService nameUsageService, final int startKey,
069    final int endKey, final NameUsageDocConverter solrDocumentConverter,
070    final VernacularNameServiceMyBatis vernacularNameService, final DescriptionServiceMyBatis descriptionService,
071    final DistributionServiceMyBatis distributionService, final SpeciesProfileServiceMyBatis speciesProfileService) {
072    this.nameUsageService = nameUsageService;
073    this.vernacularNameService = vernacularNameService;
074    this.descriptionService = descriptionService;
075    this.distributionService = distributionService;
076    this.speciesProfileService = speciesProfileService;
077    this.startKey = startKey;
078    this.endKey = endKey;
079    this.solrDocumentConverter = solrDocumentConverter;
080    this.indexWriter = solr;
081  }
082
083  /**
084   * Iterates over the assigned {@link NameUsage} objects to create the corresponding {@link SolrInputDocument}
085   * objects.
086   *
087   * @return the total number of documents added by this Thread.
088   */
089  @Override
090  public Integer call() throws Exception {
091    // Timing information initialization
092    stopWatch.start();
093    log.info("Adding usages from id {} to {}", startKey, endKey);
094    int docCount = 0;
095    // Get all usages
096    List<NameUsage> usages = nameUsageService.listRange(startKey, endKey);
097    // get all component maps into memory first
098    List<VernacularName> vernacularNames = vernacularNameService.listRange(startKey, endKey);
099    Map<Integer, List<VernacularName>> vernacularNameMap = buildComponentMap(vernacularNames);
100
101    List<Description> descriptions = descriptionService.listRange(startKey, endKey);
102    Map<Integer, List<Description>> descriptionMap = buildComponentMap(descriptions);
103
104    List<Distribution> distributions = distributionService.listRange(startKey, endKey);
105    Map<Integer, List<Distribution>> distributionMap = buildComponentMap(distributions);
106
107    List<SpeciesProfile> profiles = speciesProfileService.listRange(startKey, endKey);
108    Map<Integer, List<SpeciesProfile>> speciesProfileMap = buildComponentMap(profiles);
109
110    // now we're ready to build the solr indices quicky!
111    for (NameUsage usage : usages) {
112      try {
113        indexWriter.add(solrDocumentConverter.toObject(usage, vernacularNameMap.get(usage.getKey()),
114        descriptionMap.get(usage.getKey()),distributionMap.get(usage.getKey()), speciesProfileMap.get(usage.getKey())));
115
116      } catch (Exception e) {
117        log.error("Error indexing document for usage {}", usage.getKey(), e);
118      }
119      docCount++;
120      NameUsageIndexer.counter.incrementAndGet();
121    }
122    // job finished notice
123    stopWatch.stop();
124    log.info("Finished indexing of usages in range {}-{}. Total time: {}",
125      new Object[] {startKey, endKey, stopWatch.toString()});
126
127    return docCount;
128  }
129
130  private <T extends NameUsageComponent> Map<Integer, List<T>> buildComponentMap(List<T> components) {
131    Map<Integer, List<T>> map = Maps.newHashMap();
132    for (T c : components) {
133      if (!map.containsKey(c.getUsageKey())) {
134        map.put(c.getUsageKey(), Lists.<T>newArrayList());
135      }
136      map.get(c.getUsageKey()).add(c);
137    }
138    return map;
139  }
140}
