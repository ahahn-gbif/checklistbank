001package org.gbif.checklistbank.index.guice;
002
003import com.google.inject.Inject;
004import com.google.inject.Provides;
005import com.google.inject.Scopes;
006import com.google.inject.Singleton;
007import org.apache.solr.client.solrj.SolrServer;
008import org.gbif.checklistbank.index.NameUsageDocConverter;
009import org.gbif.checklistbank.index.NameUsageIndexer;
010import org.gbif.common.search.inject.SolrModule;
011import org.gbif.service.guice.PrivateServiceModule;
012
013import java.util.Properties;
014
015
016/**
017 * Guice module that bind the indexing classes.
018 * This module depends on the registry client and checklist bank mybatis modules to be installed too
019 * which is done in the {@link IndexingModule}.
020 */
021public class IndexingModulePrivate extends PrivateServiceModule {
022
023  private static final String PREFIX = "checklistbank.indexer.";
024
025  public IndexingModulePrivate(Properties properties) {
026    super(PREFIX, properties);
027  }
028
029  @Override
030  protected void configureService() {
031    // Object <-> Document converter binding
032    bind(NameUsageDocConverter.class).in(Scopes.SINGLETON);
033
034    // Main indexer class
035    bind(NameUsageIndexer.class).in(Scopes.SINGLETON);
036
037    install(new SolrModule());
038
039    expose(NameUsageIndexer.class);
040    expose(SolrServer.class);
041    expose(EmbeddedSolrReference.class);
042  }
043
044  @Inject
045  @Singleton
046  @Provides
047  public EmbeddedSolrReference provideSolrReference(SolrServer solr) {
048    return new EmbeddedSolrReference(solr);
049  }
050}
