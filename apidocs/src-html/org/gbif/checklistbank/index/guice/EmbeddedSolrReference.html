001package org.gbif.checklistbank.index.guice;
002
003import org.apache.solr.client.solrj.SolrServer;
004import org.apache.solr.client.solrj.embedded.EmbeddedSolrServer;
005
006/**
007 * Simple wrapper class that can be used as a singleton to reference a exchangeable solr server.
008 * The indexing tests need this so we can shutdown solr and recreate a new index without breaking existing
009 * guice bindings.
010 */
011public class EmbeddedSolrReference {
012  private EmbeddedSolrServer solr;
013
014  public EmbeddedSolrReference(SolrServer solr) {
015    this.solr = (EmbeddedSolrServer) solr;
016  }
017
018  public EmbeddedSolrServer getSolr() {
019    return solr;
020  }
021
022  public void setSolr(EmbeddedSolrServer solr) {
023    this.solr = solr;
024  }
025}
