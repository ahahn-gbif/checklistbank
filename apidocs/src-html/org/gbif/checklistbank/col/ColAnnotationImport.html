042/**
043 * Main class to run a full import of CoL GSD annotations for the GBIF Backbone, populating the col_annotation table
044 * in Checklist Bank.
045 * @see <a href="http://dev.gbif.org/issues/browse/CLB-248">CLB-248</a>
046 */
047public class ColAnnotationImport {
048  private static final Logger LOG = LoggerFactory.getLogger(ColAnnotationImport.class);
049  private Properties props;
050  private ColAnnotationService annotationService;
051  @VisibleForTesting
052  protected Pattern ANNOTATION_SPLITTER = Pattern.compile("^\\s*(.*)\\s*\\|\\s*(.*)\\s*\\|\\s*(.*)\\s*$");
053
054  public ColAnnotationImport(String propsFile) throws IOException {
055    props = PropertiesUtil.readFromFile(propsFile);
056    // init guice
057    ChecklistBankServiceMyBatisModule myBatisModule = new ChecklistBankServiceMyBatisModule(props);
058    Injector inj = Guice.createInjector(myBatisModule);
059    annotationService = inj.getInstance(ColAnnotationService.class);
060  }
061
062  public void importAnnotations() throws IOException, URISyntaxException {
063    Archive arch = downloadArchive();
064    int counter = 0;
065    int failures = 0;
066    int empty = 0;
067    for (StarRecord star : arch) {
068      Record rec = star.core();
069      counter++;
070      try {
071        int nubKey = counter;//TODO: use taxonIDs once available: Integer.parseInt(rec.value(DwcTerm.taxonID));
072        Matcher m = ANNOTATION_SPLITTER.matcher(rec.value(DwcTerm.taxonRemarks));
073        if (m.find()) {
074          ColAnnotation annotation = new ColAnnotation(nubKey, rec.value(DwcTerm.datasetName), rec.value(DwcTerm.scientificName),
075            isRejected(m.group(1)), m.group(3), m.group(2));
076          annotationService.insertAnnotation(annotation);
077
078        } else {
079          empty++;
080          LOG.warn("No annotation given for {}", rec.value(DwcTerm.scientificName));
081        }
082      } catch (NumberFormatException e) {
083        failures++;
084        LOG.warn("No valid taxonID given for {}", rec.value(DwcTerm.scientificName));
085      }
086    }
087    LOG.info("{} annotations processed, {} lack notes, {} failed to import", counter, empty, failures);
088  }
089
090  @VisibleForTesting
091  protected static boolean isRejected(String annotation) {
092    if (annotation.toLowerCase().startsWith("placed")) {
093      return false;
094    }
095    return true;
096  }
097
098  private Archive downloadArchive() throws IOException, URISyntaxException {
099    // download url
100    final String url = props.getProperty("col.annotation.url");
101    // local work dir
102    File workDir = FileUtils.createTempDir();
103    workDir.deleteOnExit();
104    // local zip file
105    File zip = new File(workDir, "annotation.zip");
106    // local decompressed dwca
107    File dwca = new File(workDir, "annotation");
108    // create folders
109    org.apache.commons.io.FileUtils.forceMkdir(dwca);
110
111    // use a 10 minutes timeout
112    HttpClient client = HttpUtil.newMultithreadedClient(600000, 10, 10);
113    // authentication
114    HttpContext authContext = new BasicHttpContext();
115    URI authUri = new URI(url);
116    AuthScope scope = new AuthScope(authUri.getHost(), AuthScope.ANY_PORT, AuthScope.ANY_REALM);
117
118    CredentialsProvider credsProvider = new BasicCredentialsProvider();
119    credsProvider.setCredentials(scope, new UsernamePasswordCredentials(props.getProperty("col.annotation.user"), props.getProperty("col.annotation.password")));
120    authContext.setAttribute(ClientContext.CREDS_PROVIDER, credsProvider);
121
122    HttpGet get = new HttpGet(url);
123    HttpResponse response = client.execute(get, authContext);
124
125    if (response.getStatusLine().getStatusCode() != 200) {
126      LOG.error("{} error downloading annotations from {}: {}", response.getStatusLine(), url,
127        response.getStatusLine().getReasonPhrase());
128      System.exit(1);
129    }
130
131    HttpEntity entity = response.getEntity();
132    if (entity != null) {
133      // copy stream to local file
134      OutputStream fos = new FileOutputStream(zip, false);
135      try {
136        entity.writeTo(fos);
137      } finally {
138        fos.close();
139      }
140    }
141
142    LOG.info("Successfully downloaded {} to {}", url, zip.getAbsolutePath());
143
144    // open archive
145    Archive arch = ArchiveFactory.openArchive(zip, dwca);
146    return arch;
147  }
148
149  public static void main (String[] args) throws Exception {
150    if (args.length != 1) {
151      System.err.println("Pass the full path to the config properties file as first argument please");
152    }
153    ColAnnotationImport imp = new ColAnnotationImport(args[0]);
154    imp.importAnnotations();
155  }
