025import java.util.List;
026import java.util.Map;
027
028import com.google.common.collect.ArrayListMultimap;
029import com.google.common.collect.ImmutableMap;
030import com.google.common.collect.Multimap;
031import com.google.inject.Inject;
032import org.apache.solr.client.solrj.SolrQuery;
033import org.apache.solr.client.solrj.SolrServer;
034
035import static org.gbif.common.search.util.SearchConstants.NOT_OP;
036
037
038/**
039 * Checklist Bank Search interface implementation using Solr/lucene.
040 * This class uses a remote to interact with CLB index.
041 */
042public class NameUsageSearchServiceImpl
043  extends SolrSearchSuggestService<NameUsageSearchResult, NameUsageSearchParameter, NameUsageSolrSearchResult,
044  NameUsageSearchRequest, NameUsageSuggestRequest> implements NameUsageSearchService {
045
046  // Default primary sort order
047  private static final Map<String, SolrQuery.ORDER> PRIMARY_SORT_ORDER =
048    new ImmutableMap.Builder<String, SolrQuery.ORDER>()
049      .put("score", SolrQuery.ORDER.desc)
050      .put("num_descendants", SolrQuery.ORDER.desc)
051      .put("scientific_name", SolrQuery.ORDER.asc).build();
052
053  // Default sort order for suggest results
054  private static final Map<String, SolrQuery.ORDER> DEFAULT_SUGGEST_ORDER =
055    new ImmutableMap.Builder<String, SolrQuery.ORDER>()
056      .put("score", SolrQuery.ORDER.desc)
057      .put("rank_key", SolrQuery.ORDER.asc).build();
058
059
060  private static final Multimap<NameUsageSearchParameter, String> SUGGEST_DEFAULT_FILTERS = ArrayListMultimap.create();
061
062  protected static final int DEFAULT_LIMIT = 10;
063
064  static {
065    // Suggest default filters for suggest service: names of type INFORMAL and BLACKLISTED are excluded by default.
066    SUGGEST_DEFAULT_FILTERS.put(NameUsageSearchParameter.NAME_TYPE, NOT_OP + NameType.INFORMAL.name());
067    SUGGEST_DEFAULT_FILTERS.put(NameUsageSearchParameter.NAME_TYPE, NOT_OP + NameType.BLACKLISTED.name());
068  }
069
070  /**
071   * Default constructor.
072   */
073  @Inject
074  public NameUsageSearchServiceImpl(SolrServer server) {
075    // Type parameter bounded to NameUsageSearch
076    super(server, null, NameUsageSearchResult.class, NameUsageSolrSearchResult.class,
077      NameUsageSearchParameter.class, PRIMARY_SORT_ORDER, DEFAULT_SUGGEST_ORDER, false);
078  }
079
080  @Override
081  public List<NameUsageSearchResult> suggest(NameUsageSuggestRequest suggestRequest) {
082    if (!suggestRequest.getParameters().containsKey(NameUsageSearchParameter.DATASET_KEY)) {
083      // if the datasetKey parameters is not in the list, the GBIF nub is used by default
084      suggestRequest.addParameter(NameUsageSearchParameter.DATASET_KEY, Constants.NUB_DATASET_KEY.toString());
085      for (NameUsageSearchParameter p : SUGGEST_DEFAULT_FILTERS.keys()) {
086        suggestRequest.addParameter(p, SUGGEST_DEFAULT_FILTERS.get(p));
087      }
088    }
089    return super.suggest(suggestRequest);
090  }
091
092}
