001package org.gbif.checklistbank.ws.nub;
002
003import org.gbif.api.model.checklistbank.NameUsageMatch;
004import org.gbif.api.model.common.LinneanClassification;
005import org.gbif.api.service.checklistbank.NameUsageMatchingService;
006import org.gbif.api.util.VocabularyUtils;
007import org.gbif.api.vocabulary.Rank;
008
009import javax.ws.rs.GET;
010import javax.ws.rs.Path;
011import javax.ws.rs.Produces;
012import javax.ws.rs.QueryParam;
013import javax.ws.rs.core.MediaType;
014
015import com.google.common.base.Strings;
016import com.google.inject.Inject;
017import com.google.inject.Singleton;
018import org.slf4j.Logger;
019import org.slf4j.LoggerFactory;
020
021@Path("/species")
022@Singleton
023public class NubResource {
024
025  private static final Logger LOG = LoggerFactory.getLogger(NubResource.class);
026
027  private final NameUsageMatchingService matchingService;
028
029  @Inject
030  public NubResource(NameUsageMatchingService matchingService) {
031    this.matchingService = matchingService;
032  }
033
034
035  @GET
036  @Produces(MediaType.APPLICATION_JSON)
037  @Path("match")
038  public NameUsageMatch search(@QueryParam("name") String scientificName, @QueryParam("rank") String rank,
039    @QueryParam("kingdom") String kingdom, @QueryParam("phylum") String phylum,
040    @QueryParam("class") String clazz, @QueryParam("order") String order, @QueryParam("family") String family,
041    @QueryParam("genus") String genus, @QueryParam("subgenus") String subgenus,
042    @QueryParam("strict") Boolean strict, @QueryParam("verbose") Boolean verbose) {
043
044    Rank r = null;
045    if (!Strings.isNullOrEmpty(rank)) {
046      try {
047        r = (Rank) VocabularyUtils.lookupEnum(rank, Rank.class);
048      } catch (Exception e) {
049        // we expect badly formed ranks and catch the exception
050        LOG.debug("Bad rank given: {}", rank);
051      }
052    }
053
054    LinneanClassification classification = new NameUsageMatch();
055    classification.setKingdom(kingdom);
056    classification.setPhylum(phylum);
057    classification.setClazz(clazz);
058    classification.setOrder(order);
059    classification.setFamily(family);
060    classification.setGenus(genus);
061    classification.setSubgenus(subgenus);
062
063    return  matchingService.match(scientificName, r, classification, bool(strict), bool(verbose));
064  }
065
066  private boolean bool(Boolean bool) {
067    return bool==null ? false : bool;
068  }
069
070}
