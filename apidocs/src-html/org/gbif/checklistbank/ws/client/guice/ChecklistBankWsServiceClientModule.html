045/**
046 * Guice module that includes implementations for all the ChecklistBank services apart from NameUsageSearchService.
047 * It required the ChecklistBankHttpClient annotated jersey client and should not be used on its own.
048 * Use the main #ChecklistBankWsClientModule(false,true) instead with its configurable constructor.
049 * The only things left for clients to do is:
050 * 1. Bind the Service class corresponding to the Client's class in configure()
051 * 2. Create a method that returns a WebResource object. The method must:
052 * a) Use the @Provides annotation, @singleton annotation, and the Client's specific binding annotation.
053 * b) Include the @ChecklistBankHttpClient binding annotation for the the Jersey Client param.
054 * c) Where possible, the WebResource returned should define its path. Our convention uses the named annotation
055 * "checklistbank.ws.url" (passed in as a parameter to the method) as the base URL.
056 * Example path for ChecklistWs would be: Names.named("checklistbank.ws.url") + "species"
057 */
058public class ChecklistBankWsServiceClientModule extends AbstractModule {
059
060  /**
061   * Use the main #ChecklistBankWsClientModule(false,true) instead.
062   */
063  protected ChecklistBankWsServiceClientModule() {
064  }
065
066  @Override
067  protected void configure() {
068    // usages
069    bind(NameUsageService.class).to(NameUsageWsClient.class).in(Scopes.SINGLETON);
070    // usage components
071    bind(DescriptionService.class).to(DescriptionWsClient.class).in(Scopes.SINGLETON);
072    bind(DistributionService.class).to(DistributionWsClient.class).in(Scopes.SINGLETON);
073    bind(ImageWsClient.class).in(Scopes.SINGLETON);
074    bind(ImageService.class).to(ImageWsClient.class).in(Scopes.SINGLETON);
075    bind(ReferenceService.class).to(ReferenceWsClient.class).in(Scopes.SINGLETON);
076    bind(SpeciesProfileService.class).to(SpeciesProfileWsClient.class).in(Scopes.SINGLETON);
077    bind(TypeSpecimenService.class).to(TypeSpecimenWsClient.class).in(Scopes.SINGLETON);
078    bind(VernacularNameService.class).to(VernacularNameWsClient.class).in(Scopes.SINGLETON);
079    // metrics
080    bind(DatasetMetricsService.class).to(DatasetMetricsWsClient.class).in(Scopes.SINGLETON);
081  }
082
083  /**
084   * @return the web resource to the name usage base url
085   */
086  @Provides
087  @Singleton
088  @ChecklistBankWs
089  private WebResource providesNameUsageWsWebResource(Client client, @Named("checklistbank.ws.url") String url) {
090    return client.resource(url);
091  }
092
093}
