001package org.gbif.checklistbank.ws.resources;
002
003import org.gbif.api.model.checklistbank.ParsedName;
004import org.gbif.checklistbank.ws.util.LineReader;
005import org.gbif.nameparser.NameParser;
006import org.gbif.nameparser.UnparsableException;
007import org.gbif.ws.util.ExtraMediaTypes;
008
009import java.io.InputStream;
010import java.io.UnsupportedEncodingException;
011import java.nio.charset.Charset;
012import java.util.Iterator;
013import java.util.List;
014import javax.ws.rs.Consumes;
015import javax.ws.rs.GET;
016import javax.ws.rs.POST;
017import javax.ws.rs.Path;
018import javax.ws.rs.Produces;
019import javax.ws.rs.QueryParam;
020import javax.ws.rs.core.MediaType;
021
022import com.google.common.base.Splitter;
023import com.google.common.base.Strings;
024import com.google.common.collect.Lists;
025import com.google.inject.Inject;
026import com.sun.jersey.multipart.FormDataParam;
027import org.slf4j.Logger;
028import org.slf4j.LoggerFactory;
029
030/**
031 * The scientific name parser exposed in the API.
032 */
033@Path("/parser/name")
034@Produces({MediaType.APPLICATION_JSON, ExtraMediaTypes.APPLICATION_JAVASCRIPT})
035public class NameParserResource {
036
037  private static final Logger LOG = LoggerFactory.getLogger(NameParserResource.class);
038  private static final Splitter NEW_LINE_SPLITTER = Splitter.on('\n').omitEmptyStrings().trimResults();
039  private final NameParser parser;
040
041  @Inject
042  public NameParserResource(NameParser parser) {
043    this.parser = parser;
044  }
045
046  /**
047   * Parsing names as GET query parameters.
048   */
049  @GET
050  public List<ParsedName> parseGet(@QueryParam("name") List<String> names) {
051    return parse(names.iterator());
052  }
053
054  /**
055   * Parsing names as a json array.
056   */
057  @POST
058  @Consumes(MediaType.APPLICATION_JSON)
059  public List<ParsedName> parseJson(List<String> names) {
060    return parse(names.iterator());
061  }
062
063  /**
064   * Parsing names by uploading a plain UTF-8 text file using one line per scientific name.
065   * <pre>
066   * curl -F names=@scientific_names.txt http://apidev.gbif.org/parser/name
067   * </pre>
068   */
069  @POST
070  @Consumes(MediaType.MULTIPART_FORM_DATA)
071  public List<ParsedName> parseFile(@FormDataParam("names") InputStream namesFile) throws UnsupportedEncodingException {
072    if (namesFile == null) {
073      LOG.debug("No names file uploaded");
074      return Lists.newArrayList();
075    }
076    LineReader iter = new LineReader(namesFile, Charset.forName("UTF8"));
077    return parse(iter.iterator());
078  }
079
080
081  /**
082   * Parsing names by posting plain text content using one line per scientific name.
083   * Make sure to preserve new lines (\n) in the posted data, for example use --data-binary with curl:
084   * <pre>
085   * curl POST -H "Content-Type:text/plain" --data-binary @scientific_names.txt http://apidev.gbif.org/parser/name
086   * </pre>
087   */
088  @POST
089  @Consumes(MediaType.TEXT_PLAIN)
090  public List<ParsedName> parsePlainText(String names) {
091    return parse(NEW_LINE_SPLITTER.split(Strings.nullToEmpty(names)).iterator());
092  }
093
094  private List<ParsedName> parse(Iterator<String> iter) {
095    int counter = 0;
096    int failed = 0;
097    List<ParsedName> pnames = Lists.newArrayList();
098
099    while (iter.hasNext()) {
100      ParsedName pn;
101      final String name = iter.next();
102      try {
103        pn = parser.parse(name);
104      } catch (UnparsableException e) {
105        pn = new ParsedName();
106        pn.setType(e.type);
107        failed++;
108      }
109      pn.setScientificName(name);
110      pnames.add(pn);
111      counter++;
112    }
113
114    LOG.debug("Parsed {} names out of which {} are unparsable", counter, failed);
115    return pnames;
116  }
117}
