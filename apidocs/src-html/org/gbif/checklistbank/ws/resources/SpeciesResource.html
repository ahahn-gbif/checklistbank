079  /**
080   * This retrieves a list of all NameUsage from ChecklistBank.
081   * It is reachable at {@code /species}.
082   *
083   * @param locale      identifier for a region
084   * @param datasetKeys the optional checklist keys to limit paging to
085   * @param page        the limit, offset paging information
086   *
087   * @return requested list of NameUsage or an empty list if none could be found
088   */
089  @GET
090  public PagingResponse<NameUsage> list(@Context Locale locale, @QueryParam(DATASET_KEY) Set<UUID> datasetKeys,
091    @QueryParam("sourceId") String sourceId, @QueryParam("name") String canonicalName, @Context Pageable page) {
092    LOG.debug("Request all usages: [pageSize({}) offset({})]", page.getLimit(), page.getOffset());
093    if (datasetKeys == null) {
094      datasetKeys = ImmutableSet.of();
095    }
096    if (Strings.isNullOrEmpty(canonicalName)) {
097      return nameUsageService.list(locale,
098        datasetKeys.isEmpty() ? null : datasetKeys.iterator().next(), sourceId, page);
099    } else {
100      return nameUsageService.listByCanonicalName(locale, canonicalName, page,
101        datasetKeys.isEmpty() ? null : datasetKeys.toArray(new UUID[datasetKeys.size()]));
102    }
103  }
