001package org.gbif.checklistbank.service.mybatis;
002
003import org.gbif.api.model.checklistbank.DatasetMetrics;
004import org.gbif.api.service.checklistbank.DatasetMetricsService;
005
006import java.util.List;
007import java.util.Map;
008import java.util.UUID;
009
010import com.google.common.collect.ComparisonChain;
011import com.google.common.collect.Maps;
012import com.google.common.collect.Ordering;
013import com.google.inject.Inject;
014import org.slf4j.Logger;
015import org.slf4j.LoggerFactory;
016
017/**
018 * Implements the NameUsageService using MyBatis.
019 * All PagingResponses will not have the count set as it can be too costly sometimes.
020 */
021public class DatasetMetricsServiceMyBatis implements DatasetMetricsService {
022
023  private static final Logger LOG = LoggerFactory.getLogger(DatasetMetricsServiceMyBatis.class);
024  private final DatasetMetricsMapper mapper;
025
026  @Inject
027  DatasetMetricsServiceMyBatis(DatasetMetricsMapper mapper) {
028    this.mapper = mapper;
029  }
030
031  public static class Count<T> implements Comparable<Count<T>>{
032    private T key;
033    private Integer count;
034
035    public Count() {
036    }
037
038    public Count(T key, Integer count) {
039      this.key = key;
040      this.count = count;
041    }
042
043    public T getKey() {
044      return key;
045    }
046
047    public void setKey(T key) {
048      this.key = key;
049    }
050
051    public Integer getCount() {
052      return count;
053    }
054
055    public void setCount(Integer count) {
056      this.count = count;
057    }
058
059    @Override
060    public int compareTo(Count<T> that) {
061      return ComparisonChain.start()
062               .compare(this.count, that.count, Ordering.natural().reverse().nullsLast())
063               .compare(this.key.toString(), that.key.toString())
064               .result();
065    }
066  }
067
068  /**
069   * @return sorted by highest count first
070   */
071  private <T> Map<T, Integer> toMap(List<? extends Count<T>> counts) {
072    Map<T, Integer> typedMap = Maps.newLinkedHashMap();
073    for (Count<T> count : counts) {
074      if (typedMap.containsKey(count.getKey())) {
075        // aggregate if the same language exists - could be that the db uses 2 iso codes for the same language or null
076        typedMap.put(count.getKey(), typedMap.get(count.getKey()) + count.getCount());
077      } else {
078        typedMap.put(count.getKey(), count.getCount());
079      }
080    }
081    return typedMap;
082  }
083
084  /**
085   * Returns the latest metric for the given dataset.
086   */
087  @Override
088  public DatasetMetrics get(UUID datasetKey) {
089    return mapper.get(datasetKey);
090  }
091
092  /**
093   * Method that calculates all dataset metrics and then persists them as a new dataset_metrics record.
094   * @param datasetKey
095   */
096  public DatasetMetrics create(UUID datasetKey) {
097    LOG.info("Create new dataset metrics for {}", datasetKey);
098    mapper.insert(datasetKey);
099    return get(datasetKey);
100  }
101
102  @Override
103  public List<DatasetMetrics> list(UUID datasetKey) {
104    return mapper.list(datasetKey);
105  }
106
107
108  /**
109   * @return The percentage of total that count covers, or 0 should the total be 0  
110   */
111  protected static int getPercentage(int count, int total) {
112    return total > 0 ? count * 100 / total : 0; 
113  }
114
115}
