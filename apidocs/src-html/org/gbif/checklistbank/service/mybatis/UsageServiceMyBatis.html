001package org.gbif.checklistbank.service.mybatis;
002
003import org.gbif.api.model.checklistbank.NameUsage;
004import org.gbif.api.model.common.paging.Pageable;
005import org.gbif.api.model.common.paging.PagingResponse;
006import org.gbif.checklistbank.service.UsageService;
007import org.gbif.checklistbank.service.mybatis.model.Usage;
008
009import java.sql.Connection;
010import java.sql.SQLException;
011import java.util.List;
012import java.util.UUID;
013import javax.sql.DataSource;
014
015import com.google.inject.Inject;
016import com.jolbox.bonecp.ConnectionHandle;
017import org.postgresql.PGConnection;
018import org.slf4j.Logger;
019import org.slf4j.LoggerFactory;
020
021/**
022 * Implements the NameUsageService using MyBatis.
023 * All PagingResponses will not have the count set as it can be too costly sometimes.
024 */
025public class UsageServiceMyBatis implements UsageService {
026
027  private static final Logger LOG = LoggerFactory.getLogger(UsageServiceMyBatis.class);
028
029  private final NameUsageMapper mapper;
030  private final UsageMapper usageMapper;
031
032  @Inject
033  private DataSource ds;
034
035  @Inject
036  UsageServiceMyBatis(NameUsageMapper mapper, UsageMapper usageMapper) {
037    this.mapper = mapper;
038    this.usageMapper = usageMapper;
039  }
040
041  @Override
042  public List<Integer> listAll() {
043    Connection con = null;
044    try {
045      con = ds.getConnection();
046      ConnectionHandle boneCon = (ConnectionHandle) con;
047      PGConnection pgcon = (PGConnection) boneCon.getInternalConnection();
048      IntArrayPgWriter intMapper = new IntArrayPgWriter();
049      pgcon.getCopyAPI().copyOut("copy (select id from name_usage) TO STDOUT WITH NULL '' ", intMapper);
050      return intMapper.result();
051    } catch (Exception e) {
052      LOG.error("Failed to load all usage ids", e);
053      throw new RuntimeException("Exception while loading usage ids", e);
054    } finally {
055      if (con != null) {
056        try {
057          con.close();
058        } catch (SQLException e) {
059        }
060      }
061    }
062  }
063
064  @Override
065  public Integer maxUsageKey(UUID datasetKey) {
066    return mapper.maxUsageKey(datasetKey);
067  }
068
069  @Override
070  public List<NameUsage> listRange(int usageKeyStart, int usageKeyEnd) {
071    return mapper.listRange(usageKeyStart, usageKeyEnd);
072  }
073
074  @Override
075  public PagingResponse<Usage> list(UUID datasetKey, Pageable page) {
076    return new PagingResponse<Usage>(page, null, usageMapper.list(datasetKey, page));
077  }
078
079}
