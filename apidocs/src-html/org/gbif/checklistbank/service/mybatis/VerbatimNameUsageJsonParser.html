001package org.gbif.checklistbank.service.mybatis;
002
003import org.gbif.api.model.checklistbank.VerbatimNameUsage;
004import org.gbif.api.vocabulary.Extension;
005
006import java.io.IOException;
007import java.util.ArrayList;
008import java.util.HashMap;
009import java.util.List;
010import java.util.Map;
011
012import com.google.common.base.Strings;
013import org.codehaus.jackson.map.ObjectMapper;
014import org.codehaus.jackson.type.TypeReference;
015import org.slf4j.Logger;
016import org.slf4j.LoggerFactory;
017
018public class VerbatimNameUsageJsonParser {
019
020  private static final Logger LOG = LoggerFactory.getLogger(VerbatimNameUsageJsonParser.class);
021
022  private ObjectMapper jsonMapper = new ObjectMapper();
023  private final TypeReference<HashMap<String, Object>> jsonMapTypeRef = new TypeReference<HashMap<String, Object>>() {
024  };
025
026  /**
027   * Parses a json object for the complete, verbatim star record as stored in clb.
028   * Example input:
029   * {"id":"100",
030   * "taxonomicStatus":"valid",
031   * "taxonRank":"Species",
032   * "scientificNameAuthorship":null,
033   * "parentNameUsageID":"86",
034   * "acceptedNameUsageID":null,
035   * "scientificName":"Spirillum beijerinckii",
036   * "extensions": {
037   * "VernacularName" : [{"vernacularName":"", "language":"en", ...}, {...}],
038   * "Distribution" : [{...}, {...}}],
039   * }
040   *
041   * @param json the raw data serialized in json as given above
042   *
043   * @return the VerbatimNameUsage or null
044   */
045  public VerbatimNameUsage jsonToVerbatim(String json) {
046    if (Strings.isNullOrEmpty(json)) {
047      return null;
048    }
049    try {
050      Map<String, Object> data = jsonMapper.readValue(json, jsonMapTypeRef);
051      VerbatimNameUsage verbatim = new VerbatimNameUsage();
052      for (String key : data.keySet()) {
053        if (!key.equals("extensions")) {
054          if (data.get(key) != null) {
055            verbatim.getFields().put(key, data.get(key).toString());
056          }
057        } else {
058          Map<String, List<Map<String, String>>> extData = (Map) data.get(key);
059          for (String rowType : extData.keySet()) {
060            Extension extension = Extension.fromRowType(rowType);
061            if (extension != null) {
062              // add extension
063              List<Map<String, String>> records = new ArrayList<Map<String, String>>();
064              verbatim.getExtensions().put(extension, records);
065              // add records
066              records.addAll(extData.get(rowType));
067            }
068          }
069        }
070      }
071      return verbatim;
072
073    } catch (IOException e) {
074      LOG.error("Cannot deserialize raw json data", e);
075    }
076    return null;
077  }
078
079}
