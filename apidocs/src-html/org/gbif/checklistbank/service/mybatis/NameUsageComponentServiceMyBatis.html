001package org.gbif.checklistbank.service.mybatis;
002
003import org.gbif.api.model.Constants;
004import org.gbif.api.model.common.paging.Pageable;
005import org.gbif.api.model.common.paging.PagingRequest;
006import org.gbif.api.model.common.paging.PagingResponse;
007import org.gbif.api.service.checklistbank.NameUsageComponentService;
008
009import java.util.List;
010import javax.annotation.Nullable;
011
012import com.google.inject.Inject;
013
014/**
015 * Implements the NameUsageComponentService using MyBatis.
016 * All PagingResponses will not have the count set as it can be too costly sometimes.
017 *
018 * @param <T> the interpreted model class.
019 */
020public class NameUsageComponentServiceMyBatis<T> implements NameUsageComponentService<T> {
021
022  private final NameUsageComponentMapper<T> mapper;
023
024  @Inject
025  NameUsageComponentServiceMyBatis(NameUsageComponentMapper<T> mapper) {
026    this.mapper = mapper;
027  }
028
029  @Override
030  public T get(int key) {
031    return mapper.get(key);
032  }
033
034  @Override
035  public PagingResponse<T> listByUsage(int usageKey, @Nullable Pageable page) {
036    if (page == null) {
037      page = new PagingRequest();
038    }
039    // is this a non nub usage?
040    List<T> result;
041    if (isNub(usageKey)) {
042      result = mapper.listByNubUsage(usageKey, page);
043    } else {
044      result = mapper.listByChecklistUsage(usageKey, page);
045    }
046
047    return new PagingResponse<T>(page, null, result);
048  }
049
050  private boolean isNub(int usageKey) {
051    return usageKey <= Constants.NUB_MAXIMUM_KEY;
052  }
053
054  /**
055   * Lists all name usages with a key between start / end.
056   *
057   * @throws IllegalArgumentException if start <= end
058   */
059  public List<T> listRange(int usageKeyStart, int usageKeyEnd) {
060    if (usageKeyStart > usageKeyEnd) {
061      throw new IllegalArgumentException("start " + usageKeyStart + " > end " + usageKeyEnd + " range");
062    }
063    if (isNub(usageKeyEnd)) {
064      // only nub usages
065      return mapper.listByNubUsageRange(usageKeyStart, usageKeyEnd);
066    } else if (!isNub(usageKeyStart)) {
067      // only checklist usages
068      return mapper.listByChecklistUsageRange(usageKeyStart, usageKeyEnd);
069    } else {
070      // mixed !!!
071      // get nubs first
072      List<T> usages = mapper.listByNubUsageRange(usageKeyStart, Constants.NUB_MAXIMUM_KEY);
073      usages.addAll(mapper.listByChecklistUsageRange(Constants.NUB_MAXIMUM_KEY + 1, usageKeyEnd));
074      return usages;
075    }
076  }
077
078}
