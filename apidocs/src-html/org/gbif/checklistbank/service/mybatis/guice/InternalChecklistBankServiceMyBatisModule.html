073/**
074 * This Module should not be used, use the
075 * {@link org.gbif.checklistbank.service.mybatis.guice.ChecklistBankServiceMyBatisModule} instead.
076 */
077public class InternalChecklistBankServiceMyBatisModule extends MyBatisModule {
078
079  public static final String DATASOURCE_BINDING_NAME = "checklistbank";
080
081  public InternalChecklistBankServiceMyBatisModule() {
082    super(DATASOURCE_BINDING_NAME);
083  }
084
085  @Override
086  protected void bindMappers() {
087    // mybatis config
088    addAlias("Usage").to(Usage.class);
089    addAlias("NameUsage").to(NameUsage.class);
090    addAlias("NameUsageMetrics").to(NameUsageMetrics.class);
091    addAlias("ParsedName").to(ParsedName.class);
092    addAlias("VernacularName").to(VernacularName.class);
093    addAlias("Reference").to(Reference.class);
094    addAlias("Description").to(Description.class);
095    addAlias("Identifier").to(Identifier.class);
096    addAlias("Distribution").to(Distribution.class);
097    addAlias("Image").to(Image.class);
098    addAlias("SpeciesProfile").to(SpeciesProfile.class);
099    addAlias("TypeSpecimen").to(TypeSpecimen.class);
100    addAlias("DatasetMetrics").to(DatasetMetrics.class);
101    addAlias("VerbatimNameUsage").to(VerbatimNameUsage.class);
102    addAlias("Count").to(DatasetMetricsServiceMyBatis.Count.class);
103
104    // mybatis mapper
105    addMapperClass(UsageMapper.class);
106    addMapperClass(NameUsageMapper.class);
107    addMapperClass(VernacularNameMapper.class);
108    addMapperClass(ReferenceMapper.class);
109    addMapperClass(DescriptionMapper.class);
110    addMapperClass(IdentifierMapper.class);
111    addMapperClass(DistributionMapper.class);
112    addMapperClass(ImageMapper.class);
113    addMapperClass(SpeciesProfileMapper.class);
114    addMapperClass(TypeSpecimenMapper.class);
115    addMapperClass(DatasetMetricsMapper.class);
116    addMapperClass(VerbatimNameUsageMapper.class);
117    addMapperClass(ParsedNameMapper.class);
118    addMapperClass(CitationMapper.class);
119    addMapperClass(NameUsageMetricsMapper.class);
120    addMapperClass(NubRelMapper.class);
121    addMapperClass(ColAnnotationMapper.class);
122  }
123
124  @Override
125  protected void bindTypeHandlers() {
126    // mybatis type handler
127    handleType(Country.class).with(CountryTypeHandler.class);
128    handleType(Language.class).with(LanguageTypeHandler.class);
129    handleType(UUID.class).with(UuidTypeHandler.class);
130  }
131
132  @Override
133  protected void bindManagers() {
134    // services. Make sure they are also exposed in the public module!
135    bind(NameUsageService.class).to(NameUsageServiceMyBatis.class).in(Scopes.SINGLETON);
136    bind(VernacularNameService.class).to(VernacularNameServiceMyBatis.class).in(Scopes.SINGLETON);
137    bind(ReferenceService.class).to(ReferenceServiceMyBatis.class).in(Scopes.SINGLETON);
138    bind(DescriptionService.class).to(DescriptionServiceMyBatis.class).in(Scopes.SINGLETON);
139    bind(DistributionService.class).to(DistributionServiceMyBatis.class).in(Scopes.SINGLETON);
140    bind(ImageService.class).to(ImageServiceMyBatis.class).in(Scopes.SINGLETON);
141    bind(SpeciesProfileService.class).to(SpeciesProfileServiceMyBatis.class).in(Scopes.SINGLETON);
142    bind(TypeSpecimenService.class).to(TypeSpecimenServiceMyBatis.class).in(Scopes.SINGLETON);
143    bind(DatasetMetricsService.class).to(DatasetMetricsServiceMyBatis.class).in(Scopes.SINGLETON);
144    // not exposed in API
145    bind(UsageService.class).to(UsageServiceMyBatis.class).in(Scopes.SINGLETON);
146    bind(ParsedNameService.class).to(ParsedNameServiceMyBatis.class).in(Scopes.SINGLETON);
147    bind(DatasetImportService.class).to(DatasetImportServiceMyBatis.class).in(Scopes.SINGLETON);
148    bind(CitationService.class).to(CitationServiceMyBatis.class).in(Scopes.SINGLETON);
149    bind(ColAnnotationService.class).to(ColAnnotationServiceMyBatis.class).in(Scopes.SINGLETON);
150  }
