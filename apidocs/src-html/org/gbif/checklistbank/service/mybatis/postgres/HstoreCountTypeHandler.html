001package org.gbif.checklistbank.service.mybatis.postgres;
002
003import java.sql.CallableStatement;
004import java.sql.PreparedStatement;
005import java.sql.ResultSet;
006import java.sql.SQLException;
007import java.util.HashMap;
008import java.util.Map;
009
010import com.google.common.annotations.VisibleForTesting;
011import com.google.common.base.Functions;
012import com.google.common.base.Strings;
013import com.google.common.collect.ImmutableSortedMap;
014import com.google.common.collect.Maps;
015import com.google.common.collect.Ordering;
016import org.apache.ibatis.type.BaseTypeHandler;
017import org.apache.ibatis.type.JdbcType;
018import org.postgresql.util.HStoreConverter;
019
020/**
021 * A mybatis type handler that translates from the typed java.util.Map<String, Integer> to the
022 * postgres hstore database type. Any non integer values in hstore are silently ignored.
023 *
024 * As we do not map all java map types to this mybatis handler apply the handler manually for the relevant hstore fields
025 * in the mapper xml, for example see DatasetMetricsMapper.xml.
026 */
027public abstract class HstoreCountTypeHandler<KEY extends Comparable> extends BaseTypeHandler<Map<KEY, Integer>> {
028
029  @Override
030  public void setNonNullParameter(PreparedStatement ps, int i, Map<KEY, Integer> parameter, JdbcType jdbcType)
031    throws SQLException {
032    ps.setString(i, HStoreConverter.toString(parameter));
033  }
034
035  @Override
036  public Map<KEY, Integer> getNullableResult(ResultSet rs, String columnName) throws SQLException {
037    return fromString(rs.getString(columnName));
038  }
039
040  @Override
041  public Map<KEY, Integer> getNullableResult(ResultSet rs, int columnIndex) throws SQLException {
042    return fromString(rs.getString(columnIndex));
043  }
044
045  @Override
046  public Map<KEY, Integer> getNullableResult(CallableStatement cs, int columnIndex) throws SQLException {
047    return fromString(cs.getString(columnIndex));
048  }
049
050  private Map<KEY, Integer> fromString(String hstring) {
051    HashMap<KEY, Integer> typedMap = Maps.newHashMap();
052    if (!Strings.isNullOrEmpty(hstring)) {
053      Map<String, String> rawMap = HStoreConverter.fromString(hstring);
054      for (Map.Entry<String, String> entry : rawMap.entrySet()) {
055        try {
056          typedMap.put(toKey(entry.getKey()), Integer.parseInt(entry.getValue()));
057        } catch (IllegalArgumentException e) {
058          // ignore this entry
059        }
060      }
061    }
062    return sortMap(typedMap);
063  }
064
065  /**
066   * Can be overridden to return sorted maps in custom manners.
067   * By default sorts the map according to the count values in descending order.
068   */
069  protected Map<KEY, Integer> sortMap(HashMap<KEY, Integer> map) {
070    return sortMapByValuesDesc(map);
071  }
072
073  @VisibleForTesting
074  protected static <KEY extends Comparable> Map<KEY, Integer> sortMapByValuesDesc(HashMap<KEY, Integer> map) {
075    final Ordering<KEY> reverseValuesAndNaturalKeysOrdering =
076        Ordering.natural().reverse().nullsLast().onResultOf(Functions.forMap(map, null)) // natural for values
077            .compound(Ordering.natural()); // secondary - natural ordering of keys
078    return ImmutableSortedMap.copyOf(map, reverseValuesAndNaturalKeysOrdering);
079  }
080
081  protected abstract KEY toKey(String key) throws IllegalArgumentException;
082
083}
