059  @Override
060  public NameUsage get(int usageKey, @Nullable Locale locale) {
061    NameUsage usage = mapper.get(usageKey);
062    addVernacularName(usage, getLanguage(locale));
063    addIdentifiers(usage);
064    return usage;
065  }

067  @Override
068  public VerbatimNameUsage getVerbatim(int usageKey) {

072    }
073    return v;
074  }

087  @Override
088  public PagingResponse<NameUsage> list(Locale locale, @Nullable UUID datasetKey, @Nullable String sourceId,
089    @Nullable Pageable page) {
090    if (page == null) {
091      page = new PagingRequest();
092    }
093
094    List<NameUsage> usages;
095    // if a sourceID has been provided as a filter, there wont be too many results
096    if (!Strings.isNullOrEmpty(sourceId)) {
097      usages = mapper.listBySourceId(datasetKey, sourceId, page);
098
099    } else {
100      usages = Lists.newArrayList();
101      // we need to page over ids only and then lookup each full usage by its key
102      // otherwise the db queries get too heavy for the database if offsets are high!
103      List<Integer> usageIds = mapper.list(datasetKey, page);
104      for (Integer uid : usageIds) {
105        usages.add(get(uid, locale));
106      }
107    }
108    return localizedPage(locale, usages, page);
109  }

122  @Override
123  public List<NameUsage> listParents(int usageKey, @Nullable Locale locale) {
124    LinkedList<NameUsage> parents = Lists.newLinkedList();
125    NameUsage curr = get(usageKey, locale);
126    if (curr != null) {
127      // make sure we don't hit some circular structures which should not exist really, but who knows...
128      Set<Integer> visited = Sets.newHashSet();
129      visited.add(usageKey);
130      while (curr.getParentKey() != null) {
131        int parentKey = curr.getParentKey();
132        if (visited.contains(parentKey)) {
133          LOG.warn("Taxonomic hierarchy for usage {} contains circular references!", parentKey);
134          break;
135        }
136        visited.add(parentKey);
137        curr = get(parentKey, locale);
138        if (curr == null) {
139          LOG.warn("Usage not found for parent key {}", parentKey);
140          break;
141        }
142        parents.addFirst(curr);
143      }
144    }
145    return parents;
146  }

181  public List<NameUsage> listRelated(int nubKey, @Nullable Locale locale, @Nullable UUID... datasetKey) {
182    List<NameUsage> usages = mapper.listRelated(nubKey, datasetKey);

188  /**
189   * @return the lowercase 2 letter iso language code or null
190   */
191  protected static String getLanguage(@Nullable Locale locale) {
192    if (locale == null || locale.getLanguage() == null || locale.getLanguage().length() != 2) {
193      return null;
194    }
195    return locale.getLanguage().toLowerCase();
196  }

198  /**
199   * Adds a matching vernacular name of a given language to a list of existing name usages.
200   * See {@link #addVernacularName(org.gbif.api.model.checklistbank.NameUsage, String)} for logic used in picking the common name.
201   *
202   * @param usages   the checklist usages to add a vernacular name to
203   * @param language the requested language
204   */
205  private void addVernacularNames(List<NameUsage> usages, @Nullable String language) {
206    for (NameUsage u : usages) {
207      addVernacularName(u, language);
208    }
209  }
210
211
212  /**
213   * Adds a vernacular name of a given language to an existing name usage.
214   * If language is null or no common name for that language exists english will be used.
215   *
216   * @param u        the checklist usage to add the vernacular name to
217   * @param language the requested language
218   */
219  private void addVernacularName(NameUsage u, @Nullable String language) {
220    if (u == null || language == null) {
221      return;
222    }
223
224    // first try with given language
225    VernacularName v = getVernacularNameInLanguage(u, language);
226
227    // try with english if nothing found and its another language
228    if (v == null && !language.equals(DescriptionServiceMyBatis.ENGLISH)) {
229      v = getVernacularNameInLanguage(u, DescriptionServiceMyBatis.ENGLISH);
230    }
231
232    if (v != null) {
233      u.setVernacularName(v.getVernacularName());
234    }
235  }
236
237  /**
238   * Gets a single vernacular name for a given language and a name usage.
239   *
240   * @param u        the name usage of the vernacular name
241   * @param language the requested language, never null
242   *
243   * @return vernacular name or null if none can be found
244   */
245  private VernacularName getVernacularNameInLanguage(NameUsage u, String language) {
246    if (u.isNub()) {
247      return vernacularNameMapper.getByNubUsage(u.getKey(), language);
248    } else {
249      return vernacularNameMapper.getByChecklistUsage(u.getKey(), language);
250    }
251  }

254  /**
255   * Adds all identifiers to a list of existing name usages.
256   *
257   * @param usages the checklist usages to add identifiers to
258   */
259  private void addIdentifiers(List<NameUsage> usages) {
260    for (NameUsage u : usages) {
261      addIdentifiers(u);
262    }
263  }
264
265  /**
266   * Adds all identifiers for an existing name usage.
267   *
268   * @param u the checklist usage to add the vernacular name to
269   */
270  private void addIdentifiers(NameUsage u) {
271    if (u == null) {
272      return;
273    }
274    // we will never have that many ids. 10 would be very many already and far more then the maximum right now.
275    PagingRequest page = new PagingRequest(0, 100);
276    // only load direct identifiers, treating nub usages the same as others
277    u.getIdentifiers().addAll(identifierMapper.listByChecklistUsage(u.getKey(), page));
278  }
