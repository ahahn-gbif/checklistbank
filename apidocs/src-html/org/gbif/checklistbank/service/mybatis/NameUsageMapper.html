001package org.gbif.checklistbank.service.mybatis;
002
003import org.gbif.api.model.checklistbank.NameUsage;
004import org.gbif.api.model.common.paging.Pageable;
005import org.gbif.api.vocabulary.Rank;
006
007import java.util.List;
008import java.util.UUID;
009
010import org.apache.ibatis.annotations.Param;
011
012/**
013 * The MyBatis mapper interface for NameUsages.
014 *
015 * @see org.gbif.api.service.checklistbank.NameUsageService
016 */
017public interface NameUsageMapper {
018
019  NameUsage get(@Param("key") int key);
020
021  /**
022   * A simple paging query for all usages in checklistbank.
023   * We only return name usage ids here to avoid extremely heavy operations for the database when the offset gets
024   * bigger.
025   *
026   * @return the name usage ids belonging to the requested page.
027   */
028  List<Integer> list(@Param("uuid") UUID datasetKey, @Param("page") Pageable page);
029
030  List<NameUsage> listBySourceId(@Param("uuid") UUID datasetKey, @Param("sourceId") String sourceId,
031    @Param("page") Pageable page);
032
033  List<NameUsage> listByCanonicalName(@Param("canonical") String canonicalName, @Param("uuids") UUID[] datasetKey,
034    @Param("page") Pageable page);
035
036  List<NameUsage> listRange(@Param("start") int usageKeyStart, @Param("end") int usageKeyEnd);
037
038  List<NameUsage> listRoot(@Param("uuid") UUID datasetKey, @Param("page") Pageable page);
039
040  List<NameUsage> listChildren(@Param("key") int parentKey, @Param("page") Pageable page);
041
042  /**
043   * To avoid long queries we use concrete nested set indices here, not a usage key.
044   *
045   * @see <a href="http://gbif.blogspot.de/2012/06/taxonomic-trees-in-postgresql.html">blog post</a>
046   */
047  List<NameUsage> listDescendants(@Param("uuid") UUID datasetKey, @Param("lft") int lft, @Param("rgt") int rgt,
048    @Param("rank") Rank rank, @Param("page") Pageable page);
049
050  List<NameUsage> listSynonyms(@Param("key") int usageKey, @Param("page") Pageable page);
051
052  /**
053   * List all related name usages that have a given nubKey.
054   */
055  List<NameUsage> listRelated(@Param("key") int nubKey, @Param("uuids") UUID[] datasetKeys);
056
057  /**
058   * @return the maximum usage key used in a dataset
059   */
060  Integer maxUsageKey(@Param("uuid") UUID datasetKey);
061
062}
