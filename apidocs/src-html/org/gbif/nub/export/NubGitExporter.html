013import com.google.common.base.Joiner;

015import com.google.common.io.Resources;
016import com.google.inject.Guice;
017import com.google.inject.Inject;
018import com.google.inject.Injector;

020import org.codehaus.jackson.map.SerializationConfig;
021import org.slf4j.Logger;
022import org.slf4j.LoggerFactory;
023
024/**
025 * Experimental export of the nub or any other dataset as README and json files in a hierarchical filestructure that
026 * can be browsed in github nicely.
027 */
028public class NubGitExporter {
029  private static final Logger LOG = LoggerFactory.getLogger(NubGitExporter.class);
030  private static final Charset UTF8 = Charset.forName("UTF8");
031
032  private int PAGE_SIZE = 10000;
033  private final String ROOTDIR = "/Users/mdoering/dev/backbone/life";
034  private final Joiner pathJoiner = Joiner.on(File.separator).skipNulls();
035  private final String readmeTempl;
036  private final UsageService usageService;
037  private ObjectMapper mapper = new ObjectMapper();
038
039  @Inject
040  public NubGitExporter(UsageService usageService) throws IOException {
041    this.usageService = usageService;
042    readmeTempl = Resources.toString(Resources.getResource("exporter_readme.txt"), UTF8);
043    mapper.enable(SerializationConfig.Feature.INDENT_OUTPUT);
044  }
045
046  public void run() {
047    final int max = usageService.maxUsageKey(Constants.NUB_DATASET_KEY) + 1;
048    int start = 0;
049    LOG.info("Retrieved max backbone usage key {}", max);
050    while (start < max) {
051      int end = start + PAGE_SIZE - 1;
052      List<NameUsage> usages = usageService.listRange(start, end);
053      LOG.info("Retrieved {} usages from {}-{}", usages.size(), start, end);
054      for (NameUsage u : usages) {
055        dumpUsage(u);
056      }
057      start += PAGE_SIZE;
058    }
059  }
060
061  private void dumpUsage(NameUsage u) {
062    try {
063      StringBuilder name = new StringBuilder();
064      if (u.isSynonym()) {
065        name.append(" Syn. ");
066      }
067//      name.append(u.getRank().ordinal());
068//      name.append(u.getRank());
069//      name.append(" ");
070      name.append(u.getCanonicalName());
071
072      File uDir = new File(pathJoiner.join(ROOTDIR, pathJoiner.join(u.getHigherClassificationMap().values()), name.toString()));
073      File readme = new File(uDir, "README.md");
074      File json = new File(uDir, "data.json");
075      // create folders if missing
076      Files.createParentDirs(readme);
077      // write readme
078      Files.write(String.format(readmeTempl, u.getScientificName(), u.getRank(), u.getTaxonomicStatus(), u.getAccordingTo(), u.getPublishedIn(), u.getBasionym(), u.getRemarks()), readme, UTF8);
079      // write json
080      mapper.writeValue(json, u);
081
082    } catch (IOException e) {
083      LOG.error("Failed to write usage {}: {}", u.getKey(), u.getScientificName(), e);
084    }
085  }
086
087  public static void main(String[] args) throws IOException {
088    Injector inj = Guice.createInjector(new ExporterModule(PropertiesUtil.loadProperties("nub.properties")));
089    NubGitExporter exp = inj.getInstance(NubGitExporter.class);
090    exp.run();
091  }
092}
