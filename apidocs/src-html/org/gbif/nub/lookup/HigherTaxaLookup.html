042/**
043 * A tool to lookup synonyms for higher taxa names. It does a rather strong match and tries to normalize names
044 * a lot so we can use them for comparisons of entire higher classifications.
045 *
046 * The class uses file based dictionaries which are maintained on rs.gbif.org with OpenRefine based on real data found
047 * in occurrence and checklist records.
048 */
049public class HigherTaxaLookup {
050  private final static Map<Rank, String> SYNONYM_FILENAMES = Maps.newHashMap();
051  static {
052    SYNONYM_FILENAMES.put(Rank.KINGDOM, "kingdom.txt");
053    SYNONYM_FILENAMES.put(Rank.PHYLUM, "phylum.txt");
054    SYNONYM_FILENAMES.put(Rank.CLASS, "class.txt");
055    SYNONYM_FILENAMES.put(Rank.ORDER, "order.txt");
056    SYNONYM_FILENAMES.put(Rank.FAMILY, "family.txt");
057  }
058  private static final Set<String> NON_NAMES = Sets.newHashSet();
059
060  private Logger log = LoggerFactory.getLogger(HigherTaxaLookup.class);
061  private Map<Rank, Map<String, String>> syn = Maps.newHashMap();
062  private Map<String, Kingdom> kingdoms = Maps.newHashMap();
063
064  /**
065   *
066   */
067  public HigherTaxaLookup() {
068    for (Kingdom k : Kingdom.values()) {
069      this.kingdoms.put(norm(k.name()), k);
070    }
071  }
072
073  /**
074   * Lookup higher taxa synonym dictionary across all ranks and return the first match found
075   *
076   * @param higherTaxon
077   * @return the looked up accepted name or the original higherTaxon
078   */
079  public String lookup(String higherTaxon) {
080    if (higherTaxon == null) {
081      return null;
082    }
083    for (Rank r : syn.keySet()) {
084      String result = lookup(higherTaxon, r);
085      if (result != null) {
086        return result;
087      }
088    }
089    return higherTaxon;
090  }
091
092  /**
093   * Lookup synonym for given higher rank.
094   * Can be null.
095   *
096   * @param higherTaxon higher rank name, case insensitive
097   * @param rank the rank to lookup for
098   * @return the looked up accepted name, null for blacklisted names or the original higherTaxon if no synonym is known
099   */
100  public String lookup(String higherTaxon, Rank rank) {
101    if (higherTaxon == null) {
102      return null;
103    }
104    if (isBlacklisted(higherTaxon)) {
105      return null;
106    }
107    if (syn.containsKey(rank)) {
108      String normedHT = norm(higherTaxon);
109      Map<String, String> x = syn.get(rank);
110      if (syn.get(rank).containsKey(normedHT)) {
111        return syn.get(rank).get(normedHT);
112      }
113    }
114    return higherTaxon;
115  }
116
117  /**
118   * Check for obvious, blacklisted garbage and return true if thats the case.
119   * The underlying set is hosted at http://rs.gbif.org/dictionaries/authority/blacklisted.txt
120   */
121  public boolean isBlacklisted(String name) {
122    if (name != null) {
123      name = norm(name);
124      if (NON_NAMES.contains(name)) {
125        return true;
126      }
127    }
128    return false;
129  }
130
131  /**
132   * @return non empty uppercased string with normalized whitespace and all non latin letters replaced. Or null
133   */
134  @VisibleForTesting
135  protected static String norm(String x) {
136    Pattern REMOVE_NON_LETTERS = Pattern.compile("[\\W\\d]+");
137    x = Strings.nullToEmpty(x);
138    x = REMOVE_NON_LETTERS.matcher(x).replaceAll(" ");
139    x = StringUtils.normalizeSpace(x).toUpperCase();
140    return StringUtils.trimToNull(x);
141  }
142
143  private Map<String, String> readSynonymStream(Rank rank, InputStream in) {
144    Map<String, String> synonyms = Maps.newHashMap();
145    try {
146      synonyms = FileUtils.streamToMap(in, 0, 1, true);
147    } catch (IOException e) {
148      log.warn("Cannot read synonym map from stream for {}. Use empty map instead.", rank, e);
149    } finally {
150      Closeables.closeQuietly(in);
151    }
152    log.debug("loaded " + synonyms.size() + " synonyms for " + rank);
153    return synonyms;
154  }
155
156  /**
157   *
158   * @param file the synonym file on rs.gbif.org
159   * @return
160   */
161  private Map<String, String> readSynonymUrl(Rank rank, String file) {
162    try {
163      URL url = RsGbifOrg.synonymUrl(file);
164      log.debug("Reading " + url.toString());
165      return readSynonymStream(rank, url.openStream());
166    } catch (IOException e) {
167      log.warn("Cannot read synonym map from " + file + ". Use empty map instead.", e);
168    }
169    return Maps.newHashMap();
170  }
171
172  /**
173   *
174   * @param file the local file to read
175   * @return
176   */
177  private Map<String, String> readSynonymFile(Rank rank, File file) {
178    try {
179      log.debug("Reading " + file.getAbsolutePath());
180      return readSynonymStream(rank, new FileInputStream(file));
181    } catch (IOException e) {
182      log.warn("Cannot read synonym map from " + file + ". Use empty map instead: ", e.getMessage());
183    }
184    return Maps.newHashMap();
185  }

187  /**
188   * Reads blacklisted names from rs.gbif.org
189   */
190  private void readOnlineBlacklist() {
191    try {
192      URL url = RsGbifOrg.authorityUrl(RsGbifOrg.FILENAME_BLACKLIST);
193      log.debug("Reading " + url.toString());
194      readBlacklistStream(url.openStream());
195    } catch (IOException e) {
196      log.warn("Cannot read online blacklist.", e);
197    }
198  }
199
200  /**
201   * Reads blacklisted names from file
202   */
203  private void readBlacklistFile(File folder) {
204    File blacklist = new File(folder, RsGbifOrg.FILENAME_BLACKLIST);
205    try {
206      readBlacklistStream(new FileInputStream(blacklist));
207    } catch (IOException e) {
208      log.warn("Cannot read local blacklist {}. {}", blacklist.getAbsoluteFile(), e.getMessage());
209    }
210  }
211
212  /**
213   * Reads blacklisted names from stream
214   */
215  private void readBlacklistStream(InputStream in) {
216    NON_NAMES.clear();
217    try {
218      NON_NAMES.addAll(FileUtils.streamToSet(in));
219    } catch (IOException e) {
220      log.warn("Cannot read blacklist. Use empty set instead.", e);
221    } finally {
222      Closeables.closeQuietly(in);
223    }
224    log.debug("loaded " + NON_NAMES.size() + " blacklisted names");
225  }
226
227   /**
228   * Reads synonym dicts from given folder.
229     * File names must be the same as on rs.gbif.org.
230   */
231  public void loadLocalDicts(File folder) {
232    log.info("Reloading dictionary files from rs.gbif.org ...");
233
234    for (Rank rank : SYNONYM_FILENAMES.keySet()) {
235      Map<String, String> synonyms = readSynonymFile(rank, new File(folder, SYNONYM_FILENAMES.get(rank)));
236      setSynonyms(rank, synonyms);
237    }
238
239    // read blacklisted names
240    readBlacklistFile(folder);
241  }
242
243  /**
244   * Reads synonym dicts from given classpath root path.
245     * File names must be the same as on rs.gbif.org.
246   */
247  public void loadClasspathDicts(String classpathFolder) throws IOException {
248    log.info("Reloading dictionary files from classpath ...");
249
250    for (Rank rank : SYNONYM_FILENAMES.keySet()) {
251      InputStream synIn = Resources.newInputStreamSupplier(Resources.getResource(classpathFolder + "/" + SYNONYM_FILENAMES.get(rank))).getInput();
252
253      Map<String, String> synonyms = readSynonymStream(rank, synIn);
254      setSynonyms(rank, synonyms);
255    }
256
257    // read blacklisted names
258    InputStream blackIn = Resources.newInputStreamSupplier(Resources.getResource(classpathFolder + "/" + RsGbifOrg.FILENAME_BLACKLIST)).getInput();
259    readBlacklistStream(blackIn);
260  }
261
262  /**
263   * Reloads all synonym files found on rs.gbif.org replacing existing mappings.
264   */
265  public void loadOnlineDicts() {
266    log.info("Reloading dictionary files from rs.gbif.org ...");
267
268    for (Rank rank : SYNONYM_FILENAMES.keySet()) {
269      Map<String, String> synonyms = readSynonymUrl(rank, SYNONYM_FILENAMES.get(rank));
270      setSynonyms(rank, synonyms);
271    }
272
273    // read blacklisted names
274    readOnlineBlacklist();
275  }

277  /**
278   * Sets the synonym lookup map for a given rank.
279   * Names will be normalised and checked for existance of the same entry as key or value.
280   *
281   * @param rank
282   * @param synonyms
283   */
284  public void setSynonyms(Rank rank, Map<String, String> synonyms) {
285    Map<String, String> synonymsNormed = Maps.newHashMap();
286
287    // normalise keys
288    for (Entry<String, String> entry : synonyms.entrySet()) {
289      synonymsNormed.put(norm(entry.getKey()), entry.getValue());
290    }
291
292    // test if synonyms show up as accepted too
293    Collection<String> syns = Sets.newHashSet(synonymsNormed.keySet());
294    for (String syn : syns) {
295      if (synonymsNormed.containsKey(synonymsNormed.get(syn))) {
296        log.warn(syn + " is both synonym and accepted - ignore synonym.");
297        synonymsNormed.remove(syn);
298      }
299    }
300
301    syn.put(rank, synonymsNormed);
302    log.debug("Loaded " + synonyms.size() + " " + rank.name() + " synonyms ");
303
304    // also create kingdom enum lookup in case of kingdom synonyms
305    if (Rank.KINGDOM == rank) {
306      Map<String, String> map = syn.get(Rank.KINGDOM);
307      if (map != null) {
308        for (String syn : map.keySet()) {
309          Kingdom k = null;
310          String key = map.get(syn);
311          if (key != null) {
312            key = key.toLowerCase();
313            key = StringUtils.capitalize(key);
314            try {
315              k = Kingdom.valueOf(key);
316            } catch (Exception e) {
317            }
318          }
319          this.kingdoms.put(norm(syn), k);
320        }
321      }
322      for (Kingdom k : Kingdom.values()) {
323        this.kingdoms.put(norm(k.name()), k);
324      }
325    }
326
327  }
328
329  /**
330   * @return the number of entries across all ranks
331   */
332  public int size() {
333    int all = 0;
334    for (Rank r : syn.keySet()) {
335      all += size(r);
336    }
337    return all;
338  }
339
340  /**
341   * @return the number of entries for a given rank
342   */
343  public int size(Rank rank) {
344    if (syn.containsKey(rank)) {
345      return syn.get(rank).size();
346    }
347    return 0;
348  }
349
350  public Kingdom toKingdom(String kingdom) {
351    if (kingdom == null) {
352      return null;
353    }
354    return kingdoms.get(kingdom.trim().toUpperCase());
355  }
356
357  public Kingdom toKingdomSynonymOnly(String kingdom) {
358    if (kingdom == null) {
359      return null;
360    }
361    Kingdom k = kingdoms.get(kingdom.trim().toUpperCase());
362    if (k != null && k.name().equalsIgnoreCase(kingdom)) {
363      // we have the accepted kingdom name, dont return it
364      return null;
365    }
366    return k;
367  }
