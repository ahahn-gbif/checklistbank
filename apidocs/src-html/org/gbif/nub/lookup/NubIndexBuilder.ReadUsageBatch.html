001package org.gbif.nub.lookup;
002
003import org.gbif.api.model.Constants;
004import org.gbif.api.model.checklistbank.NameUsage;
005import org.gbif.checklistbank.index.ThreadPoolRunner;
006import org.gbif.checklistbank.service.UsageService;
007
008import java.io.IOException;
009import java.util.List;
010import java.util.concurrent.Callable;
011import java.util.concurrent.atomic.AtomicInteger;
012
013import org.apache.lucene.index.TrackingIndexWriter;
014import org.slf4j.Logger;
015import org.slf4j.LoggerFactory;
016
017public class NubIndexBuilder extends ThreadPoolRunner<List<NameUsage>> {
018
019  private static final Logger LOG = LoggerFactory.getLogger(NubIndexBuilder.class);
020
021  private final TrackingIndexWriter writer;
022  private AtomicInteger counter = new AtomicInteger();
023  private UsageService usageService;
024  private int offset;
025  private final int max;
026  // the size of the key range used to get usages.
027  // Not all keys exist and they are not evenly distributed, so this is the max of usages returned, but its usually less
028  private final int window;
029
030  NubIndexBuilder(TrackingIndexWriter writer, UsageService usageService, int threads) {
031    super(threads);
032    this.writer = writer;
033    this.usageService = usageService;
034    this.offset = 0;
035    this.max = usageService.maxUsageKey(Constants.NUB_DATASET_KEY);
036    this.window = 20000;
037    // use sth like the following for local debugging: (puma concolor id)
038    //this.window = 100;
039    //this.offset = 2435099-window;
040    //this.max = offset + 2*window;
041  }
042
043  @Override
044  protected Callable<List<NameUsage>> newJob() {
045    if (offset >= max) {
046      // no more jobs!
047      return null;
048    }
049    int start = offset;
050    int end = Math.min(offset + window - 1, max);
051    this.offset = end + 1;
052    return new ReadUsageBatch(start, end, usageService);
053  }
054
055  @Override
056  /**
057   * Consumes the batch results and inserts them into the index.
058   */
059  protected void taskResponseHook(List<NameUsage> usages) {
060    LOG.debug("Add {} usages to index", usages.size());
061    if (!usages.isEmpty()) {
062      LOG.debug("First usage is: {}", usages.get(0).getScientificName());
063    }
064    for (NameUsage u : usages) {
065      try {
066        writer.addDocument(NubIndex.toDoc(u));
067      } catch (IOException e) {
068        throw new IllegalStateException(e);
069      }
070    }
071    counter.getAndAdd(usages.size());
072    LOG.info("{} usages added to nub index, {} in total", usages.size(), counter);
073  }
074
075
076  public class ReadUsageBatch implements Callable<List<NameUsage>> {
077      private final int start;
078      private final int end;
079      private final UsageService usageService;
080
081      public ReadUsageBatch(int start, int end, UsageService usageService) {
082        this.start = start;
083        this.end = end;
084        this.usageService = usageService;
085      }
086
087      @Override
088      public List<NameUsage> call() throws Exception {
089        LOG.debug("Retrieve usages from CLB in range {}-{}", start, end);
090        List<NameUsage> usages = usageService.listRange(start, end);
091        LOG.debug("Retrieved {} usages from CLB in range {}-{}", usages.size(), start, end);
092        return usages;
093      }
094    }
095}
