003import java.io.IOException;

005import org.apache.lucene.analysis.TokenFilter;

011/**
012 * Based on character transpositions found in Tony Reese's TaxonMatch.
013 */
014public class ScientificNameSoundAlikeFilter extends TokenFilter {
015  private static final Logger LOG = LoggerFactory.getLogger(ScientificNameSoundAlikeFilter.class);
016
017  private final CharTermAttribute termAtt = addAttribute(CharTermAttribute.class);
018
019  /**
020   * Construct a token stream filtering the given input.
021   */
022  public ScientificNameSoundAlikeFilter(TokenStream input) {
023    super(input);
024  }
025
026  @Override
027  public final boolean incrementToken() throws IOException {
028    if (!input.incrementToken()) return false;
029
030    final char[] buffer = termAtt.buffer();
031    final int bufferLength = termAtt.length();
032
033    // Do some selective replacement on the leading letter/s only:
034    if (bufferLength > 2) {
035      String start3 = new String(buffer, 0, 2);
036      if (start3.startsWith("ae")) {
037        start3 = "E" + start3.substring(2);
038      } else if (start3.startsWith("cn")) {
039        start3 = "N" + start3.substring(2);
040      } else if (start3.startsWith("ct")) {
041        start3 = "T" + start3.substring(2);
042      } else if (start3.startsWith("cz")) {
043        start3 = "C" + start3.substring(2);
044      } else if (start3.startsWith("dj")) {
045        start3 = "J" + start3.substring(2);
046      } else if (start3.startsWith("ea")) {
047        start3 = "E" + start3.substring(2);
048      } else if (start3.startsWith("eu")) {
049        start3 = "U" + start3.substring(2);
050      } else if (start3.startsWith("gn")) {
051        start3 = "N" + start3.substring(2);
052      } else if (start3.startsWith("kn")) {
053        start3 = "N" + start3.substring(2);
054      } else if (start3.startsWith("mc")) {
055        start3 = "MAC" + start3.substring(2);
056      } else if (start3.startsWith("mn")) {
057        start3 = "N" + start3.substring(2);
058      } else if (start3.startsWith("oe")) {
059        start3 = "E" + start3.substring(2);
060      } else if (start3.startsWith("qu")) {
061        start3 = "Q" + start3.substring(2);
062      } else if (start3.startsWith("ps")) {
063        start3 = "S" + start3.substring(2);
064      } else if (start3.startsWith("pt")) {
065        start3 = "T" + start3.substring(2);
066      } else if (start3.startsWith("ts")) {
067        start3 = "S" + start3.substring(2);
068      } else if (start3.startsWith("wr")) {
069        start3 = "R" + start3.substring(2);
070      } else if (start3.startsWith("x")) {
071        start3 = "Z" + start3.substring(2);
072      }
073    }
074
075    // Now keep the leading character, then do selected "soundalike" replacements.
076    // The following letters are equated: AE, OE, E, U, Y and I; IA and A are equated;
077    // K and C; Z and S; and H is dropped.
078    // Also, A and O are equated, MAC and MC are equated, and SC and S.
079
080    if (bufferLength > 1) {
081      int upto = Character.isWhitespace(buffer[0]) ? 0 : 1;
082      char c1 = ' ';
083      for(int i=1; i<bufferLength; i++) {
084        char c = buffer[i];
085        boolean skip = false;
086
087        // replace all whitespace with spaces
088        if (Character.isWhitespace(c)) {
089          c = ' ';
090
091        } else {
092
093          switch (buffer[i]) {
094            case 'a':
095              if (c1 == 'i') {
096                upto--;
097                c='a';
098              }
099              break;
100            case 'c':
101              if (c1 == 's') {
102                skip = true;
103              }
104              break;
105            case 'e':
106              if (c1 == 'a' || c1 == 'o') {
107                upto--;
108              }
109              c='i';
110              break;
111            case 'i':
112              if (c1 == 'o') {
113                upto--;
114                c='a';
115              }
116              break;
117            case 'o': c='a'; break;
118            case 'u': c='i'; break;
119            case 'y': c='i'; break;
120            case 'k': c='c'; break;
121            case 'z': c='c'; break;
122            case 'h': skip=true; break;
123          }
124        }
125
126        // drop any repeated characters (AA becomes A, BB or BBB becomes B, etc.)
127        if (!skip && upto > 0 && buffer[upto-1] == c) {
128          skip = true;
129        }
130
131        // remember original char
132        c1 = buffer[i];
133
134        // alter buffer
135        if (!skip) {
136          buffer[upto++] = c;
137        }
138      }
139      termAtt.setLength(upto);
140    }
141
142    //termAtt.setEmpty().append(treatWord(termAtt.toString(), false));
143
144    return true;
145  }
146
147  public static String treatWord(String str2, boolean isSpecies) {
148    char startLetter;
149
150    String temp = str2.toUpperCase();
151    // Do some selective replacement on the leading letter/s only:
152
153    if (temp.startsWith("AE")) {
154      temp = "E" + temp.substring(2);
155    } else if (temp.startsWith("CN")) {
156      temp = "N" + temp.substring(2);
157    } else if (temp.startsWith("CT")) {
158      temp = "T" + temp.substring(2);
159    } else if (temp.startsWith("CZ")) {
160      temp = "C" + temp.substring(2);
161    } else if (temp.startsWith("DJ")) {
162      temp = "J" + temp.substring(2);
163    } else if (temp.startsWith("EA")) {
164      temp = "E" + temp.substring(2);
165    } else if (temp.startsWith("EU")) {
166      temp = "U" + temp.substring(2);
167    } else if (temp.startsWith("GN")) {
168      temp = "N" + temp.substring(2);
169    } else if (temp.startsWith("KN")) {
170      temp = "N" + temp.substring(2);
171    } else if (temp.startsWith("MC")) {
172      temp = "MAC" + temp.substring(2);
173    } else if (temp.startsWith("MN")) {
174      temp = "N" + temp.substring(2);
175    } else if (temp.startsWith("OE")) {
176      temp = "E" + temp.substring(2);
177    } else if (temp.startsWith("QU")) {
178      temp = "Q" + temp.substring(2);
179    } else if (temp.startsWith("PS")) {
180      temp = "S" + temp.substring(2);
181    } else if (temp.startsWith("PT")) {
182      temp = "T" + temp.substring(2);
183    } else if (temp.startsWith("TS")) {
184      temp = "S" + temp.substring(2);
185    } else if (temp.startsWith("WR")) {
186      temp = "R" + temp.substring(2);
187    } else if (temp.startsWith("X")) {
188      temp = "Z" + temp.substring(2);
189    }
190    // Now keep the leading character, then do selected "soundalike" replacements. The
191    // following letters are equated: AE, OE, E, U, Y and I; IA and A are equated;
192    // K and C; Z and S; and H is dropped. Also, A and O are equated, MAC and MC are equated, and SC and S.
193    startLetter = temp.charAt(0); // quarantine the leading letter
194    temp = temp.substring(1); // snip off the leading letter
195    // now do the replacements
196    temp = temp.replaceAll("AE", "I");
197    temp = temp.replaceAll("IA", "A");
198    temp = temp.replaceAll("OE", "I");
199    temp = temp.replaceAll("OI", "A");
200    temp = temp.replaceAll("SC", "S");
201    temp = temp.replaceAll("E", "I");
202    temp = temp.replaceAll("O", "A");
203    temp = temp.replaceAll("U", "I");
204    temp = temp.replaceAll("Y", "I");
205    temp = temp.replaceAll("K", "C");
206    temp = temp.replaceAll("Z", "C");
207    temp = temp.replaceAll("H", "");
208    // add back the leading letter
209    temp = startLetter + temp;
210    // now drop any repeated characters (AA becomes A, BB or BBB becomes B, etc.)
211    temp = temp.replaceAll("(\\w)\\1+", "$1");
212
213    if (isSpecies) {
214      if (temp.endsWith("IS")) {
215        temp = temp.substring(0, temp.length() - 2) + "A";
216      } else if (temp.endsWith("IM")) {
217        temp = temp.substring(0, temp.length() - 2) + "A";
218      } else if (temp.endsWith("AS")) {
219        temp = temp.substring(0, temp.length() - 2) + "A";
220      }
221      //temp = temp.replaceAll("(\\w)\\1+", "$1");
222    }
223
224    return temp;
225  }
