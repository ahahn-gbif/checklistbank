007import java.util.Arrays;
008
009public class ModifiedDamerauLevenshtein implements StringSimilarity {
010        private final String s1;
011        private final String s2;
012        private final int pBlockLimit;
013
014        // this variable holds a shared array, to reduce heap use
015        // we use a little synchronisation to manage the space, but
016        // synchronisation is reasonably quick these days.
017
018        private static volatile int[] a_matrix = new int[64 * 64];
019        private static final Object mutex = new Object();
020
021        public ModifiedDamerauLevenshtein(String a, String b, int limit) {
022                this.s1 = a == null ? "" : a;
023                this.s2 = b == null ? "" : b;
024                this.pBlockLimit = limit;
025        }
026
027        public double getSimilarity() {
028                return DamerauLevenshtein.convertEditDistanceToSimilarity(getSimilarity(s1, s2, pBlockLimit), s1, s2);
029        }
030
031        public final static int getSimilarity(final String s1, final String s2, final int pBlockLimit) {
032                if (s1.equals(s2))
033                        return 0;
034                else if (s1.isEmpty() || s2.isEmpty())
035                        return Math.max(s1.length(), s2.length());
036                else if (s1.length() == 1 && s2.length() == 1) return 1;
037
038                //final String temp1;
039                //final String temp2;
040
041                final char[] t1;
042                final char[] t2;
043
044                {
045                        StringBuilder sb1 = new StringBuilder(s1);
046                        StringBuilder sb2 = new StringBuilder(s2);
047
048                        // these hold the index of the last character.
049                        int l1 = sb1.length()-1;
050                        int l2 = sb2.length()-1;
051
052                        while (l1>=0 && l2>=0 && sb1.charAt(0) == sb2.charAt(0)) {
053                                sb1.deleteCharAt(0);
054                                sb2.deleteCharAt(0);
055                                l1--;
056                                l2--;
057                        }
058
059                        while (l1>=0 && l2>=0 && sb1.charAt(l1) == sb2.charAt(l2)) {
060                                sb1.deleteCharAt(l1--);
061                                sb2.deleteCharAt(l2--);
062                        }
063
064                        l1++;
065                        l2++;
066                        if (l1 == 0 || l2 == 0)
067                                return Math.max(l1, l2);
068                        else if (l1 == 1 && l2 == 1) return 1;
069
070                        t1 = sb1.toString().toCharArray();
071                        t2 = sb2.toString().toCharArray();
072                }
073
074                final int temp1Len = t1.length;
075                final int temp2Len = t2.length;
076
077                // using a 1-dimensional array with bit fiddling to get to the elements
078                // saves about 12-15% off the running time.
079                // so, we replace matrix[a][b] with matrix[((a)<<6)|(b)]
080                // this limits us to 64-character words, which should be plenty. I hope.
081
082                final int[] matrix;
083                synchronized (mutex) {
084                        if (a_matrix == null) {
085                                matrix = new int[64 * 64];
086                        }
087                        else {
088                                matrix = a_matrix;
089                                a_matrix = null;
090                                Arrays.fill(matrix, 0);
091                        }
092                }
093
094                for (int i = 0; i <= temp1Len; i++) {
095                        matrix[i << 6] = i;
096                }
097
098                for (int i = 0; i <= temp2Len; i++) {
099                        matrix[i] = i;
100                }
101
102                for (int i = 1; i <= temp1Len; i++) {
103                        matrix[i << 6] = i;
104                        for (int j = 1; j <= temp2Len; j++) {
105                                int cost;
106                                if (t1[i - 1] == t2[j - 1]) {
107                                        cost = 0;
108                                }
109                                else {
110                                        cost = 1;
111                                }
112
113                                int temp_block_length = Math.max(//
114                                                Math.min(temp1Len / 2, //
115                                                                Math.min(temp2Len / 2, //
116                                                                                Math.max(pBlockLimit, 1))), //
117                                                1);
118
119                                while (temp_block_length >= 1) {
120
121                                        final int sub1 = i - ((temp_block_length * 2) - 1);
122                                        final int sub2 = j - (temp_block_length - 1);
123                                        final int sub3 = i - (temp_block_length - 1);
124                                        final int sub4 = j - ((temp_block_length * 2) - 1);
125
126                                        if (i >= (temp_block_length * 2) && j >= (temp_block_length * 2)
127                                                        && substreq(t1, sub1, t2, sub2, temp_block_length)
128                                                        && substreq(t1, sub3, t2, sub4, temp_block_length)) {
129
130                                                final int ins = matrix[((i) << 6) | (j - 1)] + 1;
131                                                final int del = matrix[((i - 1) << 6) | j] + 1;
132                                                final int tran = matrix[((i - (temp_block_length * 2)) << 6) | (j - (temp_block_length * 2))]
133                                                                + cost + (temp_block_length - 1);
134                                                matrix[(i << 6) | j] = minimum(ins, del, tran);
135
136                                                temp_block_length = 0;
137
138                                        }
139                                        else if (temp_block_length == 1) {
140
141                                                final int del = matrix[((i - 1) << 6) | j] + 1;
142                                                final int ins = matrix[(i << 6) | (j - 1)] + 1;
143                                                final int sub = matrix[((i - 1) << 6) | (j - 1)] + cost;
144
145                                                matrix[(i << 6) | j] = minimum(ins, del, sub);
146                                        }
147                                        temp_block_length--;
148                                }
149
150                        }
151                }
152
153                int ret = matrix[(temp1Len << 6) | temp2Len];
154
155                a_matrix = matrix; // does not need to be synchronized, it is atomic
156
157                return ret;
158
159        }
160
161        // The compiler inlines these, I think.
162
163        private final static int minimum(int d, int i, int s) {
164                return d < i ? (d < s ? d : s) : (i < s ? i : s);
165        }
166
167        private final static boolean substreq(char[] src1, int start1, char[] src2, int start2, int length) {
168                int at1 = start1 >= 0 ? start1 - 1 : src1.length + start1;
169                int at2 = start2 >= 0 ? start2 - 1 : src2.length + start2;
170                while (length-- > 0) {
171                        if (src1[at1++] != src2[at2++]) return false;
172                }
173                return true;
174        }
175
176}
