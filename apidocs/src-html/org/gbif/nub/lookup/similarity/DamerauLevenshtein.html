001package org.gbif.nub.lookup.similarity;
002
003public class DamerauLevenshtein implements StringSimilarity {
004
005  private String compOne;
006  private String compTwo;
007  private int[][] matrix;
008  private Boolean calculated = false;
009
010  public DamerauLevenshtein(String a, String b) {
011    if ((a.length() > 0 || !a.isEmpty()) || (b.length() > 0 || !b.isEmpty())) {
012      compOne = a;
013      compTwo = b;
014    }
015  }
016
017  public static double convertEditDistanceToSimilarity(int edits, String s1, String s2) {
018    int length = (s1.length() + s2.length()) / 2;
019    double sim = edits > length ? 0 : 100d * (1d - (double) edits / (double) length);
020    return sim;
021  }
022
023  public double getSimilarity() {
024    if (!calculated) setupMatrix();
025    return convertEditDistanceToSimilarity( matrix[compOne.length()][compTwo.length()], compOne, compTwo);
026  }
027
028  private void setupMatrix() {
029    int cost = -1;
030    int del, sub, ins;
031
032    matrix = new int[compOne.length() + 1][compTwo.length() + 1];
033
034    for (int i = 0; i <= compOne.length(); i++) {
035      matrix[i][0] = i;
036    }
037
038    for (int i = 0; i <= compTwo.length(); i++) {
039      matrix[0][i] = i;
040    }
041
042    for (int i = 1; i <= compOne.length(); i++) {
043      for (int j = 1; j <= compTwo.length(); j++) {
044        if (compOne.charAt(i - 1) == compTwo.charAt(j - 1)) {
045          cost = 0;
046        } else {
047          cost = 2;
048        }
049
050        del = matrix[i - 1][j] + 1;
051        ins = matrix[i][j - 1] + 1;
052        sub = matrix[i - 1][j - 1] + cost;
053
054        matrix[i][j] = minimum(del, ins, sub);
055
056        if ((i > 1) && (j > 1) && (compOne.charAt(i - 1) == compTwo.charAt(j - 2)) && (compOne.charAt(i - 2) == compTwo
057          .charAt(j - 1))) {
058          matrix[i][j] = minimum(matrix[i][j], matrix[i - 2][j - 2] + cost);
059        }
060      }
061    }
062
063    calculated = true;
064    //displayMatrix();
065  }
066
067  @Deprecated
068  private void displayMatrix() {
069    System.out.println("  " + compOne);
070    for (int y = 0; y <= compTwo.length(); y++) {
071      if (y - 1 < 0) {
072        System.out.print(" ");
073      } else {
074        System.out.print(compTwo.charAt(y - 1));
075      }
076      for (int x = 0; x <= compOne.length(); x++) {
077        System.out.print(matrix[x][y]);
078      }
079      System.out.println();
080    }
081  }
082
083  private int minimum(int d, int i, int s) {
084    int m = Integer.MAX_VALUE;
085
086    if (d < m) m = d;
087    if (i < m) m = i;
088    if (s < m) m = s;
089
090    return m;
091  }
092
093  private int minimum(int d, int t) {
094    int m = Integer.MAX_VALUE;
095
096    if (d < m) m = d;
097    if (t < m) m = t;
098
099    return m;
100  }
101}
