020import java.util.Arrays;
021
022/**
023 * Jaro-Winkler distance is a particularly good choice when comparing short strings of human entered data, such as names.
024 * This is due to it’s relative robustness against letter transpositions and it’s weighting of similarity toward the
025 * beginning of the string.
026 *
027 * @See <a href="http://en.wikipedia.org/wiki/Jaro%E2%80%93Winkler_distance">Jaro-Winkler distance on Wikipedia</a>.
028 *
029 * It is a variant of the Jaro distance metric which is made up of the average of three sub-calculations:
030 * <ul>
031 *   <li>The ratio of matching characters to the length of the first string</li>
032 *   <li>The ratio of matching characters to the length of the second string</li>
033 *   <li>The ratio of non-transpositions to the number matching of characters</li>
034 * </ul>
035 *
036 */
037public class JaroWinklerSimilarity implements StringSimilarity {
038
039  private final String s1;
040  private final String s2;
041
042  private float threshold = 0.7f;
043
044  public JaroWinklerSimilarity(String str1, String str2) {
045    this.s1 = str1;
046    this.s2 = str2;
047  }
048
049  private int[] matches(String s1, String s2) {
050    String max, min;
051    if (s1.length() > s2.length()) {
052      max = s1;
053      min = s2;
054    } else {
055      max = s2;
056      min = s1;
057    }
058    int range = Math.max(max.length() / 2 - 1, 0);
059    int[] matchIndexes = new int[min.length()];
060    Arrays.fill(matchIndexes, -1);
061    boolean[] matchFlags = new boolean[max.length()];
062    int matches = 0;
063    for (int mi = 0; mi < min.length(); mi++) {
064      char c1 = min.charAt(mi);
065      for (int xi = Math.max(mi - range, 0), xn = Math.min(mi + range + 1, max.length()); xi < xn; xi++) {
066        if (!matchFlags[xi] && c1 == max.charAt(xi)) {
067          matchIndexes[mi] = xi;
068          matchFlags[xi] = true;
069          matches++;
070          break;
071        }
072      }
073    }
074    char[] ms1 = new char[matches];
075    char[] ms2 = new char[matches];
076    for (int i = 0, si = 0; i < min.length(); i++) {
077      if (matchIndexes[i] != -1) {
078        ms1[si] = min.charAt(i);
079        si++;
080      }
081    }
082    for (int i = 0, si = 0; i < max.length(); i++) {
083      if (matchFlags[i]) {
084        ms2[si] = max.charAt(i);
085        si++;
086      }
087    }
088    int transpositions = 0;
089    for (int mi = 0; mi < ms1.length; mi++) {
090      if (ms1[mi] != ms2[mi]) {
091        transpositions++;
092      }
093    }
094    int prefix = 0;
095    for (int mi = 0; mi < min.length(); mi++) {
096      if (s1.charAt(mi) == s2.charAt(mi)) {
097        prefix++;
098      } else {
099        break;
100      }
101    }
102    return new int[] {matches, transpositions / 2, prefix, max.length()};
103  }
104
105  @Override
106  public double getSimilarity() {
107    int[] mtp = matches(s1, s2);
108    float m = (float) mtp[0];
109    if (m == 0) {
110      return 0;
111    }
112    float j = ((m / s1.length() + m / s2.length() + (m - mtp[1]) / m)) / 3;
113    float jw = j < getThreshold() ? j : j + Math.min(0.1f, 1f / mtp[3]) * mtp[2] * (1 - j);
114
115    //(100d - (100d-s) / Math.pow(Math.max(1,l-8), 0.2))
116
117    return 100d * jw;
118  }
119
120  /**
121   * Sets the threshold used to determine when Winkler bonus should be used.
122   * Set to a negative value to get the Jaro distance.
123   *
124   * @param threshold the new value of the threshold
125   */
126  public void setThreshold(float threshold) {
127    this.threshold = threshold;
128  }
129
130  /**
131   * Returns the current value of the threshold used for adding the Winkler bonus.
132   * The default value is 0.7.
133   *
134   * @return the current value of the threshold
135   */
136  public float getThreshold() {
137    return threshold;
138  }
139}
