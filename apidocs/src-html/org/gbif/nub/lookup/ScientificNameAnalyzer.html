001package org.gbif.nub.lookup;
002
003import org.gbif.nub.utils.Constants;
004
005import java.io.Reader;
006
007import org.apache.lucene.analysis.Analyzer;
008import org.apache.lucene.analysis.TokenStream;
009import org.apache.lucene.analysis.core.KeywordTokenizer;
010import org.apache.lucene.analysis.core.LowerCaseFilter;
011
012public class ScientificNameAnalyzer extends Analyzer {
013
014  public static final int BUFFER_SIZE = 1024;
015
016  public ScientificNameAnalyzer() {
017  }
018
019  @Override
020  protected TokenStreamComponents createComponents(final String fieldName, final Reader reader) {
021    KeywordTokenizer source = new KeywordTokenizer(reader, BUFFER_SIZE);
022
023    TokenStream result = new LowerCaseFilter(Constants.LUCENE_VERSION, source);
024    result = new ScientificNameSoundAlikeFilter(result);
025
026    return new TokenStreamComponents(source, result);
027  }
028
029}
