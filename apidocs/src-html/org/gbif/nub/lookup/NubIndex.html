046/**
047 * A memory based lucene index keeping the core attributes of a nub name usage.
048 * The index exposes matching methods that allow to select usages based on their nub key or do fuzzy matches
049 * on the canonical name alone.
050 *
051 * The index lies at the core of the nub matching service to preselect a list of potential good matches.
052 *
053 * For the entire nub with roughly 4.5 million usages this index requires 4GB of java memory.
054 */
055public class NubIndex implements ClassificationResolver {
056
057  protected static final String FIELD_ID = "id";
058  protected static final String FIELD_CANONICAL_NAME = "canonical";
059  private static final String FIELD_SCIENTIFIC_NAME = "sciname";
060  private static final String FIELD_RANK = "rank";
061  private static final String FIELD_SYNONYM = "syn";
062  private static final Map<Rank, String> HIGHER_RANK_ID_FIELD_MAP = ImmutableMap.<Rank, String>builder()
063    .put(Rank.KINGDOM, "k")
064    .put(Rank.PHYLUM, "p")
065    .put(Rank.CLASS, "c")
066    .put(Rank.ORDER, "o")
067    .put(Rank.FAMILY, "f")
068    .put(Rank.GENUS, "g")
069    .put(Rank.SPECIES, "s")
070    .build();
071  private static final Map<Rank, String> HIGHER_RANK_FIELD_MAP = ImmutableMap.<Rank, String>builder()
072    .put(Rank.KINGDOM, "kid")
073    .put(Rank.PHYLUM, "pid")
074    .put(Rank.CLASS, "cid")
075    .put(Rank.ORDER, "oid")
076    .put(Rank.FAMILY, "fid")
077    .put(Rank.GENUS, "gid")
078    .put(Rank.SPECIES, "sid")
079    .build();

084  private final Directory index;
085  private final IndexWriterConfig cfg;
086  private final Analyzer analyzer;
087  private final TrackingIndexWriter writer;
088  private SearcherManager searchManager;
089  private final ControlledRealTimeReopenThread<IndexSearcher> indexSearcherReopenThread;
090  private long reopenToken;
091
092  public NubIndex() throws IOException {
093    index = new RAMDirectory();
094    LOG.info("Init a new, empty nub index");
095    analyzer = new ScientificNameAnalyzer();
096    cfg = new IndexWriterConfig(org.gbif.nub.utils.Constants.LUCENE_VERSION, analyzer);
097    writer = new TrackingIndexWriter(new IndexWriter(index, cfg));
098    // creates initial index segments
099    writer.getIndexWriter().commit();
100    searchManager = new SearcherManager(writer.getIndexWriter(), false, null);
101    // Create the ControlledRealTimeReopenThread that reopens the index periodically having into
102    // account the changes made to the index and tracked by the TrackingIndexWriter instance
103    // The index is refreshed every 10 minutes when nobody is waiting
104    // and every 10 millis whenever is someone waiting (see search method)
105    indexSearcherReopenThread = new ControlledRealTimeReopenThread<IndexSearcher>(writer, searchManager, 600, 0.1);
106    indexSearcherReopenThread.start(); // start the refresher thread
107  }
108
109  public static NubIndex newNubIndex(UsageService usageService) throws IOException {
110    NubIndex idx = new NubIndex();
111    NubIndexBuilder nubLoader = new NubIndexBuilder(idx.writer, usageService, 4);
112    nubLoader.run();
113    idx.searchManager.maybeRefresh();
114    return idx;
115  }

117  public void addNameUsage(NameUsage usage) throws IOException {
118    reopenToken = writer.addDocument(toDoc(usage));
119  }

121  /**
122   * Adds the minimal usage information to the index required for a lookup.
123   * The returned match objects are lacking any classification information though, so make sure
124   * to use an external classification resolver with the nub matching service.
125   * Use the regular addNameUsage(usage) method for complete objects.
126   * @param u minimal usage
127   * @param canonical
128   * @throws IOException
129   */
130  public void addNameUsage(Usage u, String canonical) throws IOException {
131    reopenToken = writer.addDocument(toDoc(u.key, canonical, null, u.status.isSynonym(), u.rank, null, null));
132  }

134  public NameUsageMatch matchByUsageId(Integer usageID) {
135
136    Term tUsageID = new Term(NubIndex.FIELD_ID, usageID.toString());
137    Query q = new TermQuery(tUsageID);
138    try {
139      IndexSearcher searcher = obtainSearcher();
140      TopDocs docs = searcher.search(q, 3);
141      if (docs.totalHits > 0) {
142        Document doc = searcher.doc(docs.scoreDocs[0].doc);
143        NameUsageMatch match = fromDoc(doc);
144        match.setConfidence(100);
145        return match;
146
147      } else {
148        LOG.warn("No usage {} found in lucene index", usageID);
149      }
150    } catch (IOException e) {
151      LOG.error("Cannot load usage {} from lucene index", usageID, e.getMessage());
152    }
153
154    return null;
155  }

157  public List<NameUsageMatch> matchByName(String name, boolean fuzzySearch, int maxMatches) {
158    List<NameUsageMatch> results = Lists.newArrayList();
159
160    // use lucene analyzer to normalize input without using the full query parser
161    StringBuilder sb = new StringBuilder();
162    try {
163      TokenStream stream = new ScientificNameAnalyzer().tokenStream(NubIndex.FIELD_CANONICAL_NAME, new StringReader(name));
164      CharTermAttribute termAtt = stream.getAttribute(CharTermAttribute.class);
165      while (stream.incrementToken()) {
166        sb.append(termAtt.toString());
167      }
168    } catch (IOException e) {
169      LOG.error("An impossible error happened", e);
170    }
171
172    LOG.debug("Analyzed {} query \"{}\" becomes >>{}<<", fuzzySearch ? "fuzzy" : "straight", name, sb.toString());
173
174    // query needs to have at least 2 letters to match a real name
175    if (sb.length() < 2) {
176      return results;
177    }
178
179    final String analyzedName = sb.toString();
180    Term t = new Term(NubIndex.FIELD_CANONICAL_NAME, analyzedName);
181    Query q;
182    if (fuzzySearch) {
183      // allow 2 edits for names longer than 10 chars
184      q = new FuzzyQuery(t, analyzedName.length() > 10 ? 2 : 1, 1);
185    } else {
186      q = new TermQuery(t);
187    }
188    try {
189      IndexSearcher searcher = obtainSearcher();
190      TopDocs docs = searcher.search(q, maxMatches);
191      if (docs.totalHits > 0) {
192        for (ScoreDoc sdoc : docs.scoreDocs) {
193          NameUsageMatch match = fromDoc(searcher.doc(sdoc.doc));
194          if (name.equalsIgnoreCase(match.getCanonicalName())) {
195            match.setMatchType(NameUsageMatch.MatchType.EXACT);
196            results.add(match);
197          } else if (fuzzySearch) {
198            // even though we used a term query for straight matching the lucene analyzer has already normalized
199            // the name drastically. So we include these matches here only in case of fuzzy queries
200            match.setMatchType(NameUsageMatch.MatchType.FUZZY);
201            results.add(match);
202          }
203        }
204
205      } else {
206        LOG.debug("No {} match for name {}", fuzzySearch ? "fuzzy" : "straight", name);
207      }
208
209    } catch (IOException e) {
210      LOG.error("lucene search error", e);
211    }
212
213    return results;
214  }
215
216  private IndexSearcher obtainSearcher() {
217    try {
218      indexSearcherReopenThread.waitForGeneration(reopenToken);
219      return searchManager.acquire();
220
221    } catch (IOException e) {
222      LOG.error("Could not obtain lucene searcher", e);
223      throw new RuntimeException(e);
224
225    } catch (InterruptedException e) {
226      LOG.error("Could not obtain lucene searcher in time", e);
227      throw new RuntimeException(e);
228    }
229  }
230  /**
231   * Builds a NameUsageMatch instance from a lucene Document and populates all fields but the matching specifics
232   * i.e. confidence and matchType.
233   */
234  private static NameUsageMatch fromDoc(Document doc) {
235    NameUsageMatch u = new NameUsageMatch();
236    u.setUsageKey(toInt(doc.get(FIELD_ID)));
237
238    u.setScientificName(doc.get(FIELD_SCIENTIFIC_NAME));
239    u.setCanonicalName(doc.get(FIELD_CANONICAL_NAME));
240
241    // higher ranks
242    for (Rank r : HIGHER_RANK_FIELD_MAP.keySet()) {
243      ClassificationUtils.setHigherRank(u, r, doc.get(HIGHER_RANK_FIELD_MAP.get(r)), toInt(doc.get(HIGHER_RANK_ID_FIELD_MAP.get(r))));
244    }
245
246    Integer rankOrdinal = toInt(doc.get(FIELD_RANK));
247    if (rankOrdinal != null) {
248      u.setRank(Rank.values()[rankOrdinal]);
249    } else {
250      u.setRank(Rank.UNRANKED);
251    }
252
253    u.setSynonym(doc.get(FIELD_SYNONYM).equals("0") ? false : true);
254
255    return u;
256  }
257
258  protected static Document toDoc(NameUsage u) {
259    return toDoc(u.getKey(), u.getCanonicalName(), u.getScientificName(), u.isSynonym(), u.getRank(), u, u);
260  }
261
262  protected static Document toDoc(int key, String canonical, String sciname, boolean synonym, Rank rank,
263    LinneanClassification cl, LinneanClassificationKeys clKeys) {
264
265    Document doc = new Document();
266
267    doc.add(new StringField(FIELD_ID, String.valueOf(key), Field.Store.YES));
268
269    // analyzed name field - this is what we search upon
270    if (canonical != null) {
271      doc.add(new TextField(FIELD_CANONICAL_NAME, canonical, Field.Store.YES));
272    }
273
274    // store full name and classification only to return a full match object for hits
275    if (sciname != null) {
276      doc.add(new StoredField(FIELD_SCIENTIFIC_NAME, sciname));
277    }
278
279    // store ids as doc int values, not searchable
280    if (clKeys != null) {
281      for (Rank r : HIGHER_RANK_ID_FIELD_MAP.keySet()) {
282        Integer hrk = clKeys.getHigherRankKey(r);
283        if (hrk != null) {
284          doc.add(new StoredField(HIGHER_RANK_ID_FIELD_MAP.get(r), hrk));
285        }
286      }
287    }
288
289    // store higher ranks, not searchable
290    if (cl!= null) {
291      for (Rank r : HIGHER_RANK_FIELD_MAP.keySet()) {
292        String hr = cl.getHigherRank(r);
293        if (hr != null) {
294          doc.add(new StoredField(HIGHER_RANK_FIELD_MAP.get(r), hr));
295        }
296      }
297    }
298
299    // store synonym boolean as simple int with 0=false
300    doc.add(new StoredField(FIELD_SYNONYM, synonym ? 1 : 0));
301
302    // store rank if existing as ordinal int
303    // this lucene index is not persistent, so not risk in changing ordinal numbers
304    if (rank != null) {
305      doc.add(new StoredField(FIELD_RANK, rank.ordinal()));
306    }
307
308    return doc;
309  }
310
311  private static Integer toInt(String val) {
312    return Strings.isNullOrEmpty(val) ? null : Integer.valueOf(val);
313  }
314
315  @Override
316  public LinneanClassification getClassification(int usageKey) {
317    return matchByUsageId(usageKey);
318  }
