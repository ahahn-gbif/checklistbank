001package org.gbif.nub.build;
002
003import org.gbif.checklistbank.service.mybatis.guice.ChecklistBankServiceMyBatisModule;
004import org.gbif.checklistbank.ws.client.guice.ChecklistBankWsClientModule;
005import org.gbif.nameparser.NameParser;
006import org.gbif.nub.lookup.HigherTaxaLookup;
007import org.gbif.registry.ws.client.guice.RegistryWsClientModule;
008import org.gbif.ws.client.guice.AnonymousAuthModule;
009
010import java.io.File;
011import java.io.IOException;
012import java.util.Properties;
013
014import com.google.inject.PrivateModule;
015import com.google.inject.Provides;
016import com.google.inject.Scopes;
017import com.google.inject.Singleton;
018import org.apache.commons.io.FileUtils;
019import org.slf4j.Logger;
020import org.slf4j.LoggerFactory;
021
022/**
023 * Guice module setting up all dependencies to expose the NubGenerate.
024 */
025public class NubBuildModule extends PrivateModule {
026  private static final Logger LOG = LoggerFactory.getLogger(NubBuildModule.class);
027  private Properties properties;
028
029  public NubBuildModule(Properties properties) {
030    this.properties = properties;
031  }
032
033  @Override
034  protected void configure() {
035    install(new ChecklistBankServiceMyBatisModule(properties));
036    install(new ChecklistBankWsClientModule(properties, false, false, true));
037    install(new RegistryWsClientModule(properties));
038    // required by registry client
039    install(new AnonymousAuthModule());
040    bind(NubGenerator.class).in(Scopes.NO_SCOPE);
041    expose(NubGenerator.class);
042  }
043
044  @Provides
045  @Singleton
046  public static HigherTaxaLookup provideSynonyms() {
047    HigherTaxaLookup synonyms = new HigherTaxaLookup();
048    LOG.info("Start loading synonym dictionaries from rs.gbif.org ...");
049    synonyms.loadOnlineDicts();
050    LOG.info("Online synonym dictionaries loaded");
051    return synonyms;
052  }
053
054  @Provides
055  @Singleton
056  public static NameParser provideParser() {
057    NameParser parser = new NameParser();
058    return parser;
059  }
060
061  @Provides
062  public static File dataDir() throws IOException {
063    //TODO: make this configurable
064    File dir = new File("/tmp/nub-data");
065    if (dir.exists()) {
066      LOG.warn("Nub writer dir exists. All data will be removed");
067      FileUtils.deleteDirectory(dir);
068    }
069    // create new, empty dir
070    FileUtils.forceMkdir(dir);
071    LOG.info("New nub data directory created at {}", dir.getAbsoluteFile());
072    return dir;
073  }
074
075
076}
