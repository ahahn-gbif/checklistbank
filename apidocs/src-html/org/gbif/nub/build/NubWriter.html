001package org.gbif.nub.build;
002
003import org.gbif.api.model.Constants;
004import org.gbif.checklistbank.service.DatasetImportService;
005import org.gbif.file.TabWriter;
006import org.gbif.utils.file.FileUtils;
007
008import java.io.File;
009import java.io.FileNotFoundException;
010import java.io.FileOutputStream;
011import java.io.IOException;
012import java.util.Map;
013import java.util.Set;
014import java.util.UUID;
015
016import com.google.common.base.Preconditions;
017import com.google.common.collect.Maps;
018import com.google.common.collect.Sets;
019import com.google.inject.Inject;
020import org.apache.commons.io.LineIterator;
021import org.slf4j.Logger;
022import org.slf4j.LoggerFactory;
023
024/**
025 * Tools that generate all files for a new nub build.
026 * Should be able to share this class at some point with importing routines when we get there.
027 * The files generated are ready to be quickly inserted into teh clb database.
028 */
029public class NubWriter {
030  private static final Logger LOG = LoggerFactory.getLogger(NubWriter.class);
031  private final File dir;
032  private final DatasetImportService importService;
033  private TabWriter nubRelWriter;
034  private UUID currentDatasetKey;
035  private Set<UUID> processedDatasets = Sets.newHashSet();
036
037  @Inject
038  public NubWriter(DatasetImportService importService, File reportDir) {
039    this.importService = importService;
040    this.dir = reportDir;
041  }
042
043  /**
044   * Starts a new set of source dataset files and closes the last one if it was still open.
045   */
046  public void startDataset(UUID datasetKey) {
047    currentDatasetKey = datasetKey;
048    closeQuietly(nubRelWriter);
049    nubRelWriter = open(nubRelFile(datasetKey));
050    processedDatasets.add(datasetKey);
051  }
052
053  private File nubRelFile(UUID datasetKey) {
054    return new File(dir, "nubkeys-"+datasetKey.toString()+".tab");
055  }
056
057  public void close() {
058    closeQuietly(nubRelWriter);
059  }
060
061  public void closeQuietly(TabWriter w) {
062    if (w != null) {
063      try {
064        w.close();
065      } catch (IOException e) {
066        LOG.warn("Cannot close tab writer", e);
067      }
068    }
069  }
070
071  public void mapSource(int srcUsageKey, int nubUsageKey) {
072    Preconditions.checkNotNull("no dataset yet started", nubRelWriter);
073    try {
074      String[] row = {String.valueOf(srcUsageKey), String.valueOf(nubUsageKey)};
075      nubRelWriter.write(row);
076    } catch (IOException e) {
077      throw new IllegalStateException("Failed to write nub key map", e);
078    }
079  }
080
081  /**
082   * Deletes any previous nub from checklistbank and inserts a new one based on the nub cache.
083   */
084  public void insertNubData(ChecklistCache nubCache) {
085    LOG.info("Write nub data to files in directory {}", dir);
086
087    // delete old nub
088    importService.deleteDataset(Constants.NUB_DATASET_KEY);
089
090    // insert new nub usage
091    importService.insertUsages(Constants.NUB_DATASET_KEY, nubCache.iterate());
092  }
093
094  private TabWriter open(File f) {
095    try {
096      return new TabWriter(new FileOutputStream(f));
097    } catch (FileNotFoundException e) {
098      throw new IllegalStateException("Cannot open tab writer for file " + f.getAbsolutePath(), e);
099    }
100  }
101
102  /**
103   * Uses the mapped source files to update all nub relations for those datasets.
104   */
105  public void updateRelNub() {
106    for (UUID datasetKey : processedDatasets) {
107      try {
108        Map<Integer, Integer> relations = readRelFile(nubRelFile(datasetKey));
109        LOG.info("Updating dataset {} with {} nub relations", datasetKey, relations.size());
110        importService.insertNubRelations(datasetKey, relations);
111      } catch (FileNotFoundException e) {
112        LOG.error("Cannot read nub relation file for dataset {}. Warning, dataset relations are unchanged!", datasetKey);
113      }
114    }
115  }
116
117  private Map<Integer, Integer> readRelFile(File f) throws FileNotFoundException {
118    Map<Integer, Integer> rels = Maps.newHashMap();
119    LineIterator lines = FileUtils.getLineIterator(FileUtils.getInputStream(f));
120    while (lines.hasNext()) {
121      String[] parts = FileUtils.TAB_DELIMITED.split(lines.nextLine().trim());
122      rels.put(Integer.parseInt(parts[0]), Integer.parseInt(parts[1]));
123    }
124    return rels;
125  }
126}
