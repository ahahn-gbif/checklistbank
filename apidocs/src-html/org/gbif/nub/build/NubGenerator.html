133  /**
134   * Builds a new, complete nub checklist as a simple dwc archive.
135   * The nub build will make use of the currently indexed checklists and not run any new imports or updates.
136   *
137   * The nub is generated by adding one checklist at a time to the nub.
138   * The order of the checklists is determined by the integer clb:nubPriority machine tag, with smallest numbers
139   * having highest priority.
140   *
141   * The higher taxonomy above family level is ignored by default unless the checklist is given in the
142   * higherTaxonomyChecklists parameter.
143   *
144   * @param higherTaxonomyChecklists set of checklists where the higher taxonomy is not ignored but incorporated
145   *                                 into the nub
146   */
147  public ChecklistCache build(UUID ... higherTaxonomyChecklists) throws IOException {
148    LOG.info("Start building a new taxonomic backbone");
149
150    LOG.info("Use the higher classification from checklists:\n " + Joiner.on("\n ").join(higherTaxonomyChecklists));
151    for (UUID c : higherTaxonomyChecklists) {
152      this.higherTaxonomyChecklists.add(c);
153    }
154
155    nextNubKey = uService.maxUsageKey(Constants.NUB_DATASET_KEY) + 1;
156    LOG.info("New nub usages will start with key {}", nextNubKey);
157    LOG.info("Adding kingdoms");
158    addKingdoms();
159
160    for (Dataset d : listChecklistsFaked()) {
161      addChecklist(d, !this.higherTaxonomyChecklists.contains(d.getKey()));
162    }
163
164    reporter.close();
165
166    writer.insertNubData(nubCache);
167    writer.updateRelNub();
168
169    return nubCache;
170  }

172  private void loadSrcCache(UUID datasetKey) {
173    srcCache.reset();
174
175    // load names
176    LOG.info("Loading names into cache for checklist {}", currDataset.getTitle());
177    PagingRequest page = new PagingRequest(0, 10000);
178    PagingResponse<ParsedName> respNames;
179    do {
180      respNames = nameService.listNames(datasetKey, page);
181      for (ParsedName n : respNames.getResults()) {
182        srcCache.add(n);
183      }
184      page.nextPage();
185    }
186    while (!respNames.isEndOfRecords());
187
188    // load usages
189    LOG.info("Loading usages into cache for checklist {}", currDataset.getTitle());
190    page = new PagingRequest(0, 10000);
191    PagingResponse<Usage> respUsages;
192    do {
193      respUsages = uService.list(datasetKey, page);
194      for (Usage u : respUsages.getResults()) {
195        srcCache.add(u);
196      }
197      page.nextPage();
198    }
199    while (!respUsages.isEndOfRecords());
200
201    // verify cache consistency
202    srcCache.checkConsistency();
203  }

205  private void addChecklist(Dataset d, final boolean ignoreHigherTaxa) {
206    LOG.info("Adding checklist {}, {} higher taxa", d.getTitle(), ignoreHigherTaxa ? "ignoring" : "using");
207    currDataset = d;
208    currReport = reporter.report(d.getKey());
209    writer.startDataset(d.getKey());
210
211    // TODO: get real currSourceDatasetKey value based on citation table
212    sourceDatasetKeys.put(currSourceDatasetIntKey++, d.getKey());
213    loadSrcCache(d.getKey());
214    processed = Maps.newHashMap();
215
216    Iterator<Usage> iter = srcCache.iterate();
217    int counter = 1;
218    while (iter.hasNext()) {
219      if (counter % 10000 == 0){
220        LOG.info("Processing {}th usage", counter);
221      }
222      Usage u = iter.next();
223      LOG.debug("Process {} usage {}", u.rank, u.key);
224      processSource(u, ignoreHigherTaxa);
225      counter++;
226    }
227  }

229  /**
230   *
231   * @param src
232   * @param ignoreHigherTaxa
233   * @return the nub usage for this source usage or the next highest parent. It is
234   */
235  private NubUsage processSource(Usage src, final boolean ignoreHigherTaxa) {
236    if (processed.containsKey(src.key)) {
237      return new NubUsage(nubCache.get(processed.get(src.key)), true, NameUsageMatch.MatchType.EXACT, 100);
238    }
239
240    // get some more details
241    ParsedName srcName = srcCache.getName(src.nameKey);
242    Rank rank = getNubRank(src, srcName);
243
244    // match to existing nub
245    NubUsage nub = findNubUsage(srcName.getScientificName(), src.rank, src.key);
246
247    // IGNORE source for various reasons
248    final boolean ignore = ignoreSrc(src, srcName, rank);
249    if (ignore) {
250      currReport.incrementIgnored();
251      if (nub != null) {
252        // we should ignore this source usage and there is a matching nub usage already.
253        // Return that and stop
254        return nub;
255      } else {
256        // otherwise return the closest nub usage from the parental hierarchy
257        return getParentNubUsage(src, ignoreHigherTaxa);
258      }
259    }
260
261
262    // If we reach here, incorporate source into nub!
263    // get parental nub usage first to attach this usage
264    final boolean isIgnoredHigherTaxon = ignoreHigherTaxa(src, rank, ignoreHigherTaxa);
265    NubUsage parentNub = getParentNubUsage(src, ignoreHigherTaxa);
266
267    if (nub != null) {
268      updateNub(nub, src, srcName, parentNub, isIgnoredHigherTaxon);
269
270    } else if (isIgnoredHigherTaxon) {
271      currReport.incrementIgnored();
272      return parentNub;
273
274    } else {
275      // only create new higher taxa if it was requested
276      nub = createNub(src, srcName, parentNub, rank);
277    }
278
279    return nub;
280  }

282  private boolean ignoreHigherTaxa(Usage src, Rank nubRank, final boolean ignoreHigherTaxa) {
283      // ignore higher ranks if not asked for
284    if (ignoreHigherTaxa && (nubRank.isSuprageneric() && Rank.FAMILY != nubRank)) {
285      LOG.debug("Ignore {} taxon {}, higher ranks ignored", nubRank, CacheUtils.nameOf(src, srcCache));
286      return true;
287    }
288    return false;
289  }

291  private boolean ignoreSrc(Usage src, ParsedName srcName, Rank nubRank) {
292    // ignore usages without a rank
293    if (Rank.UNRANKED == nubRank) {
294      LOG.debug("Ignore usage {} of rank {}", srcName.getScientificName(), nubRank);
295      return true;
296    }
297
298    // completely ignore blacklisted, informal (i.e. spec. or cf.) names or incertae sedis names
299    if (NameType.BLACKLISTED == srcName.getType() || NameType.INFORMAL == srcName.getType()) {
300      LOG.debug("Ignore blacklisted or informal usage {} {}", src.key, srcName.getScientificName());
301      return true;
302    }
303
304    if (INCERTAE_SEDIS_NAME_ID == srcName.getKey()) {
305      LOG.debug("Ignore incertae sedis usage {} {}", src.key, srcName.getScientificName());
306      return true;
307    }
308
309    // ignore all cultivars and strains!
310    // http://dev.gbif.org/issues/browse/CLB-117
311    if (Rank.CULTIVAR == nubRank || Rank.CULTIVAR_GROUP == nubRank || Rank.STRAIN == nubRank){
312      LOG.debug("Ignore {} name {}, usage {}", nubRank, srcName, src.key);
313    }
314
315    if (!rankMatchesName(nubRank, srcName)) {
316      LOG.debug("Ignore usage {} with name {} contradicting rank {}", src.key, srcName.getScientificName(), nubRank);
317      return true;
318    }
319
320    return false;
321  }

323  private Rank getNubRank(Usage src, ParsedName name) {
324    Rank nubRank = Rank.inferRank(name.getGenusOrAbove(), name.getInfraGeneric(), name.getSpecificEpithet(), name.getRankMarker(), name.getInfraSpecificEpithet());
325    if (src.rank != null && !src.rank.isUncomparable()) {
326      nubRank = src.rank;
327      if (nubRank != null && !nubRank.isUncomparable() && !src.rank.equals(nubRank)) {
328        // warn if they are not the same
329        LOG.warn("Inferred rank {} is different from given rank {} for usage {}", nubRank, src.rank, src.key);
330      }
331    }
332
333    // For CoL convert generic infraspecific rank to subspecies - many names in CoL have this rank
334    if (COL_KEY.equals(currDataset.getKey()) && Rank.INFRASPECIFIC_NAME == src.rank){
335      nubRank = Rank.SUBSPECIES;
336    }
337
338    return nubRank;
339  }

341  protected static boolean rankMatchesName(Rank rank, ParsedName name) {
342    // skip assertion for hybrids or virus names which dont have a parsed name.
343    if (!name.getType().isParsable()){
344      return true;
345    }
346
347    if (Rank.SPECIES == rank){
348      // we require a genus & specific epithet and no more
349      if (!Strings.isNullOrEmpty(name.getGenusOrAbove()) && !Strings.isNullOrEmpty(name.getSpecificEpithet()) && Strings.isNullOrEmpty(name.getInfraSpecificEpithet())){
350        return true;
351      }
352    } else if (rank.isInfraspecific()){
353      // we require a genus and both epitheta
354      if (!Strings.isNullOrEmpty(name.getGenusOrAbove()) && !Strings.isNullOrEmpty(name.getSpecificEpithet()) && !Strings.isNullOrEmpty(name.getInfraSpecificEpithet())){
355        if (rank.isUncomparable()){
356          return true;
357        }
358        // check if rank marker indicates a specific rank
359        Rank markerRank = Rank.inferRank(name.getRankMarker());
360        if (markerRank == null || markerRank==rank){
361          return true;
362        }
363      }
364
365    } else if (Rank.GENUS == rank || rank.isSuprageneric()) {
366      // avoid epitheta
367      if (!Strings.isNullOrEmpty(name.getGenusOrAbove()) && Strings.isNullOrEmpty(name.getSpecificEpithet()) && Strings.isNullOrEmpty(name.getInfraSpecificEpithet())){
368        return true;
369      }
370
371    }
372    return false;
373  }

375  private NubUsage getParentNubUsage(Usage child, boolean ignoreHigherTaxa) {
376    if (child.hasParent()) {
377      Usage parentSrc = srcCache.get(child.parentKey);
378      NubUsage nub = processSource(parentSrc, ignoreHigherTaxa);
379      nub.directParentMatch = false;
380      return nub;
381    } else {
382      return incertaeSedis;
383    }

386  /**
387   * Try to match an existing nub usage using the internal matching service
388   * running on the currently created nub usages.
389   *
390   * @param srcUsageKey a source usage key used create a LinneanClassification for comparison
391   * @return the corresponding nub usage or null. Never a MatchType.NONE
392   */
393  private NubUsage findNubUsage(String scientificName, Rank rank, int srcUsageKey) {
394    LinneanClassification cl = srcCache.getClassification(srcUsageKey);
395    NameUsageMatch match = tightMatch(internalMatcher.match(scientificName, rank, cl, true, false));
396    if (match != null) {
397      return new NubUsage(nubCache.get(match.getUsageKey()), true, match.getMatchType(), match.getConfidence());
398    }
399    return null;
400  }
401
402  /**
403   * @return the given match or null if the match was not good enough for nub building purposes.
404   */
405  private NameUsageMatch tightMatch(NameUsageMatch match) {
406    if (NameUsageMatch.MatchType.NONE != match.getMatchType()) {
407      // for monomials only allow exact matches for now - fuzzy ones are sometimes wrong and would result
408      // in false merged taxa
409      if (NameUsageMatch.MatchType.EXACT == match.getMatchType() || (match.getRank().isSpeciesOrBelow() && match.getConfidence() > 90)) {
410        return match;
411      }
412    }
413    return null;
414  }

416  private NubUsage createNub(Usage src, ParsedName srcName, NubUsage nubParent, Rank nubRank) {
417    if (src.status != null && src.status.isSynonym() && !nubParent.directParentMatch) {
418      LOG.info("Synonym {} to be created based on usage {}, but accepted parent is not in the nub", srcName.getScientificName(), src.key);
419    }
420
421    // potentially create missing genus & species usages
422    if (nubRank.isSpeciesOrBelow()) {
423      if (nubParent.usage.rank.isSuprageneric()) {
424        nubParent = createOrFindImplicitUsage(srcName.getGenusOrAbove(), Rank.GENUS, nubParent.usage.key, Origin.IMPLICIT_NAME);
425      }
426      if (nubRank.isInfraspecific() && nubParent.usage.rank != Rank.SPECIES) {
427        nubParent = createOrFindImplicitUsage(srcName.canonicalSpeciesName(), Rank.SPECIES, nubParent.usage.key, Origin.IMPLICIT_NAME);
428      }
429    }
430
431    // create an autonym (BotanicalCode) / nominotypical subspecies (ZooCode) ?
432    // http://en.wikipedia.org/wiki/Autonym_(botany)
433    // http://en.wikipedia.org/wiki/Subspecies#Nominotypical_subspecies_and_subspecies_autonyms
434    //
435    // all autonyms created here for subspecies do NOT include a rank marker as the ZoologicalCode specifies.
436    // TODO: for botanical, fungi and bacterial names this should be changed at the very end
437    if (nubRank.isInfraspecific() && !srcName.isAutonym()) {
438      final String originalEpithet = srcName.getInfraSpecificEpithet();
439      srcName.setInfraSpecificEpithet(srcName.getSpecificEpithet());
440      srcName.setRank(nubRank);
441      createOrFindImplicitUsage(srcName.canonicalNameWithMarker(), nubRank, nubParent.usage.key, Origin.AUTONYM);
442      srcName.setInfraSpecificEpithet(originalEpithet);
443    }
444
445
446    // create requested usage
447    NubUsage nub = new NubUsage();
448    nub.usage.parentKey = nubParent.usage.key;
449    nub.usage.nameKey = srcName.getKey();
450    nub.usage.status = src.status == null ? TaxonomicStatus.ACCEPTED : src.status;
451    nub.usage.rank = nubRank;
452    nub.usage.accordingToKey = 0;
453    nub.usage.origin = Origin.SOURCE;
454    nub.usage.sourceKey = src.key;
455    nub.usage.namePublishedInKey = src.namePublishedInKey;
456
457    nub.usage.key = getNubUsageId(nub, srcName);
458
459    currReport.incrementCreated();
460    add(srcName);
461    add(nub.usage);
462
463    LOG.debug("Nub {} {} created: {}", nub.usage.rank, nub.usage.key, srcName.getScientificName());
464    return nub;
465  }

467  private NubUsage createOrFindImplicitUsage(String implicitName, Rank rank, int parentKey, Origin origin) {
468
469    // see if a nub usage with this name already exists
470    NubUsage nub = findNubUsage(implicitName, rank, parentKey);
471
472    if (nub == null) {
473      // create new usage
474      ParsedName pn = parseAndAdd(implicitName);
475      nub = new NubUsage();
476      nub.usage.parentKey = parentKey;
477      nub.usage.nameKey = pn.getKey();
478      nub.usage.status = TaxonomicStatus.ACCEPTED;
479      nub.usage.rank = rank;
480      nub.usage.accordingToKey = CITATION_GBIF_NUB;
481      nub.usage.origin = origin;
482
483      nub.usage.key = getNubUsageId(nub, pn);
484
485      LOG.debug("Implicit nub {} {} created: {}", nub.usage.rank, nub.usage.key, implicitName);
486      currReport.incrementCreated();
487      add(nub.usage);
488    }
489
490    return nub;
491  }

508      }

525  /**
526   * Verifies that the old parent nub key also appears in the new, more detailed classification.
527   * @param oldParentKey the old nub parent key
528   * @param newParent  the proposed new nub parent
529   * @return true if the new parent can be used instead of the old parent key without altering the taxonomic concept
530   */
531  private boolean classificationUpdateable(int oldParentKey, Usage newParent) {
532    if (oldParentKey == newParent.key) {
533      return false;
534    }
535    if (oldParentKey == incertaeSedis.usage.key) {
536      return true;
537    }
538
539    // if the old parentKey exists in the hierarchy of the new one, its all good
540    Usage p = newParent;
541    do {
542      p = nubCache.get(p.parentKey);
543      if (p.parentKey == oldParentKey) {
544        return true;
545      }
546    } while (p.parentKey > 0);
547
548    return false;
549  }
550  /**
551   * Finds an existing nub id or assigns a new identifier that has never been used before in the nub.
552   * Existing ids are detected by matching the new nub usage to the old nub via the external matching webservice.
553   * @return the nub id to be used for this nub usage
554   */
555  private int getNubUsageId(NubUsage nub, ParsedName nubName) {
556    if (previousNubMatcher != null) {
557      LinneanClassification lc = nubCache.getClassification(nub.usage.key);
558      NameUsageMatch match = tightMatch(previousNubMatcher.match(nubName.getScientificName(), nub.usage.rank, lc, true, false));
559      if (match != null) {
560        // verify that we have not used this id before in the nub build
561        Usage otherNub = nubCache.get(match.getUsageKey());
562        if (otherNub == null) {
563          return match.getUsageKey();
564        } else {
565          LOG.warn("The same previous nub id {} matches 2 usages in the new nub: {} and {} with new id {}", CacheUtils.nameOf(otherNub, nubCache), nubName.getScientificName(), nextNubKey);
566        }
567      }
568    }
569    currReport.newId(nextNubKey, nubName.getScientificName());
570    return nextNubKey++;
571  }

585  /**
586   * @return list of checklists used to build the nub. Order and selection is based on the clb:nubPriority tag
587   */
588  private List<Dataset> listChecklists() {
589    DatasetSearchRequest req = new DatasetSearchRequest();
590    req.addTypeFilter(DatasetType.CHECKLIST);
591    req.setLimit(1000);
592
593    SortedMap<String, Dataset> nubSources = Maps.newTreeMap();
594    for (DatasetSearchResult res : searchService.search(req).getResults()) {
595      Dataset d = datasetService.get(res.getKey());
596      for (MachineTag mt : d.getMachineTags()) {
597        if (mt.getNamespace().equals(MT_NAMESPACE) && mt.getName().equals(MT_NUB_PRIORITY_NAME)) {
598          nubSources.put(mt.getValue(), d);
599        }
600      }
601    }
602    LOG.info("Found {} tagged checklists in registry to build a new nub from", nubSources.size());
603
604    return Lists.newArrayList(nubSources.values());
605  }

607  private void addKingdoms() {
608    for (Kingdom k : Kingdom.values()){
609      ParsedName pn = new ParsedName();
610      pn.setScientificName(k.scientificName());
611      pn.setGenusOrAbove(k.scientificName());
612      pn.setKey(k.clbNameID());
613      add(pn);
614
615      Usage u = new Usage();
616      u.key=k.nubUsageID();
617      u.status = TaxonomicStatus.ACCEPTED;
618      u.rank=Rank.KINGDOM;
619      u.accordingToKey = CITATION_GBIF_NUB;
620      u.nameKey = k.clbNameID();
621      u.origin = Origin.DENORMED_CLASSIFICATION;
622      add(u);
623      // remember incertae sedis kingdom for last resorting parent in the nub
624      if (Kingdom.INCERTAE_SEDIS == k) {
625        incertaeSedis = new NubUsage(u, false, NameUsageMatch.MatchType.EXACT, 100);
626      }
627    }
628  }

640  private void add(Usage u) {
641    if (u.sourceKey > 0){
642      processed.put(u.sourceKey, u.key);
643    }
644    // insert to nub cache
645    nubCache.checkConsistency(u);
646    nubCache.add(u);
647    // update internal matcher.
648    // The matcher uses the nubCache as the classification resolver, so only the usage name is important here!
649    try {
650      index.addNameUsage(u, CacheUtils.canonicalNameOf(u, nubCache));
651    } catch (IOException e) {
652      throw new IllegalStateException("Failed to add usage to internal index", e);
653    }
654
655    // can be null when adding kingdoms
656    if (currReport != null) {
657      writer.mapSource(u.sourceKey, u.key);
658      currReport.incrementCreated();
659    }
660  }
