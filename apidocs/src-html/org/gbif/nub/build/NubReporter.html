001package org.gbif.nub.build;
002
003import org.gbif.file.TabWriter;
004
005import java.io.File;
006import java.io.FileNotFoundException;
007import java.io.FileOutputStream;
008import java.io.IOException;
009import java.io.Writer;
010import java.util.Map;
011import java.util.UUID;
012
013import com.google.common.collect.Maps;
014import com.google.inject.Inject;
015import org.apache.commons.io.FileUtils;
016import org.slf4j.Logger;
017import org.slf4j.LoggerFactory;
018
019public class NubReporter {
020  private static final Logger LOG = LoggerFactory.getLogger(NubReporter.class);
021
022  private final File dir;
023  // we keep an additional counter for all created nub usages so its quickly accesible
024  private int createdTotal = 0;
025  private Map<UUID, DatasetReport> reports = Maps.newHashMap();
026  private final TabWriter newIdWriter;
027
028  @Inject
029  public NubReporter(File reportDir) throws FileNotFoundException {
030    this.dir = reportDir;
031    if (!dir.exists()) {
032      try {
033        FileUtils.forceMkdir(dir);
034      } catch (IOException e) {
035        throw new IllegalArgumentException("Cannot create reporting dir", e);
036      }
037    }
038    newIdWriter = new TabWriter(new FileOutputStream(new File(reportDir, "new-ids.tab")));
039  }
040
041  public int getCreatedTotal() {
042    return createdTotal;
043  }
044
045  public class DatasetReport {
046    private int ignored;
047    private int created; // number of nub usages// based on a usage from
048    private int updated;
049    private int newIds; // number of nub usages with new ids, included in created
050
051    public int getTotal() {
052      return ignored + created + updated;
053    }
054
055    /**
056     * @return number of usages ignored during processing.
057     */
058    public int getIgnored() {
059      return ignored;
060    }
061
062    public void incrementIgnored() {
063      ignored++;
064    }
065
066    /**
067     * @return number of usages that were responsible for creating new nub usages.
068     */
069    public int getCreated() {
070      return created;
071    }
072
073    public void incrementCreated() {
074      created++;
075      createdTotal++;
076      if (createdTotal % 1000 == 0) {
077        LOG.info("{} total nub usages created", createdTotal);
078      }
079    }
080
081    /**
082     * @return number of new nub usages with new identifiers that did not exist in the previous nub
083     */
084    public int getNewIds() {
085      return newIds;
086    }
087
088    public void newId(int id, String scientificName) {
089      newIds++;
090      try {
091        String[] row = {String.valueOf(id), scientificName};
092        newIdWriter.write(row);
093      } catch (IOException e) {
094        throw new IllegalStateException(e);
095      }
096    }
097
098    /**
099     * @return number of usages that were already in the nub and updated the nub.
100     */
101    public int getUpdated() {
102      return updated;
103    }
104
105    public void incrementUpdated() {
106      updated++;
107    }
108  }
109
110  public DatasetReport report(UUID key) {
111    if (!reports.containsKey(key)) {
112      reports.put(key, new DatasetReport());
113    }
114    return reports.get(key);
115  }
116
117  public void close() throws IOException {
118    newIdWriter.close();
119  }
120
121  public void writeReport(Writer writer) throws IOException {
122    writer.write("NUB BUILDING REPORT");
123
124    for (Map.Entry<UUID, DatasetReport> entry : reports.entrySet()) {
125      DatasetReport r = entry.getValue();
126
127      writer.write("\n\nDataset ");
128      writer.write(entry.getKey().toString());
129      writer.write("\n  Total processed:" + r.getTotal());
130      writer.write("\n  Nub usages created:" + r.getCreated());
131      writer.write("\n  Nub usages with new ids:" + r.getNewIds());
132      writer.write("\n  Nub usages updated:" + r.getUpdated());
133      writer.write("\n  Usages ignored:" + r.getIgnored());
134    }
135  }
136}
