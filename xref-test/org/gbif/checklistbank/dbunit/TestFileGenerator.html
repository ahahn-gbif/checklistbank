32    public static void main(String[] args) throws Exception {
33      // database connection
34      Class<?> driverClass = Class.forName("org.postgresql.Driver");
35      // replace with your local db to be exported
36      Connection jdbcConnection = DriverManager.getConnection("jdbc:postgresql://localhost/clb-test", "postgres", "");
37      IDatabaseConnection connection = new DatabaseConnection(jdbcConnection);
38      for (Map.Entry<String, Object> prop : DatabaseDrivenChecklistBankTestRule.DB_UNIT_CLB_PROPERTIES.entrySet()) {
39        connection.getConfig().setProperty(prop.getKey(), prop.getValue());
40      }
41  
42      // full database export, automatically ordering tables by foreign key constraints
43      ITableFilter filter = new DatabaseSequenceFilter(connection);
44      IDataSet fullDataSet = new FilteredDataSet(filter, connection.createDataSet());
45      File outf = new File("checklistbank-mybatis-service/src/test/resources/dbunit/squirrels-full.xml");
46      FlatXmlDataSet.write(fullDataSet, new FileOutputStream(outf));
47  
48      // partial database export
49      QueryDataSet partialDataSet = new QueryDataSet(connection);
50      partialDataSet.addTable("name_usage", "SELECT * FROM name_usage order by lft nulls last");
51      FlatXmlDataSet.write(partialDataSet,
52        new FileOutputStream("checklistbank-mybatis-service/src/test/resources/dbunit/name_usage.xml"));
