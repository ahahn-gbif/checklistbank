80    private <T> PagingResponse<T> list(InputStream json, TypeReference<List<T>> ref, Pageable page) {
81      try {
82        List<T> data = mapper.readValue(json, ref);
83        json.close();
84        PagingResponse<T> resp = new PagingResponse<T>(page, null, data);
85        resp.setEndOfRecords(true);
86        return resp;
87      } catch (IOException e) {
88        throw new IllegalStateException(e);
89      }
90    }
