263   private Usage assertUsageExistsOnce(ChecklistCache cache, String name, Rank rank, TaxonomicStatus status,
264     int sourceID) {
265     List<Usage> usages = cache.findUsages(name);
266     assertEquals("Exactly one usage should exist for " + name, 1, usages.size());
267     Usage u = usages.get(0);
268     assertEquals("Wrong rank for " + name, rank, u.rank);
269     assertEquals("Wrong status for " + name, status, u.status);
270     //System.out.println("Source id for " + name + " is " + u.sourceKey);
271     if (sourceID != -1) {
272       assertEquals("Wrong source id for " + name, sourceID, u.sourceKey);
273     }

278   private void assertClassification(ChecklistCache cache, Usage u, String parent, LinneanClassification cl) {
279     final int usageKey = u.key;
280     // check direct parent first
281     Usage p = cache.get(u.parentKey);
282     assertEquals("Wrong direct parent", parent, CacheUtils.nameOf(p, cache));
283 
284     // verify hierarchy
285     if (!u.rank.isSpeciesOrBelow()) {
286       assertFalse("Additional higher rank for " + usageKey + " found: " + cl.getHigherRank(u.rank),
287         cl.getHigherRank(u.rank) == null);
288       assertEquals(u.rank + " classification wrong", cl.getHigherRank(u.rank), CacheUtils.canonicalNameOf(u, cache));
289       ClassificationUtils.setHigherRank(cl, u.rank, null);
290     }
291     while (u.hasParent()) {
292       u = cache.get(u.parentKey);
293       if (!u.rank.isSpeciesOrBelow()) {
294         assertFalse("Additional higher rank for " + usageKey + " found: " + cl.getHigherRank(u.rank),
295           cl.getHigherRank(u.rank) == null);
296         assertEquals(u.rank + " classification wrong", cl.getHigherRank(u.rank), CacheUtils.canonicalNameOf(u, cache));
297         ClassificationUtils.setHigherRank(cl, u.rank, null);
298       }
299     }
300 
301     assertFalse("Additional higher ranks expected: " + ClassificationUtils.getHigherClassification(cl),
302       ClassificationUtils.hasContent(cl));
303   }
