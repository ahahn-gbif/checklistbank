030  private void assertMatch(String name, LinneanClassification query, Integer expectedKey, IntRange confidence) {
031    NameUsageMatch best = matcher.match(name, null, query, false, true);
032    System.out.println(
033      name + " matches " + best.getScientificName() + " [" + best.getUsageKey() + "] with confidence " + best
034        .getConfidence());
035    System.out.println(best.getNote());
036
037    assertTrue("Wrong match type", best.getMatchType() != NameUsageMatch.MatchType.NONE);
038    if (best.getAlternatives() != null) {
039      for (NameUsageMatch m : best.getAlternatives()) {
040        System.out.println("  Alt: " + m.getScientificName() + " [" + m.getUsageKey() + "] score=" + m.getConfidence() + ". " + m.getNote());
041      }
042    }
043    assertEquals( expectedKey, best.getUsageKey());
044    if (confidence != null) {
045      assertTrue("confidence " + best.getConfidence() + " not within " + confidence, confidence.containsInteger(best.getConfidence()));
046    }
047    assertMatchConsistency(best);
048  }

050  private void assertNoMatch(String name, LinneanClassification query, IntRange confidence) {
051    NameUsageMatch best = matcher.match(name, null, query, false, true);
052    System.out.println(best.getNote());
053    assertEquals(NameUsageMatch.MatchType.NONE, best.getMatchType());
054    if (best.getAlternatives() != null && !best.getAlternatives().isEmpty()) {
055      NameUsageMatch alt = best.getAlternatives().get(0);
056      for (NameUsageMatch m : best.getAlternatives()) {
057        System.out.println("  Alt: " + m.getScientificName() + " [" + m.getUsageKey() + "] score=" + m.getConfidence() + ". " + m.getNote());
058      }
059      if (confidence != null) {
060        assertTrue("alt confidence " + alt.getConfidence() + " not within " + confidence, confidence.containsInteger(alt.getConfidence()));
061      }
062    }
063  }

066  private void assertMatchConsistency(NameUsageMatch match){
067    assertNotNull(match.getConfidence());
068    assertNotNull(match.getMatchType());
069    if (NameUsageMatch.MatchType.NONE == match.getMatchType()){
070      assertNull(match.getUsageKey());
071      assertNull(match.getScientificName());
072
073      assertNull(match.getSpeciesKey());
074      assertNull(match.getGenusKey());
075      assertNull(match.getFamilyKey());
076      assertNull(match.getOrderKey());
077      assertNull(match.getClassKey());
078      assertNull(match.getPhylumKey());
079      assertNull(match.getKingdomKey());
080
081      assertNull(match.getSpecies());
082      assertNull(match.getGenus());
083      assertNull(match.getFamily());
084      assertNull(match.getOrder());
085      assertNull(match.getClazz());
086      assertNull(match.getPhylum());
087      assertNull(match.getKingdom());
088
089    } else {
090      assertNotNull(match.getUsageKey());
091      assertNotNull(match.getScientificName());
092
093      if(match.getRank() != null){
094        Rank rank = match.getRank();
095        if (rank.isSuprageneric()){
096          assertNull(match.getSpecies());
097          assertNull(match.getSpeciesKey());
098          assertNull(match.getGenus());
099          assertNull(match.getGenusKey());
100
101        } else if (rank == Rank.GENUS){
102          assertNotNull(match.getGenus());
103          assertNull(match.getSpecies());
104          assertNull(match.getSpeciesKey());
105
106        } else if (rank == Rank.SPECIES){
107          assertNotNull(match.getGenus());
108          assertNotNull(match.getSpecies());
109          assertNotNull(match.getSpeciesKey());
110          if (!match.isSynonym()){
111            assertEquals(match.getUsageKey(), match.getSpeciesKey());
112            assertTrue(match.getScientificName().startsWith(match.getSpecies()));
113          }
114
115        } else if (rank.isInfraspecific()){
116          assertNotNull(match.getGenus());
117          assertNotNull(match.getSpecies());
118          assertNotNull(match.getSpeciesKey());
119          if (!match.isSynonym()){
120            assertFalse(match.getUsageKey().equals(match.getSpeciesKey()));
121            assertTrue(match.getScientificName().startsWith(match.getSpecies()));
122          }
123
124        }
125      }
126    }
127  }

129  private void assertNubIdNotNullAndNotEqualToAnyHigherRank(NameUsageMatch x) {
130    assertNotNull(x.getUsageKey());
131    assertFalse(x.getUsageKey() == x.getKingdomKey());
132    assertFalse(x.getUsageKey() == x.getPhylumKey());
133    assertFalse(x.getUsageKey() == x.getClassKey());
134    assertFalse(x.getUsageKey() == x.getOrderKey());
135    assertFalse(x.getUsageKey() == x.getFamilyKey());
136  }

138  @Test
139  public void testMatching() throws IOException {
140    LinneanClassification cl = new NameUsageMatch();
141    assertMatch("Anephlus", cl, 1100135, new IntRange(92,93));
142    assertMatch("Aneplus", cl, 1100050, new IntRange(92,94));
143
144    cl.setKingdom("Animalia");
145    cl.setClazz("Insecta");
146    assertMatch("Aneplus", cl, 1100050, new IntRange(97,100));
147
148    // genus Aneplus is order=Coleoptera, but Anelus is a Spirobolida in class Diplopoda
149    cl.setClazz("Diplopoda");
150    cl.setOrder("Spirobolida");
151    assertMatch("Aneplus", cl, 1027792, new IntRange(92, 95));
152
153    cl.setFamily("Atopetholidae");
154    assertMatch("Aneplus", cl, 1027792, new IntRange(99, 100));
155    assertMatch("Annepluss", cl, 1027792, new IntRange(95,99));
156
157    assertNoMatch("Annepluss", new NameUsageMatch(), new IntRange(30,80));
158  }

160  @Test
161  public void testHomonyms() throws IOException {
162    // Oenanthe
163    LinneanClassification cl = new NameUsageMatch();
164    assertNoMatch("Oenanthe", cl, null);
165
166    cl.setKingdom("Animalia");
167    assertMatch("Oenanthe", cl, 2492483, new IntRange(96, 99));
168
169    cl.setKingdom("Plantae");
170    assertMatch("Oenanthe", cl, 3034893, new IntRange(96, 99));
171    assertMatch("Oenante", cl, 3034893, new IntRange(95,99));
172
173
174    // Acanthophora
175    cl = new NameUsageMatch();
176    assertNoMatch("Acanthophora", cl, null);
177
178    cl.setKingdom("Animalia");
179    assertMatch("Acanthophora", cl, 3251480, new IntRange(93,97));
180
181    cl.setKingdom("Plantae"); // there are multiple plant genera, this should match to plantae now
182    assertMatch("Acanthophora", cl, 6, new IntRange(98,100));
183
184    cl.setFamily("Araliaceae");
185    assertMatch("Acanthophora", cl, 3036337, new IntRange(98, 100));
186    assertMatch("Acantophora", cl, 3036337, new IntRange(96,100)); // fuzzy match
187    // try matching with authors
188    assertMatch("Acantophora Merrill", cl, 3036337, new IntRange(97, 100)); // fuzzy match
189
190    cl.setFamily("Rhodomelaceae");
191    assertMatch("Acanthophora", cl, 2659277, new IntRange(97, 100));
192
193
194    // species match
195    cl = new NameUsageMatch();
196    assertMatch("Puma concolor", cl, 2435099, new IntRange(98,100));
197
198    cl.setGenus("Puma");
199    assertMatch("P. concolor", cl, 2435099, new IntRange(98, 100));
200
201    cl.setKingdom("Animalia");
202    assertMatch("P. concolor", cl, 2435099, new IntRange(99, 100));
203
204    cl.setKingdom("Plantae");
205    assertMatch("Puma concolor", cl, 2435099, new IntRange(95, 98));
206    // Picea concolor is a plant, but this fuzzy match is too far off
207    assertMatch("Pima concolor", cl, 6, null);
208    // this one should match though
209    assertMatch("Pica concolor", cl, 5284657, new IntRange(92, 97));
210
211    cl.setFamily("Pinaceae");
212    assertMatch("Pima concolor", cl, 5284657, new IntRange(93, 96));
213
214
215    // Amphibia is a homonym genus, but also and most prominently a class!
216    cl = new NameUsageMatch();
217    // non existing "species" name. Amphibia could either be the genus or the class, who knows...
218    assertNoMatch("Amphibia eyecount", cl, null);
219
220    // first try a match against the algae genus
221    cl.setKingdom("Plantae");
222    cl.setClazz("Rhodophyceae");
223    assertMatch("Amphibia eyecount", cl, 2659299, new IntRange(90, 99));
224
225    // now try with the animal class
226    cl.setKingdom("Animalia");
227    cl.setClazz("Amphibia");
228    assertMatch("Amphibia eyecount", cl, 131, new IntRange(98, 100));
229  }
