001package org.gbif.nub.lookup.similarity;
002
003import org.junit.Test;
004
005import static org.junit.Assert.assertEquals;
006
007public class DamerauLevenshteinTest {
008  final double delta = 0.01d;
009
010  @Test
011  public void testConvertEditDistanceToSimilarity() throws Exception {
012    assertEquals(100d, DamerauLevenshtein.convertEditDistanceToSimilarity(0, "1234567890", "1234567890"), delta);
013    assertEquals(75d, DamerauLevenshtein.convertEditDistanceToSimilarity(2, "12345678", "12345678"), delta);
014    assertEquals(87.5d, DamerauLevenshtein.convertEditDistanceToSimilarity(1, "1234567", "123456789"), delta);
015    // doesnt make sense, but an edit distance of zero is always 100%
016    assertEquals(100d, DamerauLevenshtein.convertEditDistanceToSimilarity(0, "12", "123456789"), delta);
017    assertEquals(0d, DamerauLevenshtein.convertEditDistanceToSimilarity(10, "1234567", "123456789"), delta);
018  }
019}
