001package org.gbif.nub.lookup;
002
003import org.gbif.api.model.checklistbank.NameUsageMatch;
004import org.gbif.api.model.common.LinneanClassification;
005import org.gbif.api.service.checklistbank.NameUsageMatchingService;
006import org.gbif.checklistbank.service.mybatis.guice.ChecklistBankServiceMyBatisModule;
007import org.gbif.utils.file.properties.PropertiesUtil;
008
009import java.io.IOException;
010import java.util.Properties;
011
012import com.google.inject.Guice;
013import com.google.inject.Injector;
014import org.slf4j.Logger;
015import org.slf4j.LoggerFactory;
016
017public class NubMatchingServiceTestManual {
018
019  private static final Logger LOG = LoggerFactory.getLogger(NubMatchingServiceTestManual.class);
020
021  private final NameUsageMatchingService matcher;
022
023  public NubMatchingServiceTestManual() throws IOException {
024
025    LOG.info("Load clb properties");
026    Properties properties = PropertiesUtil.loadProperties("checklistbank.properties");
027
028    LOG.info("Create guice injector");
029    Injector inj = Guice.createInjector(new ChecklistBankServiceMyBatisModule(properties), new NubMatchingModule());
030
031    LOG.info("Create matching service");
032    matcher = inj.getInstance(NameUsageMatchingService.class);
033
034    LOG.info("Nub Matching setup complete");
035  }
036
037  public void testMatching() throws IOException {
038    LinneanClassification cl = new NameUsageMatch();
039    // test identical
040    matcher.match("Animalia", null, cl, true, true);
041    matcher.match("Animals", null, cl, true, true);
042    matcher.match("Insects", null, cl, true, true);
043    cl.setKingdom("Animalia");
044    matcher.match("Puma concolor", null, cl, true, true);
045    cl.setKingdom("Plantae");
046    matcher.match("Puma concolor", null, cl, true, true);
047  }
048
049
050  public static void main (String[] args) throws IOException {
051    NubMatchingServiceTestManual test = new NubMatchingServiceTestManual();
052    test.testMatching();
053  }
054
055}
