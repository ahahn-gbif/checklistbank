001package org.gbif.nub.lookup;
002
003import org.gbif.api.model.checklistbank.NameUsage;
004import org.gbif.api.model.checklistbank.NameUsageMatch;
005import org.gbif.api.util.VocabularyUtils;
006import org.gbif.api.vocabulary.Rank;
007import org.gbif.file.CSVReader;
008import org.gbif.file.CSVReaderFactory;
009import org.gbif.nameparser.NameParser;
010
011import java.io.IOException;
012import java.io.InputStream;
013import java.util.List;
014
015import com.google.common.base.Preconditions;
016import com.google.common.collect.Lists;
017import com.google.common.io.Resources;
018import org.junit.BeforeClass;
019import org.junit.Test;
020
021import static org.junit.Assert.assertEquals;
022import static org.junit.Assert.assertFalse;
023import static org.junit.Assert.assertTrue;
024
025public class NubIndexTest {
026
027  private static NubIndex index;
028  private static List<NameUsage> names;
029
030  @BeforeClass
031  public static void buildMatcher() throws IOException {
032    HigherTaxaLookup syn = new HigherTaxaLookup();
033    syn.loadClasspathDicts("dicts");
034
035    names = readTestNames();
036
037    index = new NubIndex();
038    for (NameUsage u : names) {
039      index.addNameUsage(u);
040    }
041  }
042
043  public static List<NameUsage> readTestNames() throws IOException {
044    List<NameUsage> usages = Lists.newArrayList();
045
046    NameParser parser = new NameParser();
047
048    InputStream in = Resources.newInputStreamSupplier(Resources.getResource("testNames.txt")).getInput();
049    CSVReader reader = CSVReaderFactory.build(in, "UTF8", "\t", null, 0);
050    for (String[] row : reader) {
051      NameUsage n = new NameUsage();
052      n.setKey(Integer.valueOf(row[0]));
053      n.setScientificName(row[1]);
054      n.setCanonicalName(parser.parseToCanonical(n.getScientificName()));
055      n.setFamily(row[2]);
056      n.setOrder(row[3]);
057      n.setClazz(row[4]);
058      n.setPhylum(row[5]);
059      n.setKingdom(row[6]);
060      n.setSynonym(Boolean.parseBoolean(row[7]));
061      n.setRank( (Rank) VocabularyUtils.lookupEnum(row[8], Rank.class) );
062      usages.add(n);
063    }
064
065    Preconditions.checkArgument(usages.size() == 10, "Wrong number of test names");
066
067    return usages;
068  }
069
070  @Test
071  public void testMatchByName() throws Exception {
072    final Integer abiesAlbaKey = 7;
073    NameUsageMatch m = index.matchByUsageId(abiesAlbaKey);
074    assertEquals(abiesAlbaKey, m.getUsageKey());
075    assertEquals("Abies alba Mill.", m.getScientificName());
076    assertEquals(Rank.SPECIES, m.getRank());
077    assertFalse(m.isSynonym());
078
079    m = index.matchByName("Abies alba", true, 2).get(0);
080    assertEquals(abiesAlbaKey, m.getUsageKey());
081
082    m = index.matchByName("abies  alba", true, 2).get(0);
083    assertEquals(abiesAlbaKey, m.getUsageKey());
084
085    m = index.matchByName("Abbies alba", true, 2).get(0);
086    assertEquals(abiesAlbaKey, m.getUsageKey());
087
088    m = index.matchByName("abyes alba", true, 2).get(0);
089    assertEquals(abiesAlbaKey, m.getUsageKey());
090
091    m = index.matchByName(" apies  alba", true, 2).get(0);
092    assertEquals(abiesAlbaKey, m.getUsageKey());
093
094    // sciname soundalike filter enables this
095    m = index.matchByName("Aabbiess allba", true, 2).get(0);
096    assertEquals(abiesAlbaKey, m.getUsageKey());
097
098    m = index.matchByName("Aebies allba", true, 2).get(0);
099    assertEquals(abiesAlbaKey, m.getUsageKey());
100
101
102    // fuzzy searches use a minPrefix=1
103    assertTrue(index.matchByName("Obies alba", true, 2).isEmpty());
104
105    assertTrue(index.matchByName("Abies elba", false, 2).isEmpty());
106
107  }
108}
