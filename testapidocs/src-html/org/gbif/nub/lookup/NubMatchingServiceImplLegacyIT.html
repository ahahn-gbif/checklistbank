060  @Test
061  public void testClassificationSimilarity() throws IOException {
062
063    LinneanClassification cn1 = new NameUsageMatch();
064    LinneanClassification cn2 = new NameUsageMatch();
065
066    int score = matcher.classificationSimilarity(cn1, cn2);
067    assertTrue(score > -10);
068    assertTrue(score < 0);
069
070    cn1.setFamily("Asteraceae");
071    cn2.setFamily("Asteraceae");
072    score = matcher.classificationSimilarity(cn1, cn2);
073    assertTrue(score > 10);
074
075    cn2.setFamily("Asteraceaee");
076    score = matcher.classificationSimilarity(cn1, cn2);
077    assertTrue(score <= -5);
078
079    cn1.setKingdom("A");
080    cn2.setKingdom("B");
081    score = matcher.classificationSimilarity(cn1, cn2);
082    assertTrue(score < -12);
083  }

085  /**
086   * compare
087   * Plantae;Dinophyta;Dinophyceae;Gonyaulacales;;;
088   * with
089   * Protozoa;Dinophyta;;;;;
090   *
091   * @throws java.io.IOException
092   */
093  @Test
094  public void testRatingWihtKingdomSynonym() throws IOException {
095
096    LinneanClassification cn1 = new NameUsageMatch();
097    cn1.setKingdom("Plantae");
098    cn1.setPhylum("Dinophyta");
099    cn1.setClazz("Dinophyceae");
100    cn1.setOrder("Gonyaulacales");
101
102    LinneanClassification cn2 = new NameUsageMatch();
103    cn2.setKingdom("Plantae");
104    cn2.setPhylum("Dinophyta");
105    cn2.setClazz("Dinophyceae");
106    cn2.setOrder("Gonyaulacales");
107
108    // test identical
109    int score = matcher.classificationSimilarity(cn1, cn2);
110    assertTrue(score > 45);
111
112
113    // test kingdom synonym
114    cn2.setKingdom("Plants");
115    cn2.setPhylum("Dinophyta");
116    cn2.setClazz("");
117    cn2.setOrder("");
118
119    // test identical
120    score = matcher.classificationSimilarity(cn1, cn2);
121    assertTrue(score > 15);
122
123    // test vague kingdom
124    cn2.setKingdom("Protozoa");
125    score = matcher.classificationSimilarity(cn1, cn2);
126    assertTrue(score > -10);
127    assertTrue(score <= 0);
128
129    // test very different kingdom
130    cn2.setKingdom("Animalia");
131    score = matcher.classificationSimilarity(cn1, cn2);
132    assertTrue(score <= -15);
133  }

160  @Test
161  public void testRating() throws IOException {
162    final String queryName = "Acanthophora";
163
164    LinneanClassification query = new NameUsageMatch();
165    query.setKingdom("Animalia");
166    query.setPhylum("Porifera");
167    assertMatchedUsageKey(queryName, query, 3);
168
169    query.setPhylum("Arthropoda");
170    query.setClazz("Insecta");
171    assertMatchedUsageKey(queryName, query, 2);
172
173    query.setOrder("Wrongly");
174    assertMatchedUsageKey(queryName, query, 2);
175
176    query.setFamily("Geometridae");
177    assertMatchedUsageKey(queryName, query, 1);
178
179
180    query = new NameUsageMatch();
181    query.setFamily("Rhodomelaceae");
182    assertMatchedUsageKey(queryName, query, 5);
183
184    query.setFamily("Araliaceae");
185    assertMatchedUsageKey(queryName, query, 4);
186
187    query.setFamily("PHORIdae");
188    assertMatchedUsageKey(queryName, query, 2);
189
190
191    // homonym matching without a classification should return a NONE match
192    query = new NameUsageMatch();
193    query.setKingdom("");
194    query.setPhylum("");
195    query.setClazz("");
196    query.setOrder("");
197    query.setFamily("");
198    query.setGenus("");
199    NameUsageMatch best = matcher.match(queryName, null, query, false, true);
200    assertEquals(NameUsageMatch.MatchType.NONE, best.getMatchType());
201
202    // test that without a clear classification the best match is empty
203    query = new NameUsageMatch();
204    query.setKingdom("Animalia");
205    query.setGenus("Acanthophora");
206    best = matcher.match(queryName, null, query, false, true);
207    assertEquals(NameUsageMatch.MatchType.NONE, best.getMatchType());
208  }
