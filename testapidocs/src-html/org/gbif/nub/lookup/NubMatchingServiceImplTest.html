009public class NubMatchingServiceImplTest {
010
011  @Test
012  public void testInterpretGenus() throws Exception {
013    ParsedName pn = new ParsedName();
014    pn.setGenusOrAbove("P.");
015    pn.setSpecificEpithet("concolor");
016    NubMatchingServiceImpl.interpretGenus(pn, "Puma");
017    assertEquals("Puma concolor", pn.canonicalName());
018
019
020    pn.setGenusOrAbove("P.");
021    NubMatchingServiceImpl.interpretGenus(pn, "Felis");
022    assertEquals("P. concolor", pn.canonicalName());
023  }
024
