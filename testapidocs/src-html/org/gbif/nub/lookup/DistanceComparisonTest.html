071  @Test
072  public void testGetSimilarity() throws Exception {
073    for (String[] ns : names) {
074      LOG.debug(ns[0] + "  x  " + ns[1]);
075      doit("DL ", new DamerauLevenshtein(ns[0], ns[1]));
076      doit("MDL", new ModifiedDamerauLevenshtein(ns[0], ns[1], 3));
077      doit("JW ", new JaroWinklerSimilarity(ns[0], ns[1]));
078      doit("MJW", new ModifiedJaroWinklerSimilarity(ns[0], ns[1]));
079    }
080  }

092  private double doit(String name, StringSimilarity sim) {
093    double s = sim.getSimilarity();
094    LOG.debug(" {}={}", name, s);
095    return s;
096  }
097
098  private double doitTime(String name, StringSimilarity sim) {
099    long start = System.currentTimeMillis();
100    double s = sim.getSimilarity();
101    int repeat = 1000;
102    while (repeat > 0) {
103      sim.getSimilarity();
104      repeat--;
105    }
106    long stop = System.currentTimeMillis();
107    LOG.debug(" {}={}   1000x in {} ms", name, s, stop-start);
108    return s;
109  }
