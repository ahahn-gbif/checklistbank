001package org.gbif.nub.lookup;
002
003import org.gbif.api.model.checklistbank.NameUsage;
004import org.gbif.api.model.checklistbank.NameUsageMatch;
005import org.gbif.api.service.checklistbank.NameUsageMatchingService;

008
009import java.io.IOException;
010import java.io.InputStream;
011import java.util.List;
012import java.util.Map;
013
014import com.google.common.collect.Lists;
015import com.google.common.collect.Maps;
016import com.google.inject.Inject;
017import com.google.inject.PrivateModule;
018import com.google.inject.Provides;
019import com.google.inject.Singleton;
020import org.codehaus.jackson.map.ObjectMapper;
021import org.slf4j.Logger;
022import org.slf4j.LoggerFactory;
023
024/**
025 * Guice module setting up all dependencies to expose the NubMatching service.
026 */
027public class NubMatchingTestModule extends PrivateModule {
028  private static final Logger LOG = LoggerFactory.getLogger(NubMatchingTestModule.class);
029
030  @Override
031  protected void configure() {
032
033    bind(NameUsageMatchingService.class).to(NubMatchingServiceImpl.class).asEagerSingleton();
034    expose(NameUsageMatchingService.class);
035  }
036
037  @Provides
038  @Inject
039  @Singleton
040  public static NubIndex provideIndex() throws IOException {
041    NubIndex index = new NubIndex();
042    LOG.info("Start loading test nub usages into lucene index ...");
043
044    for (NameUsage u : loadIndexJson()) {
045      index.addNameUsage(u);
046    }
047    LOG.info("Lucene index initialized");
048    return index;
049  }
050
051  @Provides
052  @Singleton
053  public static HigherTaxaLookup provideSynonyms() throws IOException {
054    LOG.info("Loading synonym dictionaries from classpath ...");
055    HigherTaxaLookup syn = new HigherTaxaLookup();
056    syn.loadClasspathDicts("dicts");
057    return syn;
058  }
059
060  @Provides
061  @Singleton
062  public NameParser provideParser() {
063    NameParser parser = new NameParser();
064    return parser;
065  }
066
067  /**
068   * Load all nubXX.json files from the index resources into a distinct list of NameUsage instances.
069   * The individual nubXX.json files are regular results of a NameUsageMatch and can be added to the folder
070   * to be picked up here.
071   */
072  private static List<NameUsage> loadIndexJson() {
073    Map<Integer, NameUsage> usages = Maps.newHashMap();
074
075    InputStreamUtils isu = new InputStreamUtils();
076    ObjectMapper mapper = new ObjectMapper();
077
078    int id = 0;
079    while (id < 100) {
080      String file = "index/nub"+id+".json";
081      InputStream json = isu.classpathStream(file);
082      if (json != null) {
083        try {
084          int before = usages.size();
085          NameUsageMatch m = mapper.readValue(json, NameUsageMatch.class);
086          for (NameUsage u : extractUsages(m)) {
087            if (u != null) {
088              usages.put(u.getKey(), u);
089            }
090          }
091          System.out.println("Loaded " + (usages.size() - before) + " new usage(s) from " + file);
092        } catch (IOException e) {
093          System.err.println(" Failed to read " + file + ": " + e.getMessage());
094        }
095      }
096      id++;
097    }
098    return Lists.newArrayList(usages.values());
099  }
100
101  private static List<NameUsage> extractUsages(NameUsageMatch m) {
102    List<NameUsage> usages = Lists.newArrayList();
103    usages.add(toUsage(m));
104    if (m.getAlternatives() != null) {
105      for (NameUsageMatch m2 : m.getAlternatives()) {
106        usages.add(toUsage(m2));
107      }
108    }
109    return usages;
110  }
111
112  private static NameUsage toUsage(NameUsageMatch m) {
113    if (m.getMatchType() != NameUsageMatch.MatchType.NONE) {
114      NameUsage u = new NameUsage();
115      u.setKey(m.getUsageKey());
116      u.setCanonicalName(m.getCanonicalName());
117      u.setScientificName(m.getScientificName());
118      u.setRank(m.getRank());
119      u.setSynonym(m.isSynonym());
120
121      u.setKingdom(m.getKingdom());
122      u.setPhylum(m.getPhylum());
123      u.setClazz(m.getClazz());
124      u.setFamily(m.getFamily());
125      u.setGenus(m.getGenus());
126      u.setSubgenus(m.getSubgenus());
127      u.setSpecies(m.getSpecies());
128
129      u.setKingdomKey(m.getKingdomKey());
130      u.setPhylumKey(m.getPhylumKey());
131      u.setClassKey(m.getClassKey());
132      u.setFamilyKey(m.getFamilyKey());
133      u.setGenusKey(m.getGenusKey());
134      u.setSubgenusKey(m.getSubgenusKey());
135      u.setSpeciesKey(m.getSpeciesKey());
136      return u;
137    }
138    return null;
139  }

141}
