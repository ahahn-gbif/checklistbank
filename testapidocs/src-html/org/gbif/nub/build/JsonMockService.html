021/**
022 * Test mock implementation of the batch service loading json files from the dataset test directory
023 * to answer the list and listNames methods. Paging is entirely ignored and always the full list of
024 * records is returned!
025 */
026public class JsonMockService implements UsageService, ParsedNameService {
027  private InputStreamUtils isu = new InputStreamUtils();
028  private final ObjectMapper mapper;
029
030  public JsonMockService() {
031    mapper = new ObjectMapper();
032    mapper.configure(DeserializationConfig.Feature.FAIL_ON_UNKNOWN_PROPERTIES, false);
033  }
034
035  @Override
036  public List<Integer> listAll() {

044  }
045
046  @Override
047  public List<NameUsage> listRange(int usageKeyStart, int usageKeyEnd) {
048    // TODO: Write implementation

050  }
051
052  @Override
053  public PagingResponse<Usage> list(UUID datasetKey, Pageable page) {
054    System.out.println("Read usages for " + datasetKey);
055    InputStream json = usagesJson(datasetKey);
056    TypeReference ref = new TypeReference<List<Usage>>() { };
057    return list(json, ref, page);
058  }
059

071
072  @Override
073  public PagingResponse<ParsedName> listNames(UUID datasetKey, Pageable page) {
074    System.out.println("Read names for " + datasetKey);
075    InputStream json = namesJson(datasetKey);
076    TypeReference ref = new TypeReference<List<ParsedName>>() { };
077    return list(json, ref, page);
078  }
079
080  private <T> PagingResponse<T> list(InputStream json, TypeReference<List<T>> ref, Pageable page) {
081    try {
082      List<T> data = mapper.readValue(json, ref);
083      json.close();
084      PagingResponse<T> resp = new PagingResponse<T>(page, null, data);
085      resp.setEndOfRecords(true);
086      return resp;
087    } catch (IOException e) {
088      throw new IllegalStateException(e);
089    }
090  }
091
092  private InputStream namesJson(UUID datasetKey) {
093    String file = prefix(datasetKey) + "-names.json";
094    return isu.classpathStream(file);
095  }
096
097  private InputStream usagesJson(UUID datasetKey) {
098    String file = prefix(datasetKey) + "-usages.json";
099    return isu.classpathStream(file);
100  }
101
102  private String prefix(UUID datasetKey) {
103    return "datasets/"+datasetKey.toString();
104  }
