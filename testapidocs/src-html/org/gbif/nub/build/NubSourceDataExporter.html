024public class NubSourceDataExporter {
025  private final File dir;
026  private final UsageService uService;
027  private final ParsedNameService nService;
028  private final ObjectMapper mapper = new ObjectMapper();
029
030  public static final UUID MAMMALS = UUID.fromString("672aca30-f1b5-43d3-8a2b-c1606125fa1b");
031  public static final UUID HOMONYMS = UUID.fromString("714c64e3-2dc1-4bb7-91e4-54be5af4da12");
032  public static final UUID TOORSACORRIDOR = UUID.fromString("d9f426e7-845c-4a63-be0d-18506f235357");
033  public static final UUID GRACILLARIIDAE = UUID.fromString("bb5f507f-f7de-4a5a-ae7f-ad8abbe68bef");
034  public static final UUID CICADELLINAE = UUID.fromString("26bca1b5-3ef6-4e97-9672-0058c79185fb");
035  public static final UUID IOCBIRDS = UUID.fromString("c696e5ee-9088-4d11-bdae-ab88daffab78");
036  public static final UUID VASCAN = UUID.fromString("3f8a1297-3259-4700-91fc-acc4170b27ce");
037  // only families and above!
038  public static final UUID COL = NubGenerator.COL_KEY;
039
040  public NubSourceDataExporter(File dir, UsageService uService, ParsedNameService nService) {
041    this.dir = dir;
042    this.uService = uService;
043    this.nService = nService;
044  }
045
046  private void export(UUID datasetKey) throws IOException {
047    JsonFactory jfactory = new JsonFactory();
048    jfactory.setCodec(mapper);
049
050    exportNames(datasetKey, jfactory);
051    exportUsages(datasetKey, jfactory);
052  }
053
054  private void exportUsages(UUID datasetKey, JsonFactory jfactory) throws IOException {
055    File json = new File(dir, datasetKey.toString() + "-usages.json");
056    System.out.println("Start dumping usages for dataset " + datasetKey + " to " + json.getAbsolutePath());
057    JsonGenerator jGen = jfactory.createJsonGenerator(json, JsonEncoding.UTF8);
058    jGen.useDefaultPrettyPrinter();
059    jGen.writeStartArray();
060    PagingRequest page = new PagingRequest(0, 10000);
061    PagingResponse<Usage> resp;
062    do {
063      resp = uService.list(datasetKey, page);
064      for (Usage u : resp.getResults()) {
065        jGen.writeObject(u);
066      }
067      page.nextPage();
068    }
069    while (!resp.isEndOfRecords());
070    jGen.writeEndArray();
071    System.out.println("Done.");
072    jGen.close();
073  }
074
075  private void exportNames(UUID datasetKey, JsonFactory jfactory) throws IOException {
076    File json = new File(dir, datasetKey.toString() + "-names.json");
077    System.out.println("Start dumping names for dataset " + datasetKey + " to " + json.getAbsolutePath());
078    JsonGenerator jGen = jfactory.createJsonGenerator(json, JsonEncoding.UTF8);
079    jGen.useDefaultPrettyPrinter();
080    jGen.writeStartArray();
081    PagingRequest page = new PagingRequest(0, 10000);
082    PagingResponse<ParsedName> respPN;
083    do {
084      respPN = nService.listNames(datasetKey, page);
085      for (ParsedName n : respPN.getResults()) {
086        jGen.writeObject(n);
087      }
088      page.nextPage();
089    }
090    while (!respPN.isEndOfRecords());
091    jGen.writeEndArray();
092    System.out.println("Done.");
093    jGen.close();
094  }
095

101    NubSourceDataExporter exporter = new NubSourceDataExporter(new File("/Users/mdoering/Desktop"), uService, nService);
