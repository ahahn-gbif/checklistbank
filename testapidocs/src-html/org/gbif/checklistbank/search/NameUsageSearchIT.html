104  @Test
105  public void testCanonicalHighlighting() {
106    // build request
107    NameUsageSearchRequest searchRequest = new NameUsageSearchRequest(0L, 20);
108    searchRequest.setQ("vulgaris");
109    searchRequest.setHighlight(true);
110
111    // query
112    SearchResponse<NameUsageSearchResult, NameUsageSearchParameter> response = searchService.search(searchRequest);
113
114    // assert
115    Assert.assertEquals((Long) 10l, response.getCount());
116    // this usage also matches description and should therefore come first
117    Collection<Integer> keys = Collections2.transform(response.getResults(), extractKeyFunction);
118    assertEquals(10, keys.size());
119    assertThat(keys,
120      hasItems(100000025, 100000030, 100000031, 100000032, 100000033, 100000034, 100000035, 100000036, 100000037
121        , 100000038));
122
123    int highlighted = 0;
124    for (NameUsageSearchResult nu : response.getResults()) {
125      System.out.println(nu.getCanonicalName());
126
127      if (SolrConstants.HL_REGEX.matcher(nu.getCanonicalName()).find()) {
128        highlighted++;
129      }
130    }
131    Assert.assertEquals("wrong number of canonical highlights", 10, highlighted);
132  }

225  @Test
226  public void testVernacularNames() {
227    SearchResponse<NameUsageSearchResult, NameUsageSearchParameter> resp =
228      search("Sciurus vulgaris Linnaeus", NameUsageSearchParameter.RANK, Rank.SPECIES);
229    assertEquals(1, resp.getResults().size());
230    NameUsageSearchResult nu = resp.getResults().get(0);
231    assertEquals(2, nu.getVernacularNames().size());
232
233    resp = search("Sciurus vulgaris Linnaeus", NameUsageSearchParameter.RANK, Rank.SPECIES);
234    assertEquals(2, resp.getResults().get(0).getVernacularNames().size());
235
236    resp = search("Sciurus vulgaris Linnaeus", NameUsageSearchParameter.RANK, Rank.SPECIES);
237    assertEquals(2, resp.getResults().get(0).getVernacularNames().size());
238
239    resp = search("Eichhörnchen", null, null);
240    assertEquals(2, resp.getResults().size());
241
242    resp = search("Europäisches Eichhörnchen", null, null);
243    assertEquals(1, resp.getResults().size());
244    assertEquals(2, resp.getResults().get(0).getVernacularNames().size());
245    Set<String> vnames = getVernacularNamesSet(resp.getResults().get(0));
246    assertTrue(vnames.contains("Europäisches Eichhörnchen"));
247  }

249  private void assertSearch(String q, Long expectedCount) {
250    assertSearch(q, null, null, expectedCount, null);
251  }
252
253  private SearchResponse<NameUsageSearchResult, NameUsageSearchParameter> assertSearch(String q,
254    NameUsageSearchParameter facet,
255    Enum<?> facetFilter, Long expectedCount, Integer expectedFacetCounts) {
256
257    SearchResponse<NameUsageSearchResult, NameUsageSearchParameter> response = search(q, facet, facetFilter);
258
259    // assert
260    if (expectedCount != null) {
261      assertEquals(expectedCount, response.getCount());
262    }
263
264    if (facet != null && expectedFacetCounts != null) {
265      assertEquals(1, response.getFacets().size());
266      assertEquals(expectedFacetCounts, (Integer) response.getFacets().get(0).getCounts().size());
267    }
268    if (facet == null) {
269      Assert.assertTrue(response.getFacets().isEmpty());
270    }
271
272    return response;
273  }
274
275  private Set<String> getVernacularNamesSet(NameUsageSearchResult usage) {
276    Set<String> vnames = Sets.newHashSet();
277    for (VernacularName vn : usage.getVernacularNames()) {
278      vnames.add(vn.getVernacularName());
279    }
280    return vnames;
281  }
282
283  private SearchResponse<NameUsageSearchResult, NameUsageSearchParameter>
284    search(String q, NameUsageSearchParameter facet, Enum<?> filter) {
285
286    // build request
287    NameUsageSearchRequest searchRequest = new NameUsageSearchRequest(0L, 10);
288    searchRequest.setQ(q);
289    if (filter != null) {
290      searchRequest.addParameter(facet, filter);
291    }
292    if (facet != null) {
293      searchRequest.addFacets(facet);
294    }
295
296    // query
297    return searchService.search(searchRequest);
298  }

300  /**
301   * Utility method for testing suggest service.
302   */
303  private List<NameUsageSearchResult> searchSuggest(String q) {
304    NameUsageSuggestRequest req = new NameUsageSuggestRequest();
305    req.setQ(q);
306    req.setLimit(10);
307    req.addParameter(NameUsageSearchParameter.DATASET_KEY, SQUIRRELS_DATASET_KEY);
308    return searchService.suggest(req);
309  }
