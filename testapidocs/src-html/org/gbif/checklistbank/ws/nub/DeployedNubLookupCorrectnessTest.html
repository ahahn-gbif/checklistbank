022public class DeployedNubLookupCorrectnessTest {

024  private static final List<String> WS_URLS = ImmutableList.of(
025//    "http://mogo.gbif.org:8080/ws-nub/nub/",
026//    "http://boma.gbif.org:8080/ws-nub/nub/",
027    "http://ecat-dev.gbif.org:8080/nub-ws/nub/"
028  );
029
030  private List<NameUsageMatchingService> CLIENTS = Lists.newArrayList();
031
032  public DeployedNubLookupCorrectnessTest() {
033    final String wsProp = "checklistbank.match.ws.url";
034
035    for (String url : WS_URLS) {
036      Properties props = new Properties();
037      props.setProperty(wsProp, url);
038
039      ChecklistBankWsClientModule mod = new ChecklistBankWsClientModule(props, false, false, true);
040      Injector injector = Guice.createInjector(mod);
041      CLIENTS.add(injector.getInstance(NameUsageMatchingService.class));
042    }
043  }
044
045  private void testNubLookup() throws IOException {
046    CSVReader reader = CSVReader.build(FileUtils.getClasspathFile("lookup_checks.txt"), "UTF-8", ",", '"', 1);
047
048    while (reader.hasNext()) {
049      String[] row = reader.next();
050      if (row.length >= 9) {
051        String r = Strings.emptyToNull(getCol(row, 9));
052        Rank rank = r==null? null : Rank.valueOf(r);
053        LinneanClassification cl = new NameUsageMatch();
054        cl.setKingdom(getCol(row, 1));
055        cl.setPhylum(getCol(row, 2));
056        cl.setClazz(getCol(row, 3));
057        cl.setOrder(getCol(row, 4));
058        cl.setFamily(getCol(row, 5));
059        cl.setGenus(getCol(row, 6));
060        testQuery(getColAsInt(row, 0), cl, getCol(row, 7), getCol(row, 8), rank);
061      } else {
062        // probably a comment
063        System.out.println(StringUtils.join(row));
064      }
065    }
066  }
067
068  private String getCol(String[] row, int column) {
069    String val = StringUtils.trimToNull(row[column]);
070    if (val == null || val.equalsIgnoreCase("null")) {
071      return null;
072    }
073    return val;
074  }
075
076  private Integer getColAsInt(String[] row, int column) {
077    String val = getCol(row, column);
078    if (val != null) {
079      try {
080        return Integer.valueOf(val);
081      } catch (NumberFormatException e) {
082        // nothing to do
083      }
084    }
085    return null;
086  }
087
088  private void testQuery(Integer targetNubId, LinneanClassification cl, String scientificName, String authorship, Rank rank) {
089
090    List<NameUsageMatch> results = Lists.newArrayList();
091    for (NameUsageMatchingService client : CLIENTS) {
092      NameUsageMatch lookup = null;
093      try {
094        lookup = client.match(scientificName, rank, cl, false, true);
095        results.add(lookup);
096
097        boolean gotMatch;
098        if (targetNubId == null) {
099          gotMatch = lookup.getUsageKey() == null;
100        } else {
101          gotMatch = lookup.getUsageKey() != null && lookup.getUsageKey().equals(targetNubId);
102        }
103        if (rank != null && rank.isSuprageneric()){
104          if (lookup.getSpeciesKey()!=null || lookup.getSpecies()!=null){
105            System.err.println(
106              "Lookup for "+rank.name()+" [" + scientificName +"] contains non null species "+lookup.getSpecies()+" [" + lookup.getSpeciesKey() + "]");
107          }
108        }
109        if (gotMatch) {
110          System.out.println(
111            "Lookup good for sciName [" + scientificName + "] nubID [" + targetNubId + "] - " + client.toString());
112        } else {
113          System.err.println(
114            "Lookup mismatch for sciName [" + scientificName +"] - wanted [" + targetNubId + "], got [" + (lookup == null ? null : lookup.getUsageKey()) + "] - [" + client.toString() + "]");
115        }
116      } catch (Exception e) {
117        System.err.println("Webservice ERROR: " + e.getMessage());
118      }
119
120    }
121
122    for (NameUsageMatch outer : results) {
123      if (outer==null) continue;
124      for (NameUsageMatch inner : results) {
125        if (inner == null) continue;
126        if (System.identityHashCode(outer) != System.identityHashCode(inner)) {
127          if (!equalLookups(outer, inner)) {
128            System.err.println("Results are different from different servers for sciName [" + scientificName + "]");
129          }
130        }
131      }
132    }
133  }
134
135  private boolean equalLookups(NameUsageMatch a, NameUsageMatch b) {
136    boolean match = true;
137    if (a.getConfidence() == null || b.getConfidence() == null) {
138      match = match && a.getConfidence() == null && b.getConfidence() == null;
139    } else {
140      match = match && a.getConfidence().equals(b.getConfidence());
141    }
142
143    if (a.getUsageKey() == null || b.getUsageKey() == null) {
144      match = match && a.getUsageKey() == null && b.getUsageKey() == null;
145    } else {
146      match = match && a.getUsageKey().equals(b.getUsageKey());
147    }
148
149    if (a.getClassKey() == null || b.getClassKey() == null) {
150      match = match && a.getClassKey() == null && b.getClassKey() == null;
151    } else {
152      match = match && a.getClassKey().equals(b.getClassKey());
153    }
154
155    if (a.getFamilyKey() == null || b.getFamilyKey() == null) {
156      match = match && a.getFamilyKey() == null && b.getFamilyKey() == null;
157    } else {
158      match = match && a.getFamilyKey().equals(b.getFamilyKey());
159    }
160
161    if (a.getGenusKey() == null || b.getGenusKey() == null) {
162      match = match && a.getGenusKey() == null && b.getGenusKey() == null;
163    } else {
164      match = match && a.getGenusKey().equals(b.getGenusKey());
165    }
166
167    if (a.getKingdomKey() == null || b.getKingdomKey() == null) {
168      match = match && a.getKingdomKey() == null && b.getKingdomKey() == null;
169    } else {
170      match = match && a.getKingdomKey().equals(b.getKingdomKey());
171    }
172
173    if (a.getOrderKey() == null || b.getOrderKey() == null) {
174      match = match && a.getOrderKey() == null && b.getOrderKey() == null;
175    } else {
176      match = match && a.getOrderKey().equals(b.getOrderKey());
177    }
178
179    if (a.getPhylumKey() == null || b.getPhylumKey() == null) {
180      match = match && a.getPhylumKey() == null && b.getPhylumKey() == null;
181    } else {
182      match = match && a.getPhylumKey().equals(b.getPhylumKey());
183    }
184
185    if (a.getSpeciesKey() == null || b.getSpeciesKey() == null) {
186      match = match && a.getSpeciesKey() == null && b.getSpeciesKey() == null;
187    } else {
188      match = match && a.getSpeciesKey().equals(b.getSpeciesKey());
189    }
190
191    return match;
192  }
