001package org.gbif.checklistbank.ws;
002
003import com.google.inject.AbstractModule;
004import com.google.inject.name.Names;
005
006import java.net.URI;
007
008public class UrlBindingModule extends AbstractModule {
009
010  private URI uri;
011  private String baseResourceUrl;
012
013  public UrlBindingModule(URI uri, String baseResourceUrl) {
014    this.uri = uri;
015    this.baseResourceUrl = baseResourceUrl;
016  }
017
018  /**
019   * Configures a {@link com.google.inject.Binder} via the exposed methods.
020   */
021  @Override
022  protected void configure() {
023    bindConstant().annotatedWith(Names.named("checklistbank.search.ws.url")).to(uri.toString() + baseResourceUrl);
024    bindConstant().annotatedWith(Names.named("checklistbank.ws.url")).to(uri.toString() + baseResourceUrl);
025  }
026}
