001package org.gbif.checklistbank.ws.resources;
002
003import org.gbif.api.model.checklistbank.NameUsage;
004import org.gbif.api.model.checklistbank.TypeSpecimen;
005import org.gbif.api.model.common.paging.Pageable;
006import org.gbif.api.model.common.paging.PagingRequest;
007import org.gbif.api.model.common.paging.PagingResponse;
008import org.gbif.api.service.checklistbank.DescriptionService;
009import org.gbif.api.service.checklistbank.DistributionService;
010import org.gbif.api.service.checklistbank.ImageService;
011import org.gbif.api.service.checklistbank.NameUsageService;
012import org.gbif.api.service.checklistbank.ReferenceService;
013import org.gbif.api.service.checklistbank.SpeciesProfileService;
014import org.gbif.api.service.checklistbank.TypeSpecimenService;
015import org.gbif.api.service.checklistbank.VernacularNameService;
016import org.gbif.api.vocabulary.TypeStatus;
017
018import java.util.LinkedList;
019import java.util.List;
020import java.util.Locale;
021
022import org.junit.Before;
023import org.junit.Test;
024
025import static org.junit.Assert.assertEquals;
026import static org.mockito.Mockito.mock;
027import static org.mockito.Mockito.when;
028
029public class SpeciesResourceTest {
030
031  private NameUsageService mockNameUsageService;
032  private VernacularNameService mockVernacularNameService;
033  private TypeSpecimenService mockTypeSpecimenService;
034  private SpeciesProfileService mockSpeciesProfileService;
035  private ReferenceService mockReferenceService;
036  private ImageService mockImageService;
037  private DescriptionService mockDescriptionService;
038  private DistributionService mockDistributionService;
039
040  private SpeciesResource resource;
041
042  private NameUsage nameUsage;
043  private Locale locale;
044  private static final int NAME_USAGE_KEY = 103350120;
045  private Pageable page;
046
047  @Before
048  public void setUp() throws Exception {
049    mockNameUsageService = mock(NameUsageService.class);
050    mockVernacularNameService = mock(VernacularNameService.class);
051    mockTypeSpecimenService = mock(TypeSpecimenService.class);
052    mockSpeciesProfileService = mock(SpeciesProfileService.class);
053    mockReferenceService = mock(ReferenceService.class);
054    mockImageService = mock(ImageService.class);
055    mockDescriptionService = mock(DescriptionService.class);
056    mockDistributionService = mock(DistributionService.class);
057
058    resource = new SpeciesResource(mockNameUsageService, mockVernacularNameService, mockTypeSpecimenService,
059      mockSpeciesProfileService, mockReferenceService, mockImageService, mockDescriptionService,
060      mockDistributionService);
061
062    locale = Locale.US;
063    nameUsage = new NameUsage();
064    nameUsage.setKey(NAME_USAGE_KEY);
065
066    page = new PagingRequest(0, 1);
067  }
068
069  @Test
070  public void testGet() throws Exception {
071    when(mockNameUsageService.get(NAME_USAGE_KEY, locale)).thenReturn(nameUsage);
072    NameUsage nameUsage = resource.get(NAME_USAGE_KEY, locale);
073    assertEquals(String.valueOf(NAME_USAGE_KEY), nameUsage.getKey().toString());
074  }
075
076  @Test
077  public void testListTypeSpecimensByNameUsage() {
078    TypeSpecimen ts = new TypeSpecimen();
079    ts.setUsageKey(NAME_USAGE_KEY);
080    ts.setTypeStatus(TypeStatus.PARATYPE);
081    List<TypeSpecimen> list = new LinkedList<TypeSpecimen>();
082    list.add(ts);
083    PagingResponse<TypeSpecimen> response = new PagingResponse<TypeSpecimen>(0, 1, 1l, list);
084
085    when(mockTypeSpecimenService.listByUsage(NAME_USAGE_KEY, page)).thenReturn(response);
086
087    PagingResponse<TypeSpecimen> result = resource.listTypeSpecimensByNameUsage(NAME_USAGE_KEY, page);
088    assertEquals(result.getResults().get(0).getUsageKey().toString(), String.valueOf(NAME_USAGE_KEY));
089  }
090
091}
