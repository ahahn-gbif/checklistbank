063  /**
064   * Starts up a solr server and indexes the test database.
065   * Wrapped in a static method so we can set the solr server in the ChecklistBankSearchWsTestModule
066   * which must have a default, empty constructor.
067   */
068  public static EmbeddedSolrReference setup() {
069    if (solrRef != null) {
070      return solrRef;
071    }
072
073    // Creates the injector, merging properties taken from default test indexing and checklistbank
074    try {
075      Properties props = PropertiesUtil.loadProperties(PROPERTY_INDEXER_DEFAULT);
076      Properties props2 = PropertiesUtil.loadProperties(ClientMyBatisITBase.PROPERTIES_FILE);
077      props.putAll(props2);
078      Injector injector = Guice.createInjector(new IndexingTestModule(props));
079
080      // Gets the indexer instance
081      solrRef = injector.getInstance(EmbeddedSolrReference.class);
082
083      // build the solr index
084      NameUsageIndexer nameUsageIndexer = injector.getInstance(NameUsageIndexer.class);
085      nameUsageIndexer.run();
086
087      return solrRef;
088
089    } catch (IOException e) {
090      throw new RuntimeException("Cant load properties to build solr index", e);
091    }
092  }

160  public void searchSearchFacets() {
161    // RANK
162    SearchResponse<NameUsageSearchResult, NameUsageSearchParameter> result = assertSearch("Sciurus vulgaris", NameUsageSearchParameter.RANK, 10L, null);
163    result = assertSearch("Sciurus vulgaris", NameUsageSearchParameter.RANK, Rank.VARIETY, 2l, null);

166    assertEquals("Sciurus vulgaris Linnaeus, 1758", result.getResults().get(0).getScientificName());
167
168    // CHECKLIST
169    result = assertSearch("Animalia", NameUsageSearchParameter.DATASET_KEY, 46L, null);

171      assertSearch("Animalia", NameUsageSearchParameter.DATASET_KEY, Constants.NUB_DATASET_KEY.toString(), 2l, null);
172    assertEquals((Integer) 1, result.getResults().get(0).getKey());
173    result = assertSearch("Animalia", NameUsageSearchParameter.DATASET_KEY, UUID.randomUUID().toString(), 0l, null);
174
175    // HIGHERTAXON
176    result = assertSearch("Rodentia", NameUsageSearchParameter.HIGHERTAXON_KEY, 42L, null);
177    result = assertSearch("Rodentia", NameUsageSearchParameter.HIGHERTAXON_KEY, "1", 1l, null);
178    assertEquals((Integer) 10, result.getResults().get(0).getKey());
179    result = assertSearch("Rodentia", NameUsageSearchParameter.HIGHERTAXON_KEY, "100000024", 0l, null);
180
181    // TAXSTATUS
182    result = assertSearch("Sciurus", NameUsageSearchParameter.STATUS, 30L, null);
183    result = assertSearch("Sciurus", NameUsageSearchParameter.STATUS, TaxonomicStatus.SYNONYM, 15l, null);
184    result = assertSearch("Sciurus", NameUsageSearchParameter.STATUS, TaxonomicStatus.HETEROTYPIC_SYNONYM, 1l, null);
185    result = assertSearch("Sciurus", NameUsageSearchParameter.STATUS, TaxonomicStatus.MISAPPLIED, 0l, null);
186
187    // EXTINCT
188    result = assertSearch("Sciurus", NameUsageSearchParameter.EXTINCT, 30L, null);
189    result = assertSearch("Sciurus", NameUsageSearchParameter.EXTINCT, "false", 1l, null);
190    assertEquals((Integer) 100000025, result.getResults().get(0).getKey());
191    result = assertSearch("Sciurus", NameUsageSearchParameter.EXTINCT, "true", 0l, null);
192
193    // MARINE
194    result = assertSearch("Sciurus", NameUsageSearchParameter.HABITAT, 30L, null);
195    result = assertSearch("Sciurus", NameUsageSearchParameter.HABITAT, "false", 1l, null);
196    assertEquals((Integer) 100000025, result.getResults().get(0).getKey());
197    result = assertSearch("Sciurus", NameUsageSearchParameter.HABITAT, "true", 0l, null);
198

200    result = assertSearch("Sciurillus", NameUsageSearchParameter.THREAT, 5L, null);
201    result = assertSearch("Sciurillus", NameUsageSearchParameter.THREAT, ThreatStatus.NEAR_THREATENED, 1l, null);
202    assertEquals((Integer) 100000007, result.getResults().get(0).getKey());
203    result = assertSearch("Sciurillus", NameUsageSearchParameter.THREAT, ThreatStatus.EXTINCT_IN_THE_WILD, 0l, null);
204
205    // HIGHERTAXON & RANK
206    NameUsageSearchRequest searchRequest = new NameUsageSearchRequest(0L, 20);
207    searchRequest.setQ("Sciurus");
208    searchRequest.addFacets(NameUsageSearchParameter.HIGHERTAXON_KEY, NameUsageSearchParameter.RANK);
209    assertSearch(searchRequest, true, 30l, null);
210
211    searchRequest.addParameter(NameUsageSearchParameter.HIGHERTAXON_KEY, "100000025");
212    assertSearch(searchRequest, true, 14l, null);
213
214    searchRequest.addParameter(NameUsageSearchParameter.RANK, Rank.VARIETY);
215    assertSearch(searchRequest, true, 2l, null);
216  }

219  private SearchResponse<NameUsageSearchResult, NameUsageSearchParameter> assertSearch(String q, NameUsageSearchParameter facet,
220    Long expectedCount, String expectedFacetCounts) {
221    return assertSearch(q, facet, (String) null, expectedCount, expectedFacetCounts);
222  }
223
224  private SearchResponse<NameUsageSearchResult, NameUsageSearchParameter> assertSearch(String q, NameUsageSearchParameter facet,
225    Enum<?> facetFilter, Long expectedCount, String expectedFacetCounts) {
226
227    NameUsageSearchRequest searchRequest = prepareRequest(q, facet);
228    if (facetFilter != null) {
229      searchRequest.addParameter(facet, facetFilter);
230    }
231
232    return assertSearch(searchRequest, facet != null, expectedCount, expectedFacetCounts);
233  }
234
235  private SearchResponse<NameUsageSearchResult, NameUsageSearchParameter> assertSearch(String q, NameUsageSearchParameter facet,
236    String facetFilter, Long expectedCount, String expectedFacetCounts) {
237
238    NameUsageSearchRequest searchRequest = prepareRequest(q, facet);
239    if (facetFilter != null) {
240      searchRequest.addParameter(facet, facetFilter);
241    }
242    return assertSearch(searchRequest, facet != null, expectedCount, expectedFacetCounts);
243  }
244
245  private NameUsageSearchRequest prepareRequest(String q, NameUsageSearchParameter facet) {
246    // build request
247    NameUsageSearchRequest searchRequest = new NameUsageSearchRequest(0L, 20);
248    searchRequest.setQ(q);
249
250    if (facet != null) {
251      searchRequest.addFacets(facet);
252    }
253
254    return searchRequest;
255  }
256  private SearchResponse<NameUsageSearchResult, NameUsageSearchParameter> assertSearch(NameUsageSearchRequest searchRequest, boolean useFacets,
257    Long expectedCount, String expectedFacetCounts) {
258
259    // query
260    SearchResponse<NameUsageSearchResult, NameUsageSearchParameter> response = wsClient.search(searchRequest);
261
262    // assert
263    if (expectedCount != null) {
264      assertEquals(expectedCount, response.getCount());
265    }
266
267    if (useFacets && expectedFacetCounts != null) {
268      assertEquals(1, response.getFacets().size());
269      assertEquals(expectedFacetCounts, response.getFacets().get(0).getCounts().size());
270    }
271    if (!useFacets) {
272      Assert.assertTrue(response.getFacets().isEmpty());
273    }
274
275    for (NameUsageSearchResult u : response.getResults()) {
276      assertNotNull(u);
277      assertEquals(NameUsageSearchResult.class, u.getClass());
278    }
279
280    return response;
281  }
