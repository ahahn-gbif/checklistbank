140  @Test
141  public void testGetMybatisStyle() {
142    final NameUsage rodentia = wsClient.get(100000004, Locale.UK);
143    final NameUsageMetrics rodentiaM = wsClient.getMetrics(100000004);
144    assertNotNull(rodentia);
145    assertEquals((Integer) 10, rodentia.getNubKey());
146    assertFalse(rodentia.isSynonym());
147    assertEquals("1000", rodentia.getSourceId());
148    assertEquals("Rodentia", rodentia.getCanonicalName());
149    assertEquals("Rodentia Bowdich, 1821", rodentia.getScientificName());
150    assertEquals("Bowdich, 1821", rodentia.getAuthorship());
151    assertEquals(Rank.ORDER, rodentia.getRank());
152    assertEquals((Integer) 100000003, rodentia.getParentKey());
153
154    assertEquals("Animalia", rodentia.getKingdom());
155    assertEquals((Integer) 100000001, rodentia.getKingdomKey());
156
157    assertEquals("Chordata", rodentia.getPhylum());
158    assertEquals((Integer) 100000002, rodentia.getPhylumKey());
159    assertEquals(0, rodentiaM.getNumPhylum());
160
161    assertEquals("Mammalia", rodentia.getClazz());
162    assertEquals((Integer) 100000003, rodentia.getClassKey());
163    assertEquals(0, rodentiaM.getNumClass());
164
165    assertEquals(1, rodentiaM.getNumOrder());
166    assertEquals(1, rodentiaM.getNumFamily());
167    assertEquals(2, rodentiaM.getNumGenus());
168    assertEquals(3, rodentiaM.getNumSpecies());
169    assertEquals(8, rodentiaM.getNumSynonyms());
170    assertEquals(1, rodentiaM.getNumChildren());
171    assertEquals(24, rodentia.getNumDescendants());
172
173    assertEquals(Origin.SOURCE, rodentia.getOrigin());
174
175    assertEquals(UUID.fromString("109aea14-c252-4a85-96e2-f5f4d5d088f4"), rodentia.getDatasetKey());
176    assertNull(rodentia.getPublishedIn());
177
178    // identifiers are eagerly loaded
179    // <identifier id="4" usage_fk="100000004" identifier="1000" type_fk="2007"/>
180    assertEquals(1, rodentia.getIdentifiers().size());
181    assertEquals(rodentia.getKey(), rodentia.getIdentifiers().get(0).getUsageKey());
182    assertEquals("1000", rodentia.getIdentifiers().get(0).getIdentifier());
183    assertEquals(IdentifierType.SOURCE_ID, rodentia.getIdentifiers().get(0).getType());
184    assertEquals("Hau Ruck", rodentia.getIdentifiers().get(0).getTitle());
185
186
187    NameUsage squirrel = wsClient.get(100000025, Locale.UK);
188    final NameUsageMetrics squirrelM = wsClient.getMetrics(100000025);
189    assertNotNull(squirrel);
190    assertNull(squirrel.getNubKey());
191    assertFalse(squirrel.isSynonym());
192    assertEquals("Sciurus vulgaris", squirrel.getCanonicalName());
193    assertEquals("Sciurus vulgaris Linnaeus, 1758", squirrel.getScientificName());
194    assertEquals("Linnaeus, 1758", squirrel.getAuthorship());
195    assertEquals("Eurasian Red Squirrel", squirrel.getVernacularName());
196    assertEquals(Rank.SPECIES, squirrel.getRank());
197    assertEquals((Integer) 100000024, squirrel.getParentKey());
198
199    assertEquals("Animalia", squirrel.getKingdom());
200    assertEquals((Integer) 100000001, squirrel.getKingdomKey());
201
202    assertEquals("Chordata", squirrel.getPhylum());
203    assertEquals((Integer) 100000002, squirrel.getPhylumKey());
204    assertEquals(0, squirrelM.getNumPhylum());
205
206    assertEquals("Mammalia", squirrel.getClazz());
207    assertEquals((Integer) 100000003, squirrel.getClassKey());
208    assertEquals(0, squirrelM.getNumClass());
209
210    assertEquals("Rodentia", squirrel.getOrder());
211    assertEquals((Integer) 100000004, squirrel.getOrderKey());
212    assertEquals(0, squirrelM.getNumOrder());
213
214    assertEquals("Sciuridae", squirrel.getFamily());
215    assertEquals((Integer) 100000005, squirrel.getFamilyKey());
216    assertEquals(0, squirrelM.getNumFamily());
217
218    assertEquals("Sciurus", squirrel.getGenus());
219    assertEquals((Integer) 100000011, squirrel.getGenusKey());
220    assertEquals(0, squirrelM.getNumGenus());
221
222    assertEquals(1, squirrelM.getNumSpecies());
223    assertEquals(9, squirrelM.getNumChildren());
224    assertEquals(4, squirrelM.getNumSynonyms());
225
226    assertEquals(UUID.fromString("109aea14-c252-4a85-96e2-f5f4d5d088f4"), squirrel.getDatasetKey());
227    assertEquals("Syst. Nat. , 10th ed. vol. 1 p. 63", squirrel.getPublishedIn());
228
229    assertEquals("2010030", squirrel.getSourceId());
230
231
232    // TEST VERNACULAR
233    squirrel = wsClient.get(100000040, null);
234    assertNull(squirrel.getVernacularName());
235
236    squirrel = wsClient.get(100000040, Locale.GERMANY);
237    assertEquals("Kaukasischen Eichhörnchen", squirrel.getVernacularName());
238
239    // TEST non existing language VERNACULAR
240    squirrel = wsClient.get(100000040, Locale.CHINESE);
241    assertEquals("Caucasian Squirrel", squirrel.getVernacularName());
242
243    // TEST MULTIPLE IDENTIFIERS
244    squirrel = wsClient.get(100000007, Locale.GERMANY);
245    Set<Integer> ids = Sets.newHashSet(7, 102, 103, 104, 105);
246    for (Identifier id : squirrel.getIdentifiers()) {
247      assertEquals(squirrel.getKey(), id.getUsageKey());
248      switch (id.getKey()) {
249        case 7:
250          assertEquals("6905528", id.getIdentifier());
251          assertEquals(IdentifierType.SOURCE_ID, id.getType());
252          assertEquals("Bonjour mon capitain", id.getTitle());
253          break;
254        case 102:
255          assertEquals("http://www.catalogueoflife.org/details/species/id/6905528", id.getIdentifier());
256          assertEquals(IdentifierType.URL, id.getType());
257          assertNull(id.getTitle());
258          break;
259        case 103:
260          assertEquals("urn:lsid:catalogueoflife.org:taxon:df0a319c-29c1-102b-9a4a-00304854f820:col20120721",
261            id.getIdentifier());
262          assertEquals(IdentifierType.LSID, id.getType());
263          assertNull(id.getTitle());
264          break;
265        case 104:
266          assertEquals("doi:10.1038/6905528", id.getIdentifier());
267          assertEquals(IdentifierType.DOI, id.getType());
268          assertNull(id.getTitle());
269          break;
270        case 105:
271          assertEquals("http://www.itis.gov/servlet/SingleRpt/SingleRpt?search_topic=TSN&search_value=632417",
272            id.getIdentifier());
273          assertEquals(IdentifierType.URL, id.getType());
274          assertNull(id.getTitle());
275          break;
276        default:
277          fail("Unkown Identifier");
278      }
279      ids.remove(id.getKey());
280    }
281    assertTrue(ids.isEmpty());
282
283    // TEST SYNONYM
284    NameUsage syn = wsClient.get(100000027, Locale.FRENCH);
285    assertNotNull(syn);
286    assertTrue(syn.isSynonym());
287    assertEquals("Sciurus nadymensis", syn.getCanonicalName());
288    assertEquals("Sciurus nadymensis Serebrennikov, 1928", syn.getScientificName());
289    assertNull(syn.getVernacularName());
290    assertEquals(Rank.SPECIES, syn.getRank());
291    assertEquals((Integer) 100000024, syn.getParentKey());
292    assertEquals((Integer) 100000025, syn.getAcceptedKey());
293    assertEquals("Sciurus vulgaris Linnaeus, 1758", syn.getAccepted());
294    assertFalse(syn.isProParte());
295  }
