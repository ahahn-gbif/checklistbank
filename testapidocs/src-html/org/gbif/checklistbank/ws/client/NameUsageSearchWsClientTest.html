013package org.gbif.checklistbank.ws.client;
014
015import org.gbif.api.model.checklistbank.search.NameUsageSearchParameter;
016import org.gbif.api.model.checklistbank.search.NameUsageSearchRequest;
017import org.gbif.api.model.checklistbank.search.NameUsageSearchResult;
018import org.gbif.api.model.common.search.SearchResponse;
019
020import javax.ws.rs.core.MultivaluedMap;
021
022import com.sun.jersey.api.client.GenericType;
023import com.sun.jersey.api.client.WebResource;
024import org.junit.Assert;
025import org.junit.Before;
026import org.junit.Test;
027import org.junit.runner.RunWith;
028import org.mockito.Matchers;
029import org.mockito.Mock;
030import org.powermock.api.mockito.PowerMockito;
031import org.powermock.core.classloader.annotations.PrepareForTest;
032import org.powermock.modules.junit4.PowerMockRunner;
033
034import static org.gbif.api.model.common.paging.PagingConstants.DEFAULT_PARAM_LIMIT;
035import static org.gbif.api.model.common.paging.PagingConstants.DEFAULT_PARAM_OFFSET;
036
037import static org.mockito.Matchers.any;
038import static org.mockito.Mockito.when;
039
040@RunWith(PowerMockRunner.class)
041public class NameUsageSearchWsClientTest {
042
043  @Mock
044  private WebResource resource;
045  private NameUsageSearchWsClient client;
046
047
048  @Before
049  public void setUp() {
050    client = new NameUsageSearchWsClient(resource);
051  }
052
053  @Test
054  @PrepareForTest(WebResource.Builder.class)
055  public void testList() {
056    WebResource.Builder builder = PowerMockito.mock(WebResource.Builder.class);
057    NameUsageSearchRequest searchRequest = new NameUsageSearchRequest(DEFAULT_PARAM_OFFSET, DEFAULT_PARAM_LIMIT);
058    searchRequest.setQ("*");
059    when(resource.type(any(String.class))).thenReturn(builder);
060    when(resource.accept(any(String.class))).thenReturn(builder);
061    when(resource.path(any(String.class))).thenReturn(resource);
062    when(resource.queryParam(Matchers.<String>any(), Matchers.<String>any())).thenReturn(resource);
063    when(resource.queryParams(any(MultivaluedMap.class))).thenReturn(resource);
064    // there is no path, it uses the default URL /checklist
065    SearchResponse<NameUsageSearchResult, NameUsageSearchParameter> defaultResponse =
066      new SearchResponse<NameUsageSearchResult, NameUsageSearchParameter>(DEFAULT_PARAM_OFFSET, DEFAULT_PARAM_LIMIT);
067    when(builder.get(any(GenericType.class))).thenReturn(defaultResponse);
068    when(resource.get(any(GenericType.class))).thenReturn(defaultResponse);
069
070    SearchResponse<NameUsageSearchResult, NameUsageSearchParameter> response = client.search(searchRequest);
071    Assert.assertNotNull(response);
072  }
073}
