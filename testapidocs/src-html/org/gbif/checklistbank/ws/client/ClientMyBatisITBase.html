001package org.gbif.checklistbank.ws.client;
002
003import org.gbif.api.service.checklistbank.NameUsageService;
004import org.gbif.checklistbank.service.mybatis.postgres.DatabaseDrivenChecklistBankTestRule;
005import org.gbif.checklistbank.ws.UrlBindingModule;
006import org.gbif.checklistbank.ws.client.guice.ChecklistBankWsClientModule;
007import org.gbif.checklistbank.ws.guice.ChecklistBankWsModule;
008import org.gbif.utils.file.properties.PropertiesUtil;
009import org.gbif.ws.client.BaseResourceTest;
010
011import java.io.IOException;
012import java.util.UUID;
013
014import com.google.inject.Guice;
015import com.google.inject.Injector;
016import org.junit.Before;
017import org.junit.Rule;
018
019/**
020 * Base class for webservice client integration tests that access the mybatis based webservices.
021 *
022 * @param <T> the specific client to test.
023 */
024public class ClientMyBatisITBase<T> extends BaseResourceTest {
025
026  protected static final String PROPERTIES_FILE = "checklistbank.properties";
027
028  private Injector clientInjector;
029  protected T wsClient;
030  private final Class<T> wsClientClass;
031  protected final UUID CHECKLIST_KEY_1001 = UUID.fromString("109aea14-c252-4a85-96e2-f5f4d5d088f4");
032
033
034  @Rule
035  public DatabaseDrivenChecklistBankTestRule<NameUsageService> ddt =
036    new DatabaseDrivenChecklistBankTestRule<NameUsageService>(NameUsageService.class);
037
038  public ClientMyBatisITBase(Class<T> wsClientClass) {
039    super("org.gbif.checklistbank.ws", "clb", ChecklistBankWsModule.class);
040    this.wsClientClass = wsClientClass;
041  }
042
043  @Before
044  public void init() throws IOException {
045    clientInjector = Guice.createInjector(new UrlBindingModule(getBaseURI(), contextPath),
046      new ChecklistBankWsClientModule(PropertiesUtil.loadProperties(PROPERTIES_FILE), false, true, false));
047    wsClient = clientInjector.getInstance(wsClientClass);
048  }
049}
