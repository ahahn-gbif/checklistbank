001package org.gbif.checklistbank.ws.client;
002
003import org.gbif.api.model.common.paging.Pageable;
004import org.gbif.api.model.common.paging.PagingRequest;
005import org.gbif.api.model.common.paging.PagingResponse;
006import org.gbif.ws.client.BaseWsGetClient;
007
008import java.util.List;
009import java.util.Locale;
010import javax.ws.rs.core.MediaType;
011import javax.ws.rs.core.MultivaluedMap;
012
013import com.google.common.collect.Lists;
014import com.sun.jersey.api.client.WebResource;
015import org.junit.Before;
016import org.junit.Test;
017import org.junit.runner.RunWith;
018import org.mockito.Matchers;
019import org.powermock.core.classloader.annotations.PrepareForTest;
020import org.powermock.modules.junit4.PowerMockRunner;
021
022import static org.junit.Assert.assertEquals;
023import static org.mockito.Matchers.eq;
024import static org.mockito.Mockito.when;
025import static org.powermock.api.mockito.PowerMockito.mock;
026
027@RunWith(PowerMockRunner.class)
028@PrepareForTest(WebResource.Builder.class)
029public abstract class WsClientBaseTest<T> {
030
031  protected final Class<T> resourceClass;
032  protected WebResource resource;
033  protected WebResource.Builder resourceBuilder;
034  private BaseWsGetClient<T, Integer> client;
035  protected final Pageable page = new PagingRequest(0l, 1);
036  protected static final int key = 1111;
037
038  public WsClientBaseTest(Class<T> resourceClass) {
039    this.resourceClass = resourceClass;
040  }
041
042  @Before
043  public void setUp() {
044    resource = mock(WebResource.class);
045    when(resource.path(Matchers.<String>any())).thenReturn(resource);
046    when(resource.queryParam(Matchers.<String>any(), Matchers.<String>any())).thenReturn(resource);
047    when(resource.queryParams(Matchers.<MultivaluedMap<String, String>>any())).thenReturn(resource);
048
049    resourceBuilder = mock(WebResource.Builder.class);
050    when(resource.type(eq(MediaType.APPLICATION_JSON))).thenReturn(resourceBuilder);
051
052    when(resourceBuilder.acceptLanguage(Matchers.<Locale>any())).thenReturn(resourceBuilder);
053    when(resourceBuilder.type(eq(MediaType.APPLICATION_JSON))).thenReturn(resourceBuilder);
054
055    client = getClient(resource);
056    when(resource.get(resourceClass)).thenReturn(getMockObject());
057    when(resourceBuilder.get(eq(resourceClass))).thenReturn(getMockObject());
058  }
059
060  protected PagingResponse<T> getMockResponse() {
061    List<T> results = Lists.newArrayList();
062    T obj = getMockObject();
063    results.add(obj);
064
065    return new PagingResponse<T>(page, 1L, results);
066  }
067
068  /**
069   * @return mocked object with key and usageKey being set in this class
070   */
071  abstract T getMockObject();
072
073  abstract BaseWsGetClient<T, Integer> getClient(WebResource resource);
074
075  @Test
076  public void testGet() {
077    T obj = client.get(key);
078    T obj2 = getMockObject();
079    assertEquals(obj2, obj);
080  }
081
082}
