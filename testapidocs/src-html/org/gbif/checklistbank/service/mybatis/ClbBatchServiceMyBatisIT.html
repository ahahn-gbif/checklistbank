001package org.gbif.checklistbank.service.mybatis;
002
003import org.gbif.api.model.checklistbank.NameUsage;
004import org.gbif.checklistbank.service.UsageService;
005import org.gbif.checklistbank.service.mybatis.postgres.DatabaseDrivenChecklistBankTestRule;
006
007import java.util.List;
008import java.util.Set;
009import java.util.UUID;
010
011import com.google.common.collect.Sets;
012import org.junit.Rule;
013import org.junit.Test;
014
015import static org.junit.Assert.assertEquals;
016import static org.junit.Assert.assertNotNull;
017import static org.junit.Assert.assertNull;
018import static org.junit.Assert.assertTrue;
019import static org.junit.Assert.fail;
020
021public class ClbBatchServiceMyBatisIT {
022
023  private static final UUID CHECKLIST_KEY = UUID.fromString("109aea14-c252-4a85-96e2-f5f4d5d088f4");
024
025  @Rule
026  public DatabaseDrivenChecklistBankTestRule<UsageService> ddt =
027    new DatabaseDrivenChecklistBankTestRule<UsageService>(UsageService.class);
028
029   @Test
030  public void testListAll() {
031    List<Integer> ids = ddt.getService().listAll();
032    assertEquals(46, ids.size());
033  }
034
035  @Test
036  public void testListRange() {
037    List<NameUsage> usages = ddt.getService().listRange(100000001, 100000020);
038    assertEquals(20, usages.size());
039
040    boolean found = false;
041    for (NameUsage nu : usages) {
042      assertNull(nu.getVernacularName());
043      assertNotNull(nu.getScientificName());
044      assertTrue(nu.getKey() >= 100000001 && nu.getKey() <= 100000020);
045      assertEquals(CHECKLIST_KEY, nu.getDatasetKey());
046
047      if (nu.getKey().equals(100000007)) {
048        found = true;
049        assertEquals("6905528", nu.getSourceId());
050        Set<String> expectedLinks = Sets.newHashSet("http://www.catalogueoflife.org/details/species/id/6905528",
051          "http://www.itis.gov/servlet/SingleRpt/SingleRpt?search_topic=TSN&search_value=632417");
052        assertTrue(expectedLinks.contains(nu.getLink()));
053      }
054    }
055    if (!found) {
056      fail("usage 100000007 missing in range result");
057    }
058  }
059
060}
