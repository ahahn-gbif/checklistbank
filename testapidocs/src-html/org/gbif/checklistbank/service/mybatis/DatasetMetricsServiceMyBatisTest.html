001package org.gbif.checklistbank.service.mybatis;
002
003import java.util.Collections;
004import java.util.List;
005
006import com.google.common.collect.Lists;
007import org.junit.Test;
008
009import static org.junit.Assert.assertEquals;
010
011public class DatasetMetricsServiceMyBatisTest {
012
013  @Test
014  public void testPercentage() {
015    assertEquals(9, DatasetMetricsServiceMyBatis.getPercentage(103, 1078));
016    assertEquals(9, DatasetMetricsServiceMyBatis.getPercentage(90, 1000));
017    assertEquals(0, DatasetMetricsServiceMyBatis.getPercentage(1, 1078));
018    assertEquals(0, DatasetMetricsServiceMyBatis.getPercentage(0, 1078));
019    assertEquals(99, DatasetMetricsServiceMyBatis.getPercentage(1077, 1078));
020    assertEquals(100, DatasetMetricsServiceMyBatis.getPercentage(1077, 1077));
021  }
022
023  @Test
024  public void testCount() {
025    List<DatasetMetricsServiceMyBatis.Count<String>> counts = Lists.newArrayList();
026
027    counts.add(new DatasetMetricsServiceMyBatis.Count<String>("c", 19));
028    counts.add(new DatasetMetricsServiceMyBatis.Count<String>("d", 9));
029    counts.add(new DatasetMetricsServiceMyBatis.Count<String>("b", 229));
030    counts.add(new DatasetMetricsServiceMyBatis.Count<String>("few", 3214129));
031    counts.add(new DatasetMetricsServiceMyBatis.Count<String>("a", -4));
032    counts.add(new DatasetMetricsServiceMyBatis.Count<String>("g", -4));
033
034    Collections.sort(counts);
035
036    assertEquals("g", counts.get(5).getKey());
037    assertEquals("a", counts.get(4).getKey());
038    assertEquals("d", counts.get(3).getKey());
039    assertEquals("c", counts.get(2).getKey());
040    assertEquals("b", counts.get(1).getKey());
041    assertEquals("few", counts.get(0).getKey());
042  }
043
044}
