001package org.gbif.checklistbank.service.mybatis;
002
003import org.gbif.api.model.checklistbank.Image;
004import org.gbif.api.model.common.paging.Pageable;
005import org.gbif.api.model.common.paging.PagingRequest;
006import org.gbif.api.service.checklistbank.ImageService;
007import org.gbif.checklistbank.service.mybatis.postgres.DatabaseDrivenChecklistBankTestRule;
008
009import java.text.ParseException;
010import java.util.List;
011import java.util.UUID;
012
013import org.junit.Rule;
014import org.junit.Test;
015
016import static org.junit.Assert.assertEquals;
017import static org.junit.Assert.assertNull;
018
019public class ImageServiceMyBatisIT {
020
021  private final Integer USAGE_ID = 100000025;
022  private final UUID SQUIRREL_UUID = UUID.fromString("109aea14-c252-4a85-96e2-f5f4d5d088f4");
023
024  @Rule
025  public DatabaseDrivenChecklistBankTestRule<ImageService> ddt =
026    new DatabaseDrivenChecklistBankTestRule<ImageService>(ImageService.class);
027
028  @Test
029  public void testGet() throws ParseException {
030    Image image = ddt.getService().get(100010);
031    assertEquals(USAGE_ID, image.getUsageKey());
032    assertEquals(SQUIRREL_UUID, image.getDatasetKey());
033    assertEquals("2008-01-01", image.getCreated());
034    assertEquals("Prashanthns", image.getCreator());
035    assertEquals("Eurasian Red Squirrel", image.getTitle());
036    assertEquals("A Eurasian Red Squirrel seen from below. Photographed in Wagenigen, Nederlands.", image.getDescription());
037    assertEquals("http://upload.wikimedia.org/wikipedia/commons/e/e2/Eurasian_Red_Squirrel.jpg", image.getImage());
038    assertEquals("Creative Commons Attribution 3.0 Unported", image.getLicense());
039    assertNull(image.getPublisher());
040    assertEquals("http://en.wikipedia.org/wiki/File:Eurasian_Red_Squirrel.jpg", image.getLink());
041  }
042
043  @Test
044  public void testListByChecklistUsage() {
045    List<Image> images = ddt.getService().listByUsage(USAGE_ID, null).getResults();
046    assertEquals(8, images.size());
047    assertEquals((Integer) 100010, images.get(0).getKey());
048    assertEquals((Integer) 100011, images.get(1).getKey());
049    assertEquals((Integer) 100012, images.get(2).getKey());
050    assertEquals((Integer) 100013, images.get(3).getKey());
051
052    // TEST PAGING
053    Pageable page = new PagingRequest(0, 1);
054    Image d1 = ddt.getService().listByUsage(USAGE_ID, page).getResults().get(0);
055
056    page = new PagingRequest(1, 1);
057    Image d2 = ddt.getService().listByUsage(USAGE_ID, page).getResults().get(0);
058    assertEquals(d1, images.get(0));
059    assertEquals(d2, images.get(1));
060  }
061}
