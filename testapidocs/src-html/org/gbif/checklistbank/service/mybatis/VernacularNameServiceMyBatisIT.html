001package org.gbif.checklistbank.service.mybatis;
002
003import org.gbif.api.model.checklistbank.VernacularName;
004import org.gbif.api.model.common.paging.Pageable;
005import org.gbif.api.model.common.paging.PagingRequest;
006import org.gbif.api.service.checklistbank.VernacularNameService;
007import org.gbif.api.vocabulary.Country;
008import org.gbif.api.vocabulary.Language;
009import org.gbif.checklistbank.service.mybatis.postgres.DatabaseDrivenChecklistBankTestRule;
010
011import java.util.List;
012
013import org.junit.Rule;
014import org.junit.Test;
015
016import static org.junit.Assert.assertEquals;
017import static org.junit.Assert.assertNotNull;
018
019public class VernacularNameServiceMyBatisIT {
020
021  @Rule
022  public DatabaseDrivenChecklistBankTestRule<VernacularNameService> ddt =
023    new DatabaseDrivenChecklistBankTestRule<VernacularNameService>(VernacularNameService.class);
024
025  @Test
026  public void testGet() {
027    VernacularName squirrel = ddt.getService().get(100007);
028    assertEquals("Eurasian Red Squirrel", squirrel.getVernacularName());
029    assertEquals(Language.ENGLISH, squirrel.getLanguage());
030    assertEquals(Country.GERMANY, squirrel.getCountry());
031    assertEquals((Integer) 100000025, squirrel.getUsageKey());
032
033    squirrel = ddt.getService().get(100010);
034    assertEquals("Kaukasischen Eichhörnchen", squirrel.getVernacularName());
035    assertEquals(Language.GERMAN, squirrel.getLanguage());
036    assertEquals((Integer) 100000040, squirrel.getUsageKey());
037  }
038
039  @Test
040  public void testListByChecklistUsage() {
041    // TEST VERNACULAR
042    List<VernacularName> squirrels = ddt.getService().listByUsage(100000025, null).getResults();
043    assertEquals(2, squirrels.size());
044    assertEquals("Eurasian Red Squirrel", squirrels.get(0).getVernacularName());
045    assertEquals(Language.ENGLISH, squirrels.get(0).getLanguage());
046    assertEquals("Europäisches Eichhörnchen", squirrels.get(1).getVernacularName());
047    assertEquals(Language.GERMAN, squirrels.get(1).getLanguage());
048    // TEST PAGING
049    Pageable page = new PagingRequest(0, 1);
050    VernacularName v1 = ddt.getService().listByUsage(100000025, page).getResults().get(0);
051
052    page = new PagingRequest(1, 1);
053    VernacularName v2 = ddt.getService().listByUsage(100000025, page).getResults().get(0);
054    assertEquals(v1, squirrels.get(0));
055    assertEquals(v2, squirrels.get(1));
056  }
057
058  @Test
059  public void testListByRange() {
060    // TEST VERNACULAR
061    List<VernacularName> records = ((VernacularNameServiceMyBatis) ddt.getService()).listRange(1, 100000020);
062    assertEquals(2, records.size());
063    for (VernacularName v : records) {
064      assertNotNull(v.getUsageKey());
065      assertNotNull(v.getVernacularName());
066    }
067  }
068
069}
