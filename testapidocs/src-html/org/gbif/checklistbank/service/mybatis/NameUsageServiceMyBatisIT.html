041  @Test
042  public void testGet() {
043    final NameUsage rodentia = ddt.getService().get(100000004, Locale.UK);
044    final NameUsageMetrics rodentiaM = ddt.getService().getMetrics(100000004);
045    assertNotNull(rodentia);
046    assertEquals((Integer) 10, rodentia.getNubKey());
047    assertFalse(rodentia.isSynonym());
048    assertEquals("1000", rodentia.getSourceId());
049    assertEquals("Rodentia", rodentia.getCanonicalName());
050    assertEquals("Rodentia Bowdich, 1821", rodentia.getScientificName());
051    assertEquals("Bowdich, 1821", rodentia.getAuthorship());
052    assertEquals(Rank.ORDER, rodentia.getRank());
053    assertEquals((Integer) 100000003, rodentia.getParentKey());
054
055    assertEquals("Animalia", rodentia.getKingdom());
056    assertEquals((Integer) 100000001, rodentia.getKingdomKey());
057
058    assertEquals("Chordata", rodentia.getPhylum());
059    assertEquals((Integer) 100000002, rodentia.getPhylumKey());
060    assertEquals(0, rodentiaM.getNumPhylum());
061
062    assertEquals("Mammalia", rodentia.getClazz());
063    assertEquals((Integer) 100000003, rodentia.getClassKey());
064    assertEquals(0, rodentiaM.getNumClass());
065
066    assertEquals(1, rodentiaM.getNumOrder());
067    assertEquals(1, rodentiaM.getNumFamily());
068    assertEquals(2, rodentiaM.getNumGenus());
069    assertEquals(3, rodentiaM.getNumSpecies());
070    assertEquals(8, rodentiaM.getNumSynonyms());
071    assertEquals(1, rodentiaM.getNumChildren());
072    assertEquals(24, rodentia.getNumDescendants());
073
074    assertEquals(Origin.SOURCE, rodentia.getOrigin());
075
076    assertEquals(UUID.fromString("109aea14-c252-4a85-96e2-f5f4d5d088f4"), rodentia.getDatasetKey());
077    assertNull(rodentia.getPublishedIn());
078
079    // identifiers are eagerly loaded
080    // <identifier id="4" usage_fk="100000004" identifier="1000" type_fk="2007"/>
081    assertEquals(1, rodentia.getIdentifiers().size());
082    assertEquals(rodentia.getKey(), rodentia.getIdentifiers().get(0).getUsageKey());
083    assertEquals("1000", rodentia.getIdentifiers().get(0).getIdentifier());
084    assertEquals(IdentifierType.SOURCE_ID, rodentia.getIdentifiers().get(0).getType());
085    assertEquals("Hau Ruck", rodentia.getIdentifiers().get(0).getTitle());
086
087
088    NameUsage squirrel = ddt.getService().get(100000025, Locale.UK);
089    final NameUsageMetrics squirrelM = ddt.getService().getMetrics(100000025);
090    assertNotNull(squirrel);
091    assertNull(squirrel.getNubKey());
092    assertFalse(squirrel.isSynonym());
093    assertEquals("Sciurus vulgaris", squirrel.getCanonicalName());
094    assertEquals("Sciurus vulgaris Linnaeus, 1758", squirrel.getScientificName());
095    assertEquals("Linnaeus, 1758", squirrel.getAuthorship());
096    assertEquals("Eurasian Red Squirrel", squirrel.getVernacularName());
097    assertEquals(Rank.SPECIES, squirrel.getRank());
098    assertEquals((Integer) 100000024, squirrel.getParentKey());
099
100    assertEquals("Animalia", squirrel.getKingdom());
101    assertEquals((Integer) 100000001, squirrel.getKingdomKey());
102
103    assertEquals("Chordata", squirrel.getPhylum());
104    assertEquals((Integer) 100000002, squirrel.getPhylumKey());
105    assertEquals(0, squirrelM.getNumPhylum());
106
107    assertEquals("Mammalia", squirrel.getClazz());
108    assertEquals((Integer) 100000003, squirrel.getClassKey());
109    assertEquals(0, squirrelM.getNumClass());
110
111    assertEquals("Rodentia", squirrel.getOrder());
112    assertEquals((Integer) 100000004, squirrel.getOrderKey());
113    assertEquals(0, squirrelM.getNumOrder());
114
115    assertEquals("Sciuridae", squirrel.getFamily());
116    assertEquals((Integer) 100000005, squirrel.getFamilyKey());
117    assertEquals(0, squirrelM.getNumFamily());
118
119    assertEquals("Sciurus", squirrel.getGenus());
120    assertEquals((Integer) 100000011, squirrel.getGenusKey());
121    assertEquals(0, squirrelM.getNumGenus());
122
123    assertEquals(1, squirrelM.getNumSpecies());
124    assertEquals(9, squirrelM.getNumChildren());
125    assertEquals(4, squirrelM.getNumSynonyms());
126
127    assertEquals(UUID.fromString("109aea14-c252-4a85-96e2-f5f4d5d088f4"), squirrel.getDatasetKey());
128    assertEquals("Syst. Nat. , 10th ed. vol. 1 p. 63", squirrel.getPublishedIn());
129
130    assertEquals("2010030", squirrel.getSourceId());
131
132
133    // TEST VERNACULAR
134    squirrel = ddt.getService().get(100000040, null);
135    assertNull(squirrel.getVernacularName());
136
137    squirrel = ddt.getService().get(100000040, Locale.GERMANY);
138    assertEquals("Kaukasischen Eichhörnchen", squirrel.getVernacularName());
139
140    // TEST non existing language VERNACULAR
141    squirrel = ddt.getService().get(100000040, Locale.CHINESE);
142    assertEquals("Caucasian Squirrel", squirrel.getVernacularName());
143
144    // TEST MULTIPLE IDENTIFIERS
145    squirrel = ddt.getService().get(100000007, Locale.GERMANY);
146    Set<Integer> ids = Sets.newHashSet(7, 102, 103, 104, 105);
147    for (Identifier id : squirrel.getIdentifiers()) {
148      assertEquals(squirrel.getKey(), id.getUsageKey());
149      switch (id.getKey()) {
150        case 7:
151          assertEquals("6905528", id.getIdentifier());
152          assertEquals(IdentifierType.SOURCE_ID, id.getType());
153          assertEquals("Bonjour mon capitain", id.getTitle());
154          break;
155        case 102:
156          assertEquals("http://www.catalogueoflife.org/details/species/id/6905528", id.getIdentifier());
157          assertEquals(IdentifierType.URL, id.getType());
158          assertNull(id.getTitle());
159          break;
160        case 103:
161          assertEquals("urn:lsid:catalogueoflife.org:taxon:df0a319c-29c1-102b-9a4a-00304854f820:col20120721",
162            id.getIdentifier());
163          assertEquals(IdentifierType.LSID, id.getType());
164          assertNull(id.getTitle());
165          break;
166        case 104:
167          assertEquals("doi:10.1038/6905528", id.getIdentifier());
168          assertEquals(IdentifierType.DOI, id.getType());
169          assertNull(id.getTitle());
170          break;
171        case 105:
172          assertEquals("http://www.itis.gov/servlet/SingleRpt/SingleRpt?search_topic=TSN&search_value=632417",
173            id.getIdentifier());
174          assertEquals(IdentifierType.URL, id.getType());
175          assertNull(id.getTitle());
176          break;
177        default:
178          fail("Unkown Identifier");
179      }
180      ids.remove(id.getKey());
181    }
182    assertTrue(ids.isEmpty());
183
184    // TEST SYNONYM
185    NameUsage syn = ddt.getService().get(100000027, Locale.FRENCH);
186    assertNotNull(syn);
187    assertTrue(syn.isSynonym());
188    assertEquals("Sciurus nadymensis", syn.getCanonicalName());
189    assertEquals("Sciurus nadymensis Serebrennikov, 1928", syn.getScientificName());
190    assertNull(syn.getVernacularName());
191    assertEquals(Rank.SPECIES, syn.getRank());
192    assertEquals((Integer) 100000024, syn.getParentKey());
193    assertEquals((Integer) 100000025, syn.getAcceptedKey());
194    assertEquals("Sciurus vulgaris Linnaeus, 1758", syn.getAccepted());
195    assertFalse(syn.isProParte());
196
197  }

228  @Test
229  public void testList() {
230    List<NameUsage> usages = ddt.getService().list(Locale.UK, null, null, null).getResults();
231    assertEquals(PagingConstants.DEFAULT_PARAM_LIMIT, usages.size());
232
233    // test paging
234    Pageable page = new PagingRequest(1l, 1);
235    usages = ddt.getService().list(Locale.UK, null, null, page).getResults();
236    assertEquals(1, usages.size());
237    NameUsage u1 = usages.get(0);
238
239    page = new PagingRequest(0l, 2);
240    usages = ddt.getService().list(Locale.UK, null, null, page).getResults();
241    assertEquals(2, usages.size());
242    assertEquals(u1, usages.get(1));
243
244    // test by source id
245    usages = ddt.getService().list(Locale.UK, null, "1", null).getResults();
246    assertEquals(1, usages.size());
247    assertEquals((Integer) 100000001, usages.get(0).getKey());
248
249    // test by checklist key
250    usages = ddt.getService().list(Locale.UK, UUID.fromString("d7dddbf4-2cf0-4f39-9b2a-bb099caae36c"), null, null)
251      .getResults();
252    assertEquals(2, usages.size());
253
254    // test combined
255    usages =
256      ddt.getService().list(Locale.UK, UUID.fromString("109aea14-c252-4a85-96e2-f5f4d5d088f4"), "1", null).getResults();
257    assertEquals(1, usages.size());
258    assertEquals((Integer) 100000001, usages.get(0).getKey());
