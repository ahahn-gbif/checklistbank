001package org.gbif.checklistbank.service.mybatis;
002
003import org.gbif.api.model.checklistbank.Reference;
004import org.gbif.api.model.common.paging.Pageable;
005import org.gbif.api.model.common.paging.PagingRequest;
006import org.gbif.api.service.checklistbank.ReferenceService;
007import org.gbif.checklistbank.service.mybatis.postgres.DatabaseDrivenChecklistBankTestRule;
008
009import java.util.List;
010
011import org.junit.Rule;
012import org.junit.Test;
013
014import static org.junit.Assert.assertEquals;
015import static org.junit.Assert.assertNull;
016
017public class ReferenceServiceMyBatisIT {
018
019  private final Integer USAGE_ID = 100000025;
020
021  @Rule
022  public DatabaseDrivenChecklistBankTestRule<ReferenceService> ddt =
023    new DatabaseDrivenChecklistBankTestRule<ReferenceService>(ReferenceService.class);
024
025  @Test
026  public void testGet() {
027    Reference ref = ddt.getService().get(17);
028    assertEquals("Watson, Jeremy (30 December 2007) \"Tufty's saviour to the rescue\". Scotland on Sunday. Edinburgh.",
029      ref.getCitation());
030    assertNull(ref.getAuthor());
031    assertNull(ref.getTitle());
032    assertNull(ref.getDate());
033    assertNull(ref.getType());
034    assertNull(ref.getLink());
035    assertNull(ref.getDoi());
036    assertNull(ref.getRemarks());
037    assertEquals(USAGE_ID, ref.getUsageKey());
038  }
039
040  @Test
041  public void testListByChecklistUsage() {
042    List<Reference> refs = ddt.getService().listByUsage(USAGE_ID, null).getResults();
043    assertEquals(20, refs.size());
044    assertEquals("\"A new era for Scotland's red squirrels?\" in Scottish Wildlife (November 2008) No. 66. Edinburgh.",
045      refs.get(0).getCitation());
046    assertEquals("\"Black squirrels set to dominate\". BBC News. 2009-01-20. Retrieved 2009-04-26.",
047      refs.get(1).getCitation());
048
049    // TEST PAGING
050    Pageable page = new PagingRequest(0, 1);
051    Reference r1 = ddt.getService().listByUsage(USAGE_ID, page).getResults().get(0);
052
053    page = new PagingRequest(1, 1);
054    Reference r2 = ddt.getService().listByUsage(USAGE_ID, page).getResults().get(0);
055    assertEquals(r1, refs.get(0));
056    assertEquals(r2, refs.get(1));
057  }
058}
