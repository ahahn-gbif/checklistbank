001package org.gbif.checklistbank.service.mybatis;
002
003import org.gbif.api.model.checklistbank.DatasetMetrics;
004import org.gbif.api.service.checklistbank.DatasetMetricsService;
005import org.gbif.api.vocabulary.Kingdom;
006import org.gbif.api.vocabulary.Language;
007import org.gbif.api.vocabulary.Rank;
008import org.gbif.checklistbank.service.mybatis.postgres.DatabaseDrivenChecklistBankTestRule;
009
010import java.util.List;
011import java.util.UUID;
012
013import org.junit.Rule;
014import org.junit.Test;
015
016import static org.junit.Assert.assertEquals;
017
018public class DatasetMetricsServiceMyBatisIT {
019
020  private static final UUID CHECKLIST_KEY = UUID.fromString("109aea14-c252-4a85-96e2-f5f4d5d088f4");
021
022  @Rule
023  public DatabaseDrivenChecklistBankTestRule<DatasetMetricsService> ddt =
024    new DatabaseDrivenChecklistBankTestRule<DatasetMetricsService>(DatasetMetricsService.class);
025
026  @Test
027  public void testGet() {
028    DatasetMetrics d = ddt.getService().get(CHECKLIST_KEY);
029    assertEquals(CHECKLIST_KEY, d.getDatasetKey());
030    assertEquals(1000, d.getUsagesCount());
031    assertEquals(25, d.getColCoveragePct());
032    assertEquals(250, d.getColMatchingCount());
033    assertEquals(100, d.getCountByKingdom(Kingdom.ANIMALIA));
034    assertEquals(700, d.getCountByKingdom(Kingdom.PLANTAE));
035    assertEquals(0, d.getCountByKingdom(Kingdom.FUNGI));
036    assertEquals(120, d.getCountByRank(Rank.GENUS));
037    assertEquals(10, d.getCountByRank(Rank.PHYLUM));
038    assertEquals(4, d.getCountNamesByLanguage(Language.DANISH));
039    assertEquals(132, d.getCountNamesByLanguage(Language.GERMAN));
040  }
041
042  @Test
043  public void testList() {
044    List<DatasetMetrics> ds = ddt.getService().list(CHECKLIST_KEY);
045    assertEquals(3, ds.size());
046    for (DatasetMetrics d : ds) {
047      assertEquals(CHECKLIST_KEY, d.getDatasetKey());
048    }
049    assertEquals(1000, ds.get(0).getUsagesCount());
050    assertEquals(200, ds.get(1).getUsagesCount());
051    assertEquals(100, ds.get(2).getUsagesCount());
052  }
053
054}
