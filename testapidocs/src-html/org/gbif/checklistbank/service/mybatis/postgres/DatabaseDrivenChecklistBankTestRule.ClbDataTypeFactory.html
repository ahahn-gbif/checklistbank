014import com.google.common.collect.ImmutableMap;

018import org.dbunit.dataset.datatype.DataTypeException;
019import org.dbunit.ext.postgresql.PostgresqlDataTypeFactory;

021/**
022 * Provides a JUnit {@link org.junit.rules.TestRule} to allow database driven integration tests in ChecklistBank.
023 * You need to add this rule to each of your tests where you want to use it. One example on the usage:
024 * <pre>
025 * {@code
026 *
027 * @param <T> the class of the service you need for this test. This is initialized using Guice and backed by a database
028 *            as specified in the properties provided.
029 */
030public class DatabaseDrivenChecklistBankTestRule<T> extends DatabaseDrivenTestRule<T> {
031
032  private static final String DEFAULT_PROPERTY_FILE = "checklistbank.properties";
033  private static final String DEFAULT_DBUNIT_FILE = "squirrels-full.xml";
034
035  public static final Map<String, Object> DB_UNIT_CLB_PROPERTIES = new ImmutableMap.Builder<String, Object>()
036    .put(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new ClbDataTypeFactory())
037    .put("http://www.dbunit.org/features/caseSensitiveTableNames", true)
038    .put(DatabaseConfig.PROPERTY_ESCAPE_PATTERN, "\"?\"").build();
039
040  private static Properties buildDefaultProperties() {
041    try {
042      return PropertiesUtil.loadProperties(DEFAULT_PROPERTY_FILE);
043    } catch (IOException e) {
044      throw new RuntimeException(e);
045    }
046  }
047
048  /**
049   * @param serviceClass   the class for the service we want to wire up and test
050   * @param properties     the verbatim configuration properties
051   * @param dbUnitFileName the optional unqualified filename within the dbunit package to be used in setting up the db
052   */
053  public DatabaseDrivenChecklistBankTestRule(Properties properties, Class<T> serviceClass, String dbUnitFileName) {
054    super(new ChecklistBankServiceMyBatisModule(properties),
055      InternalChecklistBankServiceMyBatisModule.DATASOURCE_BINDING_NAME, serviceClass, dbUnitFileName,
056      DB_UNIT_CLB_PROPERTIES);
057  }
058
059  public DatabaseDrivenChecklistBankTestRule(Class<T> serviceClass) {
060    this(buildDefaultProperties(), serviceClass, DEFAULT_DBUNIT_FILE);
061  }
062
063  /**
064   * Creates a db driven test rule using the default checklistbank.properties and the squirrels dbunit file.
065   */
066  public DatabaseDrivenChecklistBankTestRule() {
067    this(null);
068  }
069
070  /**
071   * Extending the regular postgres data type factory with our custom clb enumerations.
072   */
073  public static class ClbDataTypeFactory extends PostgresqlDataTypeFactory{
074    private static Set<String> ENUM_NAMES = ImmutableSet.of("cites_appendix", "establishment_means", "identifier_type",
075      "kingdom", "life_stage", "name_part", "name_type", "nomenclatural_status", "occurrence_status", "origin_type",
076      "rank", "sex", "taxonomic_status", "threat_status", "type_designation_type", "type_status");
077
078    @Override
079    public boolean isEnumType(String sqlTypeName) {
080      if (ENUM_NAMES.contains(sqlTypeName)) {
081        return true;
082      }
083      return false;
084    }
085
086    @Override
087    public DataType createDataType(int sqlType, String sqlTypeName) throws DataTypeException {
088      if (sqlType == Types.OTHER && "hstore".equals(sqlTypeName)) {
089        return new HstoreType();
090      }
091      if (sqlType == Types.ARRAY) {
092        return new ArrayType(sqlTypeName.substring(1));
093      }
094      return super.createDataType(sqlType, sqlTypeName);
095    }
096  }
