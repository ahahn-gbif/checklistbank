001package org.gbif.checklistbank.service.mybatis;
002
003import org.gbif.api.model.checklistbank.TypeSpecimen;
004import org.gbif.api.model.common.paging.Pageable;
005import org.gbif.api.model.common.paging.PagingRequest;
006import org.gbif.api.service.checklistbank.TypeSpecimenService;
007import org.gbif.api.vocabulary.Rank;
008import org.gbif.api.vocabulary.TypeDesignationType;
009import org.gbif.api.vocabulary.TypeStatus;
010import org.gbif.checklistbank.service.mybatis.postgres.DatabaseDrivenChecklistBankTestRule;
011
012import java.text.ParseException;
013import java.util.List;
014import java.util.UUID;
015
016import org.junit.Rule;
017import org.junit.Test;
018
019import static org.junit.Assert.assertEquals;
020
021public class TypeSpecimenServiceMyBatisTestIT {
022
023  private final Integer USAGE_ID = 100000006;
024  private final UUID SQUIRREL_DATASET = UUID.fromString("109aea14-c252-4a85-96e2-f5f4d5d088f4");
025
026  @Rule
027  public DatabaseDrivenChecklistBankTestRule<TypeSpecimenService> ddt =
028    new DatabaseDrivenChecklistBankTestRule<TypeSpecimenService>(TypeSpecimenService.class);
029
030  @Test
031  public void testGet() throws ParseException {
032
033    TypeSpecimen typeSpecimen = ddt.getService().get(15);
034    assertEquals(USAGE_ID, typeSpecimen.getUsageKey());
035    assertEquals(SQUIRREL_DATASET, typeSpecimen.getDatasetKey());
036    assertEquals(TypeStatus.TYPE_SPECIES, typeSpecimen.getTypeStatus());
037    assertEquals("Raciurus pusillus", typeSpecimen.getScientificName());
038    assertEquals("Thomas", typeSpecimen.getTypeDesignatedBy());
039    assertEquals("Country File, BBC, 28. 89. 2008", typeSpecimen.getCitation());
040    assertEquals(Rank.SPECIES, typeSpecimen.getTaxonRank());
041    assertEquals("555", typeSpecimen.getOccurrenceId());
042    assertEquals("BGBM", typeSpecimen.getInstitutionCode());
043    assertEquals("Squirrels", typeSpecimen.getCollectionCode());
044    assertEquals("Z22", typeSpecimen.getCatalogNumber());
045    assertEquals("French Guiana, Cayenne", typeSpecimen.getLocality());
046    assertEquals("Forestry Commission - Red Squirrels", typeSpecimen.getSource());
047    assertEquals("Charles III", typeSpecimen.getRecordedBy());
048    assertEquals("Around summer 2000", typeSpecimen.getVerbatimEventDate());
049    assertEquals("label", typeSpecimen.getVerbatimLabel());
050    assertEquals("55 degrees", typeSpecimen.getVerbatimLongitude());
051    assertEquals("10 degrees", typeSpecimen.getVerbatimLatitude());
052    assertEquals(TypeDesignationType.ORIGINAL_DESIGNATION, typeSpecimen.getTypeDesignationType());
053  }
054
055  @Test
056  public void testListByChecklistUsage() {
057    List<TypeSpecimen> typeSpecimens = ddt.getService().listByUsage(USAGE_ID, null).getResults();
058    assertEquals(2, typeSpecimens.size());
059    assertEquals((Integer) 15, typeSpecimens.get(0).getKey());
060
061    // TEST PAGING
062    Pageable page = new PagingRequest(0, 1);
063    TypeSpecimen d1 = ddt.getService().listByUsage(USAGE_ID, page).getResults().get(0);
064
065    page = new PagingRequest(1, 1);
066    TypeSpecimen d2 = ddt.getService().listByUsage(USAGE_ID, page).getResults().get(0);
067    assertEquals(d1, typeSpecimens.get(0));
068    assertEquals(d2, typeSpecimens.get(1));
069  }
070
071}
