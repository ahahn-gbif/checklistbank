004package org.gbif.checklistbank.index;
005
006import org.gbif.api.service.registry.DatasetService;
007import org.gbif.checklistbank.index.guice.IndexingModulePrivate;
008import org.gbif.checklistbank.service.mybatis.guice.ChecklistBankServiceMyBatisModule;
009
010import java.util.Properties;
011
012import com.google.inject.AbstractModule;
013import com.google.inject.Scopes;
014
015
016/**
017 * Guice module that initializes the required classes and dependencies for the CLB indexer
018 * using a mocked dataset service.
019 */
020public class IndexingTestModule extends AbstractModule {
021
022  private final Properties properties;
023
024  public IndexingTestModule(Properties properties) {
025    this.properties = properties;
026  }
027
028  @Override
029  protected void configure() {
030    // Installs the MyBatis service layer
031    install(new ChecklistBankServiceMyBatisModule(properties));
032
033    // bind dataset service mock
034    bind(DatasetService.class).to(DatasetServiceMock.class).in(Scopes.SINGLETON);
035
036    // Installs private indexing module
037    install(new IndexingModulePrivate(properties));
038  }
039
040}
