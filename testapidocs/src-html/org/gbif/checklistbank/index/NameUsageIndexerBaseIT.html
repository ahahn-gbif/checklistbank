001package org.gbif.checklistbank.index;
002
003import com.google.inject.Guice;
004import com.google.inject.Injector;
005import org.apache.solr.client.solrj.embedded.EmbeddedSolrServer;
006import org.gbif.checklistbank.index.guice.EmbeddedSolrReference;
007import org.gbif.checklistbank.service.mybatis.postgres.DatabaseDrivenChecklistBankTestRule;
008import org.gbif.utils.file.properties.PropertiesUtil;
009import org.junit.BeforeClass;
010import org.junit.Rule;
011import org.slf4j.Logger;
012import org.slf4j.LoggerFactory;
013import org.xml.sax.SAXException;
014
015import javax.xml.parsers.ParserConfigurationException;
016import java.io.IOException;
017import java.util.Properties;
018
019/**
020 * Base class for integration tests using the squirrels test solr index.
021 * The class builds a fresh index before all tests run.
022 */
023public abstract class NameUsageIndexerBaseIT {
024  private static final String PROPERTY_FILE = "checklistbank.properties";
025  private static final String PROPERTY_DEFAULT_FILE = "checklistbank-indexer-default.properties";
026
027  private static final Logger LOG = LoggerFactory.getLogger(NameUsageIndexerBaseIT.class);
028  private static EmbeddedSolrReference solrRef;
029  private static NameUsageIndexer nameUsageIndexer;
030
031  @Rule
032  public DatabaseDrivenChecklistBankTestRule<?> squirrel = new DatabaseDrivenChecklistBankTestRule();
033
034  @BeforeClass
035  public static void setup() throws IOException, SAXException, ParserConfigurationException {
036    // Creates the injector, merging properties taken from default test indexing and checklistbank
037    Properties props = PropertiesUtil.loadProperties(PROPERTY_DEFAULT_FILE);
038    Properties props2 = PropertiesUtil.loadProperties(PROPERTY_FILE);
039    props.putAll(props2);
040    Injector injector = Guice.createInjector(new IndexingTestModule(props));
041    // Gets the indexer instance
042    solrRef = injector.getInstance(EmbeddedSolrReference.class);
043    nameUsageIndexer = injector.getInstance(NameUsageIndexer.class);
044    nameUsageIndexer.run();
045  }
046
047  protected static EmbeddedSolrServer solr() {
048    return solrRef.getSolr();
049  }
050
051  protected static NameUsageIndexer indexer() {
052    return nameUsageIndexer;
053  }
054
055}
