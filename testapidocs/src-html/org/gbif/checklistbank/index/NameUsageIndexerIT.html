001package org.gbif.checklistbank.index;
002
003import org.gbif.api.model.checklistbank.Description;
004import org.gbif.api.vocabulary.Rank;
005import org.gbif.checklistbank.index.model.NameUsageSolrSearchResult;
006
007import java.io.IOException;
008import java.util.List;
009
010import org.apache.solr.client.solrj.SolrQuery;
011import org.apache.solr.client.solrj.SolrServerException;
012import org.apache.solr.client.solrj.response.QueryResponse;
013import org.apache.solr.common.SolrDocumentList;
014import org.junit.Test;
015
016import static org.junit.Assert.assertEquals;
017import static org.junit.Assert.assertFalse;
018import static org.junit.Assert.assertNull;
019
020/**
021 * Test the index generation.
022 */
023public class NameUsageIndexerIT extends NameUsageIndexerBaseIT {
024
025  @Test
026  public void testIndexBuild() throws IOException, SolrServerException, InterruptedException {
027    // test index and make sure we use the squirrels test data
028    // number of all records
029    SolrQuery query = new SolrQuery();
030    query.setQuery("*:*");
031    QueryResponse rsp = solr().query(query);
032    assertEquals(46l, rsp.getResults().getNumFound());
033
034    // vernacular name with umlaut
035    query = new SolrQuery();
036    query.setQuery("vernacular_name:\"Europäisches Eichhörnchen\"");
037    rsp = solr().query(query);
038    SolrDocumentList docs = rsp.getResults();
039    assertEquals(1, docs.size());
040
041    // single species
042    query = new SolrQuery();
043    query.setQuery("canonical_name:\"Sciurillus pusillus\"");
044    rsp = solr().query(query);
045
046    docs = rsp.getResults();
047    assertEquals(4, docs.size());
048    List<NameUsageSolrSearchResult> usages = rsp.getBeans(NameUsageSolrSearchResult.class);
049    NameUsageSolrSearchResult u1 = usages.get(0);
050    assertEquals((Integer) 100000007, u1.getKey());
051
052    // extinct
053    query = new SolrQuery();
054    query.setQuery("key:100000025");
055    rsp = solr().query(query);
056    docs = rsp.getResults();
057    assertEquals(1, docs.size());
058    usages = rsp.getBeans(NameUsageSolrSearchResult.class);
059    u1 = usages.get(0);
060    assertEquals((Integer) 100000025, u1.getKey());
061    assertFalse(u1.isSynonym());
062    assertFalse(u1.isExtinct());
063    assertFalse(u1.isMarine());
064    assertEquals(Rank.SPECIES, u1.getRank());
065    assertEquals((Integer) 100000024, u1.getSubgenusKey());
066    assertEquals((Integer) 100000011, u1.getGenusKey());
067    assertEquals((Integer) 100000005, u1.getFamilyKey());
068    assertEquals((Integer) 100000004, u1.getOrderKey());
069    assertEquals((Integer) 100000003, u1.getClassKey());
070    assertEquals((Integer) 100000002, u1.getPhylumKey());
071    assertEquals((Integer) 100000001, u1.getKingdomKey());
072    assertEquals((Integer) 100000024, u1.getParentKey());
073    assertNull(u1.getAcceptedKey());
074    assertEquals(6, u1.getDescriptions().size());
075    assertEquals("2010030", u1.getSourceId());
076
077    query = new SolrQuery();
078    query.setQuery("extinct:false");
079    rsp = solr().query(query);
080    docs = rsp.getResults();
081    assertEquals(1, docs.size());
082    usages = rsp.getBeans(NameUsageSolrSearchResult.class);
083    u1 = usages.get(0);
084    assertEquals((Integer) 100000025, u1.getKey());
085
086    // threat status
087    query = new SolrQuery();
088    query.setQuery("threat_status_key:*");
089    rsp = solr().query(query);
090    docs = rsp.getResults();
091    assertEquals(1, docs.size());
092    usages = rsp.getBeans(NameUsageSolrSearchResult.class);
093    u1 = usages.get(0);
094    assertEquals((Integer) 100000007, u1.getKey());
095
096    // nub key
097    query = new SolrQuery();
098    query.setQuery("nub_key:1");
099    rsp = solr().query(query);
100    docs = rsp.getResults();
101    assertEquals(1, docs.size());
102    usages = rsp.getBeans(NameUsageSolrSearchResult.class);
103    u1 = usages.get(0);
104    assertEquals((Integer) 100000001, u1.getKey());
105
106    query = new SolrQuery();
107    query.setQuery("nub_key:57453456");
108    rsp = solr().query(query);
109    docs = rsp.getResults();
110    assertEquals(0, docs.size());
111
112    query = new SolrQuery();
113    query.setQuery("key:100000004");
114    rsp = solr().query(query);
115    docs = rsp.getResults();
116    assertEquals(1, docs.size());
117    usages = rsp.getBeans(NameUsageSolrSearchResult.class);
118    u1 = usages.get(0);
119    for (Description d : u1.getDescriptions()) {
120      // make sure we don't have html tags in the descriptions
121      assertFalse(d.getDescription().contains("<"));
122    }
123  }
124
125
126}
