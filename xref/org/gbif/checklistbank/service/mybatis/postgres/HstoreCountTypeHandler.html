50    private Map<KEY, Integer> fromString(String hstring) {
51      HashMap<KEY, Integer> typedMap = Maps.newHashMap();
52      if (!Strings.isNullOrEmpty(hstring)) {
53        Map<String, String> rawMap = HStoreConverter.fromString(hstring);
54        for (Map.Entry<String, String> entry : rawMap.entrySet()) {
55          try {
56            typedMap.put(toKey(entry.getKey()), Integer.parseInt(entry.getValue()));
57          } catch (IllegalArgumentException e) {
58            // ignore this entry
59          }
60        }
61      }
62      return sortMap(typedMap);
63    }

65    /**

67     * By default sorts the map according to the count values in descending order.
68     */
69    protected Map<KEY, Integer> sortMap(HashMap<KEY, Integer> map) {
70      return sortMapByValuesDesc(map);
71    }

73    @VisibleForTesting
74    protected static <KEY extends Comparable> Map<KEY, Integer> sortMapByValuesDesc(HashMap<KEY, Integer> map) {
75      final Ordering<KEY> reverseValuesAndNaturalKeysOrdering =
76          Ordering.natural().reverse().nullsLast().onResultOf(Functions.forMap(map, null)) // natural for values
77              .compound(Ordering.natural()); // secondary - natural ordering of keys
78      return ImmutableSortedMap.copyOf(map, reverseValuesAndNaturalKeysOrdering);
79    }
