11  /**
12   * A facade to a description stored in a serialized format in solr that allows to support solr highlighting on the
13   * description field of the complex Description class.
14   */
15  public class HighlightableDescription implements HighlightableList {
16  
17    private final List<Description> descriptions;
18  
19    public HighlightableDescription(List<Description> descriptions) {
20      this.descriptions = descriptions;
21    }
22  
23    @Override
24    public List<String> getValues() {
25      List<String> values = Lists.newArrayList();
26      for (Description d : descriptions) {
27        values.add(NameUsageDocConverter.serializeDescription(d));
28      }
29      return values;
30    }
31  
32    @Override
33    public void replaceValue(int index, String newValue) {
34      // watch out, we get a marked up serialized string here!
35      Description d = NameUsageDocConverter.deserializeDescription(newValue);
36      descriptions.get(index).setDescription(d.getDescription());
37    }
