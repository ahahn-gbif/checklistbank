48  /**
49   * Checklist Bank multithreaded name usage solr indexer.
50   * This class creates a pool of configurable <i>threads</i> that concurrently execute a number of jobs
51   * each processing a configurable number of name usages (<i>batchSize</i>)
52   * using a configurable number of concurrent lucene <i>writers</i>.
53   * The indexer makes direct use of the mybatis layer and requires a checklist bank datasource to be configured.

98    private class CountReporter extends Thread {
99  
100     /**
101      * Timer to measure the total time of execution
102      */
103     private StopWatch stopWatch = new StopWatch();
104     private final long total;
105     private final DecimalFormat twoDForm = new DecimalFormat("#.##");
106 
107     CountReporter(long total) {
108       this.total = total;
109     }
110 
111     @Override
112     public void run() {
113       stopWatch.start();
114       LOG.info("Started reporting thread with expected {} total records.", total);
115       LOG.info("Logging every {} seconds. Use logInterval property to change interval.", logInterval);
116       boolean interrupted = false;
117       while (!interrupted) {
118         log();
119         try {
120           Thread.sleep(logInterval * 1000);
121         } catch (InterruptedException e) {
122           LOG.info("Reporter thread interrupted, exiting");
123           interrupted = true;
124         }
125       }
126     }
127 
128     /**
129      * Log total progress every minute.
130      */
131     private void log() {
132       long cnt = counter.get();
133       double percCompleted = (double) cnt / (double) total;
134       double percRemaining = 1d - percCompleted;
135       long timeRemaining = (long) (stopWatch.getTime() * (percRemaining / percCompleted));
136       LOG.info("{} documents ({}%) added in {}",
137         new Object[] {cnt, twoDForm.format(percCompleted * 100), stopWatch.toString()});
138       LOG.info("Expected remaining time to finish {}", DurationFormatUtils.formatDurationHMS(timeRemaining));
139     }

182   private static Properties loadProperties(String propertiesFile) throws IOException {
183     InputStream inputStream = null;
184     Properties tempProperties;
185     try {
186       inputStream = Files.newInputStreamSupplier(new File(propertiesFile)).getInput();
187       tempProperties = new Properties();
188       tempProperties.load(inputStream);

195   private void setupServers() {
196     writers = new EmbeddedSolrServer[numWriters];
197     if (numWriters == 1) {
198       // use main server
199       writers[0] = solrRef.getSolr();
200     } else {
201       // create others
202       LOG.debug("Setting up {} embedded solr servers ...", numWriters);
203       for (int idx = 0; idx < numWriters; idx++) {
204         writers[idx] = setupSolr(getWriterHome(idx));
205       }
206     }

209   private void mergeIndices() throws IOException, SolrServerException {
210     if (numWriters == 1) {
211       LOG.info("Optimizing single solr index ...");
212       solrRef.getSolr().optimize();
213 
214     } else {
215       File solrHome = solrRef.getSolr().getCoreContainer().getConfigFile().getParentFile();
216       // shutdown solr before we can merge into its index
217       solrRef.getSolr().getCoreContainer().shutdown();
218       File luceneDir = getLuceneDir(solrHome);
219       LOG.debug("Opening main lucene index at {}", luceneDir);
220       FSDirectory mainDir = FSDirectory.open(luceneDir);
221       IndexWriterConfig cfg = new IndexWriterConfig(VERSION, new StandardAnalyzer(VERSION));
222       IndexWriter fsWriter = new IndexWriter(mainDir, cfg);
223 
224       LOG.info("Start merging of {} solr indices", jobCounter);
225       Directory[] parts = new Directory[jobCounter];
226       for (int idx = 0; idx < jobCounter; idx++) {
227         File threadDir = getLuceneDir(getWriterHome(idx));
228         LOG.info("Add lucene dir {} for merging", threadDir);
229         parts[idx] = FSDirectory.open(threadDir);
230       }
231       fsWriter.addIndexes(parts);
232       fsWriter.close();
233       mainDir.close();
234       LOG.info("Lucene dirs merged! Startup main solr again");
235 
236       //startup solr again, keeping it in the same singleton wrapper that is accessible to the other tests
237       solrRef.setSolr(setupSolr(solrHome));
238     }
239   }

250    * Setup an embedded solr only for with a given solr home.
251    * Creates a checklistbank solr index schema, solr.xml and all other config files needed.
252    *
253    * @return the created server
254    */
255   private EmbeddedSolrServer setupSolr(File solrHome) {
256 
257     try {
258       // copy solr resource files
259       ResourcesUtil.copy(solrHome, "solr/", false, "solr.xml");
260       // copy default configurations
261       File conf = new File(solrHome, "conf");
262       ResourcesUtil.copy(conf, "solr/default/", false, "synonyms.txt", "protwords.txt", "stopwords.txt");
263       // copy specific configurations, overwriting above defaults
264       ResourcesUtil.copy(conf, "solr/conf/", false, "schema.xml", "solrconfig.xml");
265 
266       // create container
267       CoreContainer coreContainer = CoreContainer.createAndLoad(solrHome.getAbsolutePath(), new File(solrHome, "solr.xml"));
268       EmbeddedSolrServer solrServer = new EmbeddedSolrServer(coreContainer, "");
269 
270       LOG.info("Created embedded solr server with solr dir {}", solrHome.getAbsolutePath());
271 
272       // test solr
273       SolrPingResponse solrPingResponse = solrServer.ping();
274       LOG.info("Solr server configured at {}, ping response in {}", solrHome.getAbsolutePath(),
275         solrPingResponse.getQTime());
276 
277       return solrServer;
278 
279     } catch (Exception e) {
280       throw new IllegalStateException("Solr unavailable", e);
281     }

295   /**
296    * Creates a list of NameUsageIndexingJob by loading all usage ids and splitting up the jobs between those ids.
297    *
298    * @return a {@link List} of {@link NameUsageIndexingJob}.
299    */
300   @Override
301   protected Callable<Integer> newJob() {
302     if (allIds == null) {
303       init();
304     }
305 
306     // any new job to be created?
307     if (allIds.size() <= jobCounter * batchSize) {
308       LOG.info("No more jobs to create. Created {} jobs in total each processing {} records.", jobCounter, batchSize);
309       return null;
310     }
311 
312     // produce new job with a new slice
313     final int startKey = allIds.get(jobCounter * batchSize);
314     int endIdx = (jobCounter + 1) * batchSize - 1;
315     final int endKey = endIdx > allIds.size() ? allIds.get(allIds.size() - 1) : allIds.get(endIdx);
316     jobCounter++;
317 
318     // round robin on configured solr servers?
319     SolrServer solr = writers[jobCounter % numWriters];
320 
321     return new NameUsageIndexingJob(solr, nameUsageService, startKey, endKey, solrDocumentConverter,
322       vernacularNameService, descriptionService, distributionService, speciesProfileService);
323   }

325   private void init() {
326     StopWatch stopWatch = new StopWatch();
327 
328     LOG.debug("Start retrieving all usage ids ...");
329     stopWatch.start();
330     allIds = nameUsageService.listAll();
331     LOG.info("Retrieved all {} usage ids in {}", allIds.size(), stopWatch.toString());
332     stopWatch.reset();
333     stopWatch.start();
334     Collections.sort(allIds);
335     LOG.info("Sorted all {} usage ids in {}", allIds.size(), stopWatch.toString());
336     LOG.info("{} full jobs each processing {} records to be created.", allIds.size() / batchSize, batchSize);
337 
338     // create solr servers if multiple writers are configured
339     setupServers();
340 
341 
342     // start global reporter
343     reporterThread = new CountReporter(allIds.size());
344     reporterThread.start();
345   }
