24  /**
25   * Experimental export of the nub or any other dataset as README and json files in a hierarchical filestructure that
26   * can be browsed in github nicely.
27   */
28  public class NubGitExporter {
29    private static final Logger LOG = LoggerFactory.getLogger(NubGitExporter.class);
30    private static final Charset UTF8 = Charset.forName("UTF8");
31  
32    private int PAGE_SIZE = 10000;
33    private final String ROOTDIR = "/Users/mdoering/dev/backbone/life";
34    private final Joiner pathJoiner = Joiner.on(File.separator).skipNulls();
35    private final String readmeTempl;
36    private final UsageService usageService;
37    private ObjectMapper mapper = new ObjectMapper();
38  
39    @Inject
40    public NubGitExporter(UsageService usageService) throws IOException {
41      this.usageService = usageService;
42      readmeTempl = Resources.toString(Resources.getResource("exporter_readme.txt"), UTF8);
43      mapper.enable(SerializationConfig.Feature.INDENT_OUTPUT);
44    }
45  
46    public void run() {
47      final int max = usageService.maxUsageKey(Constants.NUB_DATASET_KEY) + 1;
48      int start = 0;
49      LOG.info("Retrieved max backbone usage key {}", max);
50      while (start < max) {
51        int end = start + PAGE_SIZE - 1;
52        List<NameUsage> usages = usageService.listRange(start, end);
53        LOG.info("Retrieved {} usages from {}-{}", usages.size(), start, end);
54        for (NameUsage u : usages) {
55          dumpUsage(u);
56        }
57        start += PAGE_SIZE;
58      }
59    }
60  
61    private void dumpUsage(NameUsage u) {
62      try {
63        StringBuilder name = new StringBuilder();
64        if (u.isSynonym()) {
65          name.append(" Syn. ");
66        }
67  //      name.append(u.getRank().ordinal());
68  //      name.append(u.getRank());
69  //      name.append(" ");
70        name.append(u.getCanonicalName());
71  
72        File uDir = new File(pathJoiner.join(ROOTDIR, pathJoiner.join(u.getHigherClassificationMap().values()), name.toString()));
73        File readme = new File(uDir, "README.md");
74        File json = new File(uDir, "data.json");
75        // create folders if missing
76        Files.createParentDirs(readme);
77        // write readme
78        Files.write(String.format(readmeTempl, u.getScientificName(), u.getRank(), u.getTaxonomicStatus(), u.getAccordingTo(), u.getPublishedIn(), u.getBasionym(), u.getRemarks()), readme, UTF8);
79        // write json
80        mapper.writeValue(json, u);
81  
82      } catch (IOException e) {
83        LOG.error("Failed to write usage {}: {}", u.getKey(), u.getScientificName(), e);
84      }
85    }
86  
87    public static void main(String[] args) throws IOException {
88      Injector inj = Guice.createInjector(new ExporterModule(PropertiesUtil.loadProperties("nub.properties")));
89      NubGitExporter exp = inj.getInstance(NubGitExporter.class);
90      exp.run();
91    }
