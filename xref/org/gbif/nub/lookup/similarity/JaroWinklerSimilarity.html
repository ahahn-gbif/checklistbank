20  import java.util.Arrays;
21  
22  /**
23   * Jaro-Winkler distance is a particularly good choice when comparing short strings of human entered data, such as names.
24   * This is due to it’s relative robustness against letter transpositions and it’s weighting of similarity toward the
25   * beginning of the string.
26   *
27   * @See <a href="http://en.wikipedia.org/wiki/Jaro%E2%80%93Winkler_distance">Jaro-Winkler distance on Wikipedia</a>.
28   *
29   * It is a variant of the Jaro distance metric which is made up of the average of three sub-calculations:
30   * <ul>
31   *   <li>The ratio of matching characters to the length of the first string</li>
32   *   <li>The ratio of matching characters to the length of the second string</li>
33   *   <li>The ratio of non-transpositions to the number matching of characters</li>
34   * </ul>
35   *
36   */
37  public class JaroWinklerSimilarity implements StringSimilarity {
38  
39    private final String s1;
40    private final String s2;
41  
42    private float threshold = 0.7f;
43  
44    public JaroWinklerSimilarity(String str1, String str2) {
45      this.s1 = str1;
46      this.s2 = str2;
47    }
48  
49    private int[] matches(String s1, String s2) {
50      String max, min;
51      if (s1.length() > s2.length()) {
52        max = s1;
53        min = s2;
54      } else {
55        max = s2;
56        min = s1;
57      }
58      int range = Math.max(max.length() / 2 - 1, 0);
59      int[] matchIndexes = new int[min.length()];
60      Arrays.fill(matchIndexes, -1);
61      boolean[] matchFlags = new boolean[max.length()];
62      int matches = 0;
63      for (int mi = 0; mi < min.length(); mi++) {
64        char c1 = min.charAt(mi);
65        for (int xi = Math.max(mi - range, 0), xn = Math.min(mi + range + 1, max.length()); xi < xn; xi++) {
66          if (!matchFlags[xi] && c1 == max.charAt(xi)) {
67            matchIndexes[mi] = xi;
68            matchFlags[xi] = true;
69            matches++;
70            break;
71          }
72        }
73      }
74      char[] ms1 = new char[matches];
75      char[] ms2 = new char[matches];
76      for (int i = 0, si = 0; i < min.length(); i++) {
77        if (matchIndexes[i] != -1) {
78          ms1[si] = min.charAt(i);
79          si++;
80        }
81      }
82      for (int i = 0, si = 0; i < max.length(); i++) {
83        if (matchFlags[i]) {
84          ms2[si] = max.charAt(i);
85          si++;
86        }
87      }
88      int transpositions = 0;
89      for (int mi = 0; mi < ms1.length; mi++) {
90        if (ms1[mi] != ms2[mi]) {
91          transpositions++;
92        }
93      }
94      int prefix = 0;
95      for (int mi = 0; mi < min.length(); mi++) {
96        if (s1.charAt(mi) == s2.charAt(mi)) {
97          prefix++;
98        } else {
99          break;
100       }
101     }
102     return new int[] {matches, transpositions / 2, prefix, max.length()};
103   }
104 
105   @Override
106   public double getSimilarity() {
107     int[] mtp = matches(s1, s2);
108     float m = (float) mtp[0];
109     if (m == 0) {
110       return 0;
111     }
112     float j = ((m / s1.length() + m / s2.length() + (m - mtp[1]) / m)) / 3;
113     float jw = j < getThreshold() ? j : j + Math.min(0.1f, 1f / mtp[3]) * mtp[2] * (1 - j);
114 
115     //(100d - (100d-s) / Math.pow(Math.max(1,l-8), 0.2))
116 
117     return 100d * jw;
118   }

120   /**
121    * Sets the threshold used to determine when Winkler bonus should be used.
122    * Set to a negative value to get the Jaro distance.
123    *
124    * @param threshold the new value of the threshold
125    */
126   public void setThreshold(float threshold) {
127     this.threshold = threshold;
128   }

130   /**
131    * Returns the current value of the threshold used for adding the Winkler bonus.
132    * The default value is 0.7.
133    *
134    * @return the current value of the threshold
135    */
136   public float getThreshold() {
137     return threshold;
138   }
