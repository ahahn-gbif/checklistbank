31  	public final static int getSimilarity(final String s1, final String s2, final int pBlockLimit) {
32  		if (s1.equals(s2))
33  			return 0;
34  		else if (s1.isEmpty() || s2.isEmpty())
35  			return Math.max(s1.length(), s2.length());
36  		else if (s1.length() == 1 && s2.length() == 1) return 1;
37  
38  		//final String temp1;
39  		//final String temp2;
40  
41  		final char[] t1;
42  		final char[] t2;
43  
44  		{
45  			StringBuilder sb1 = new StringBuilder(s1);
46  			StringBuilder sb2 = new StringBuilder(s2);
47  
48  			// these hold the index of the last character.
49  			int l1 = sb1.length()-1;
50  			int l2 = sb2.length()-1;
51  
52  			while (l1>=0 && l2>=0 && sb1.charAt(0) == sb2.charAt(0)) {
53  				sb1.deleteCharAt(0);
54  				sb2.deleteCharAt(0);
55  				l1--;
56  				l2--;
57  			}
58  
59  			while (l1>=0 && l2>=0 && sb1.charAt(l1) == sb2.charAt(l2)) {
60  				sb1.deleteCharAt(l1--);
61  				sb2.deleteCharAt(l2--);
62  			}
63  
64  			l1++;
65  			l2++;
66  			if (l1 == 0 || l2 == 0)
67  				return Math.max(l1, l2);
68  			else if (l1 == 1 && l2 == 1) return 1;
69  
70  			t1 = sb1.toString().toCharArray();
71  			t2 = sb2.toString().toCharArray();
72  		}
73  
74  		final int temp1Len = t1.length;
75  		final int temp2Len = t2.length;
76  
77  		// using a 1-dimensional array with bit fiddling to get to the elements
78  		// saves about 12-15% off the running time.
79  		// so, we replace matrix[a][b] with matrix[((a)<<6)|(b)]
80  		// this limits us to 64-character words, which should be plenty. I hope.
81  
82  		final int[] matrix;
83  		synchronized (mutex) {
84  			if (a_matrix == null) {
85  				matrix = new int[64 * 64];
86  			}
87  			else {
88  				matrix = a_matrix;
89  				a_matrix = null;
90  				Arrays.fill(matrix, 0);
91  			}
92  		}
93  
94  		for (int i = 0; i <= temp1Len; i++) {
95  			matrix[i << 6] = i;
96  		}
97  
98  		for (int i = 0; i <= temp2Len; i++) {
99  			matrix[i] = i;
100 		}
101 
102 		for (int i = 1; i <= temp1Len; i++) {
103 			matrix[i << 6] = i;
104 			for (int j = 1; j <= temp2Len; j++) {
105 				int cost;
106 				if (t1[i - 1] == t2[j - 1]) {
107 					cost = 0;
108 				}
109 				else {
110 					cost = 1;
111 				}
112 
113 				int temp_block_length = Math.max(//
114 						Math.min(temp1Len / 2, //
115 								Math.min(temp2Len / 2, //
116 										Math.max(pBlockLimit, 1))), //
117 						1);
118 
119 				while (temp_block_length >= 1) {
120 
121 					final int sub1 = i - ((temp_block_length * 2) - 1);
122 					final int sub2 = j - (temp_block_length - 1);
123 					final int sub3 = i - (temp_block_length - 1);
124 					final int sub4 = j - ((temp_block_length * 2) - 1);
125 
126 					if (i >= (temp_block_length * 2) && j >= (temp_block_length * 2)
127 							&& substreq(t1, sub1, t2, sub2, temp_block_length)
128 							&& substreq(t1, sub3, t2, sub4, temp_block_length)) {
129 
130 						final int ins = matrix[((i) << 6) | (j - 1)] + 1;
131 						final int del = matrix[((i - 1) << 6) | j] + 1;
132 						final int tran = matrix[((i - (temp_block_length * 2)) << 6) | (j - (temp_block_length * 2))]
133 								+ cost + (temp_block_length - 1);
134 						matrix[(i << 6) | j] = minimum(ins, del, tran);
135 
136 						temp_block_length = 0;
137 
138 					}
139 					else if (temp_block_length == 1) {
140 
141 						final int del = matrix[((i - 1) << 6) | j] + 1;
142 						final int ins = matrix[(i << 6) | (j - 1)] + 1;
143 						final int sub = matrix[((i - 1) << 6) | (j - 1)] + cost;
144 
145 						matrix[(i << 6) | j] = minimum(ins, del, sub);
146 					}
147 					temp_block_length--;
148 				}
149 
150 			}
151 		}
152 
153 		int ret = matrix[(temp1Len << 6) | temp2Len];
154 
155 		a_matrix = matrix; // does not need to be synchronized, it is atomic
156 
157 		return ret;
158 
159 	}

163 	private final static int minimum(int d, int i, int s) {
164 		return d < i ? (d < s ? d : s) : (i < s ? i : s);
165 	}

167 	private final static boolean substreq(char[] src1, int start1, char[] src2, int start2, int length) {
168 		int at1 = start1 >= 0 ? start1 - 1 : src1.length + start1;
169 		int at2 = start2 >= 0 ? start2 - 1 : src2.length + start2;
170 		while (length-- > 0) {
171 			if (src1[at1++] != src2[at2++]) return false;
172 		}
173 		return true;
174 	}
