104   private NameUsageMatch matchInternal(String scientificName, @Nullable Rank rank, @Nullable LinneanClassification classification, boolean strict, boolean verbose) {
105 
106     ParsedName pn = null;
107     if (classification == null) {
108       classification = new NameUsageMatch();
109     }
110     try {
111       // use name parser to make the name a canonical one
112       pn = parser.parse(scientificName);
113       interpretGenus(pn, classification.getGenus());
114       scientificName = pn.canonicalName();
115 
116     } catch (UnparsableException e) {
117       // hybrid names, virus names & blacklisted ones - dont provide any parsed name
118       LOG.debug("Unparsable [{}] name [{}]", e.type, scientificName);
119     }
120 
121 
122     NameUsageMatch match1 = match(scientificName, rank, classification, true, MIN_CONFIDENCE, verbose);
123     if (isMatch(match1) || strict) {
124       return match1;
125     }
126 
127 
128     // try to MATCH TO HIGHER RANKS if we can
129     // include species or genus only matches ?
130     NameUsageMatch match;
131     if (pn != null && pn.getGenusOrAbove() != null) {
132       if (pn.getSpecificEpithet() != null) {
133         if (pn.getInfraSpecificEpithet() != null) {
134           // try with species
135           String species = pn.canonicalSpeciesName();
136           match = match(species, Rank.SPECIES, classification, true, MIN_CONFIDENCE_FOR_HIGHER_MATCHES, false);
137           if (isMatch(match)) {
138             return higherMatch(match, match1);
139           }
140         }
141 
142         // try with genus
143         // we're not sure if this is really a genus, so dont set the rank
144         // we get non species names sometimes like "Chaetognatha eyecount" that refer to a phylum called
145         // "Chaetognatha"
146         match = match(pn.getGenusOrAbove(), null, classification, false, MIN_CONFIDENCE_FOR_HIGHER_MATCHES, false);
147         if (isMatch(match)) {
148           return higherMatch(match, match1);
149         }
150       }
151 
152     } else if (!Strings.isNullOrEmpty(classification.getGenus())) {
153       // no parsed name, try genus if given
154       match = match(classification.getGenus(), Rank.GENUS, classification, false, MIN_CONFIDENCE_FOR_HIGHER_MATCHES, false);
155       if (isMatch(match)) {
156         return higherMatch(match, match1);
157       }
158     }
159 
160     // last resort - try higher ranks above genus
161     for (Rank qr : HIGHER_QUERY_RANK) {
162       String name = ClassificationUtils.getHigherRank(classification, qr);
163       if (!StringUtils.isEmpty(name)) {
164         match = match(name, qr, classification, false, MIN_CONFIDENCE_FOR_HIGHER_MATCHES, false);
165         if (isMatch(match)) {
166           return higherMatch(match, match1);
167         }
168       }
169     }
170 
171     // if finally we cant find anything, return empty match object - but not null!
172     LOG.debug("No match for name {}", scientificName);
173     return noMatch(100, match1.getNote(), verbose ? match1.getAlternatives() : null);
174   }

177    * Expands abbreviated genus names with the full genus if provided in the separate classification.
178    * @param pn
179    * @param genus
180    */
181   @VisibleForTesting
182   protected static void interpretGenus(ParsedName pn, String genus) {
183         // test if name has an abbreviated genus
184     if (pn != null && !Strings.isNullOrEmpty(genus) && pn.getGenusOrAbove() != null && genus.length() > 1) {
185       if (pn.getGenusOrAbove().length() == 2
186         && pn.getGenusOrAbove().charAt(1) == '.'
187         && pn.getGenusOrAbove().charAt(0) == genus.charAt(0)
188         || pn.getGenusOrAbove().length() == 1 && pn.getGenusOrAbove().charAt(0) == genus.charAt(0)) {
189         pn.setGenusOrAbove(genus);
190       }
191     }
192   }

194   /**
195    * Use our custom similarity algorithm and compare the higher classifications to select the best match
196    * @return the best match, might contain no usageKey
197    */
198   @VisibleForTesting
199   protected NameUsageMatch match(String canonicalName, Rank rank, LinneanClassification lc, boolean fuzzySearch, int minConfidence, boolean verbose){
200     if (Strings.isNullOrEmpty(canonicalName)) {
201       throw new IllegalArgumentException("A scientific name is required for matching");
202     }
203 
204     // first try our manual hackmap
205     if (hackMap.containsKey(canonicalName.toLowerCase())) {
206       return hackMap.get(canonicalName.toLowerCase());
207     }
208 
209     // do a lucene matching
210     List<NameUsageMatch> matches = nubIndex.matchByName(canonicalName, fuzzySearch, 50);
211     for (NameUsageMatch m : matches) {
212       // 0 - 100
213       final int nameSimilarity = nameSimilarity(canonicalName, m);
214       // -50 - 50
215       LinneanClassification mCl = externalResolver == null ? m : externalResolver.getClassification(m.getUsageKey());
216       final int classificationSimilarity = classificationSimilarity(lc, mCl);
217       // -10 - 5
218       final int rankSimilarity = rankSimilarity(rank, m.getRank());
219       // 0 - 1
220       final int boostAccepted = m.isSynonym() ? 0 : 1;
221       // preliminary total score, -5 - 20 distance to next best match coming below!
222       m.setConfidence(nameSimilarity + classificationSimilarity + rankSimilarity + boostAccepted);
223 
224       if (verbose) {
225         addNote(m, "Individual confidence: name="+nameSimilarity);
226         addNote(m, "classification="+classificationSimilarity);
227         addNote(m, "rank="+rankSimilarity);
228         addNote(m, "accepted="+boostAccepted);
229       }
230     }
231 
232     // order by confidence
233     Collections.sort(matches, CONFIDENCE_ORDER);
234     if (matches.size() > 0) {
235       // add -5 - 20 confidence based on distance to next best match
236       NameUsageMatch best = matches.get(0);
237       int bestConfidence = best.getConfidence();
238       int nextMatchDistance;
239 
240       if (matches.size() == 1) {
241         // boost results with a single match to pick from
242         nextMatchDistance = 20;
243 
244       } else {
245         // we have more than one match to chose from
246         int secondBestConfidence = matches.get(1).getConfidence();
247         if (bestConfidence == secondBestConfidence) {
248           // equally good matches, bummer!
249           return noMatch(99, "Multiple equal matches for " + canonicalName, verbose ? matches : null);
250         }
251         // boost up to 15 based on distance to next match
252         nextMatchDistance = Math.min(15, (bestConfidence - secondBestConfidence - 10) / 2);
253       }
254       best.setConfidence(bestConfidence + nextMatchDistance);
255       if (verbose) {
256         addNote(best, "nextMatch=" + nextMatchDistance);
257       }
258 
259       if (bestConfidence < minConfidence) {
260         return noMatch(99, "No match because of too little confidence", verbose ? matches : null);
261       }
262 
263       // finally normalize confidence values into the range of 0 to 100
264       best.setConfidence(normConfidence(best.getConfidence()));
265       if (verbose && matches.size() > 1) {
266         // the first match is best
267         matches.remove(0);
268         best.setAlternatives(matches);
269         for (NameUsageMatch alt : matches) {
270           alt.setConfidence(normConfidence(alt.getConfidence()));
271         }
272       }
273 
274       return best;
275     }
276 
277     return noMatch(100, null, null);
278   }

296   private int nameSimilarity(String canonicalName, NameUsageMatch m) {
297     // calculate name distance
298     int confidence;
299     if (canonicalName.equalsIgnoreCase(m.getCanonicalName())) {
300       // straight match
301       confidence = 100;
302       // binomial straight match? That is pretty trustworthy
303       if (canonicalName.contains(" ")) {
304         confidence += 20;
305       }
306 
307     } else {
308       // fuzzy - be careful!
309       StringSimilarity sim = new ModifiedJaroWinklerSimilarity(canonicalName, m.getCanonicalName());
310       // regular JW is only good for very high similarities
311       confidence = (int) sim.getSimilarity() - 5;
312       // binomial at least? That is slightly more trustworthy
313       if (m.getCanonicalName().contains(" ")) {
314         confidence += 5;
315       }
316 
317     }
318     return confidence;
319   }

321   @VisibleForTesting
322   protected int classificationSimilarity(LinneanClassification query, LinneanClassification reference) {
323     // kingdom is super important
324     int rate = compareHigherRank(Rank.KINGDOM, query, reference, 8, -10, -1);
325     // plant and animal kingdoms are better delimited than Chromista, Fungi, etc. , so punish those mismatches higher
326     if (rate == -10 && isInKingdoms(query, Kingdom.ANIMALIA, Kingdom.PLANTAE) && isInKingdoms(reference, Kingdom.ANIMALIA, Kingdom.PLANTAE)){
327       rate = -25;
328     }
329     // phylum to family
330     rate += compareHigherRank(Rank.PHYLUM, query, reference, 10, -10, -1);
331     rate += compareHigherRank(Rank.CLASS, query, reference, 15, -7, 0);
332     rate += compareHigherRank(Rank.ORDER, query, reference, 15, -5, 0);
333     rate += compareHigherRank(Rank.FAMILY, query, reference, 25, -3, 0);
334 
335     return minMax(-50, 50, rate);
336   }

338   /**
339    * Compares a single higher rank and returns the matching confidence supplied.
340    * @param rank the rank to be compared
341    * @param query the classification of the query
342    * @param ref the classification of the nub reference usage
343    * @param match confidence returned if the classifications match for the given rank
344    * @param mismatch confidence returned if the classifications do not match for the given rank
345    * @param missing confidence returned if one or both classifications have missing information for the given rank
346    * @return match, mismatch or missing confidence depending on match
347    */
348   private int compareHigherRank(Rank rank, LinneanClassification query, LinneanClassification ref, int match, int mismatch, int missing) {
349     if (!StringUtils.isBlank(query.getHigherRank(rank)) && !StringUtils.isBlank(ref.getHigherRank(rank))) {
350       String querySyn = synonyms.lookup(query.getHigherRank(rank), rank);
351       String refSyn = synonyms.lookup(ref.getHigherRank(rank), rank);
352       if (!StringUtils.isBlank(querySyn) && !StringUtils.isBlank(refSyn) && querySyn.equalsIgnoreCase(refSyn)){
353         return match;
354       } else {
355         return mismatch;
356       }
357     }
358     return missing;
359   }

361   private boolean isInKingdoms(LinneanClassification n, Kingdom ... kingdoms){
362     String syn = synonyms.lookup(n.getKingdom(), Rank.KINGDOM);
363     if (!Strings.isNullOrEmpty(syn)){
364       for (Kingdom kingdom : kingdoms){
365         if (syn.equalsIgnoreCase(kingdom.name())){
366           return true;
367         }
368       }
369     }
370     return false;
371   }

373   // rate ranks from -10 to +5, zero if nothing is know
374   private int rankSimilarity(Rank query, Rank ref) {
375     int similarity = 0;
376     if (ref != null) {
377         // rate ranks lower that are not represented in the canonical, e.g. cultivars
378       if (Rank.CULTIVAR == ref || Rank.CULTIVAR_GROUP == ref){
379         similarity -= 7;
380       }else if (Rank.STRAIN == ref){
381         similarity -= 7;
382       }else if (Rank.INFORMAL == ref){
383         similarity -= 5;
384       }
385       if (ref.isUncomparable()){
386         // this also includes informal again
387         similarity -= 3;
388       }
389 
390       if(query != null){
391         // both ranks exist. Compare directly
392         if (query.equals(ref)) {
393           similarity += 10;
394         } else if (Rank.UNCOMPARABLE_RANKS.contains(query)) {
395           // uncomparable query ranks
396           similarity -= 5;
397         } else {
398           // rate lower the further away the ranks are
399           similarity -= 1 * Math.abs(ref.ordinal() - query.ordinal());
400         }
401       }
402 
403     } else if (query != null) {
404       // reference has no rank, rate it lower
405       similarity -= 1;
406     }
407     return minMax(-10, 5, similarity);
408   }

410   /**
411    * Produces a value between 0 and 100 by taking the not properly normalized confidence in the expected range of
412    * 0 to 175. This function is optimized to deal with acceptable matches being above 80, good matches above 90 and
413    * very good matches incl and above 100. The maximum of 100 is reached for an input of 175 or above.
414    */
415   @VisibleForTesting
416   protected static int normConfidence(int s) {
417     return minMax(0, 100, s <= 80 ? s : (int) Math.round(75.8 + (26d * (Math.log10((s - 70d) * 1.5) - 1))));
418   }

420   private static int minMax(int min, int max, int value) {
421     return Math.max(min, Math.min(max, value));
422   }

425   /**
426    * Ordering based on contact types with head of delegation and node managers coming first.
427    */
428   public static class ConfidenceOrder implements Comparator<NameUsageMatch> {
429 
430     @Override
431     public int compare(NameUsageMatch o1, NameUsageMatch o2) {
432       return ComparisonChain.start()
433         .compare(o1.getConfidence(), o2.getConfidence(), Ordering.natural().reverse().nullsLast())
434         .compare(o1.getScientificName(), o2.getScientificName(), Ordering.natural().nullsLast())
435         .result();
436     }
