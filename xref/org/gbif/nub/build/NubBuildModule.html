63      //TODO: make this configurable
64      File dir = new File("/tmp/nub-data");
65      if (dir.exists()) {
66        LOG.warn("Nub writer dir exists. All data will be removed");
67        FileUtils.deleteDirectory(dir);
68      }
69      // create new, empty dir
70      FileUtils.forceMkdir(dir);
